|| ಚಾಣಕ್ಯಾನಿ/ ಕೌಟಿಲೀಯಾನಿ ನೀತಿಸೂತ್ರಾಣಿ ||

ಪ್ರಥಮೋಽಧ್ಯಾಯಃ ||
ಸಾ ಶ್ರೀರ್ವೋಽವ್ಯಾತ್ |1|
ಸುಖಸ್ಯ ಮೂಲಂ ಧರ್ಮಃ |2|
ಧರ್ಮಸ್ಯ ಮೂಲಮರ್ಥಃ |3|
ಅರ್ಥಸ್ಯ ಮೂಲಂ ರಾಜ್ಯಂ |4|
ರಾಜ್ಯಮೂಲಮಿಂದ್ರಿಯಜಯಃ |5|
ಇಂದ್ರಿಯಜಯಸ್ಯ ಮೂಲಂ ವಿನಯಃ |6|
ವಿನಯಸ್ಯ ಮೂಲಂ ವೃದ್ಧೋಪಸೇವಾ |7|
ವೃದ್ಧಸೇವಾಯಾ ವಿಜ್ಞಾನಂ |8|
ವಿಜ್ಞಾನೇನಾತ್ಮಾನಂ ಸಂಪಾದಯೇತ್ |9|
ಸಂಪಾದಿತಾತ್ಮಾ ಜಿತಾತ್ಮಾ ಭವತಿ |10|
ಜಿತಾತ್ಮಾ ಸರ್ವಾರ್ಥೈಃ ಸಂಯುಜ್ಯತೇ |11|
ಸ್ವಾಮಿಸಂಪತ್ ಪ್ರಕೃತಿಸಂಪದಂ ಕರೋತಿ |12|
ಪ್ರಕೃತಿಸಂಪದಾ ಹ್ಯನಾಯಕಮಪಿ ರಾಜ್ಯಂ ನೀಯತೇ |13|
ಪ್ರಕೃತಿಕೋಪಃ ಸರ್ವಕೋಪೇಭ್ಯೋ ಗರೀಯಾನ್ |14|
ಅವಿನೀತಸ್ವಾಮಿಭಾವಾದಸ್ವಾಮಿಭಾವಃ ಶ್ರೇಯಾನ್ |15|
ಸಂಪಾದ್ಯಾತ್ಮಾನಮನ್ವಿಚ್ಛೇತ್ ಸಹಾಯಾನ್ |16|
ನಾಽಸಹಾಸ್ಯ ಮಂತ್ರನಿಶ್ಚಯಃ |18|
ನೈಕಂ ಚಕ್ರಂ ಪರಿಭ್ರಮತಿ |19|
ಸಹಾಯಃ ಸಮೋ ದುಃಖಸುಖಯೋಃ |17|
ಮಾನೀ ಪ್ರತಿಮಾನಿನಮಾತ್ಮದ್ವಿತೀಯಂ ಮಂತ್ರಿಣಮುತ್ಪಾದಯೇತ್ |20|
ಅವಿನೀತಂ ಸ್ನೇಹಮಾತ್ರೇಣ ನ ಮಂತ್ರೇ ಕುರ್ವೀತ |21|
ಶ್ರುತವಂತಮುಪಧಾಶುದ್ಧಂ ಮಂತ್ರಿಣಂ ಕುರ್ವೀತ |21|
ಮಂತ್ರಮೂಲಾಃ ಸರ್ವಾರಂಭಾಃ |22|
ಮಂತ್ರಸಂವರಣೇ ಕಾರ್ಯಸಿದ್ಧಿರ್ಭವತಿ |23|
ಮಂತ್ರನಿಃಸ್ರಾವಃ ಸರ್ವಂ ನಾಶಯತಿ |24|
ಪ್ರಮಾದಾದ್ ದ್ವಿಷತಾಂ ವಶಮುಪಯಾಸ್ಯತಿ |25|
ಸರ್ವದ್ವಾರೇಭ್ಯೋ ಮಂತ್ರೋ ರಕ್ಷಿತವ್ಯಃ |26|
ಮಂತ್ರಸಂಪದಾ ಹಿ ರಾಜ್ಯಂ ವಿವರ್ಧತೇ |27|
ಶ್ರೇಷ್ಠತಮಾಂ ಮಂತ್ರಗುಪಿತಮಾಹುಃ |28|
ಕಾರ್ಯಾಕಾರ್ಯ ಪ್ರದೀಪೋ ಮಂತ್ರಃ |29|
ಮಂತ್ರಚಕ್ಷುಷಾ ಪರಚ್ಛಿದ್ರಾಣ್ಯವಲೋಕಯಂತಿ ||31 |
ಮಂತ್ರಕಾಲೇ ನ ಮತ್ಸರಃ ಕರ್ತವ್ಯಃ |31|
ಅಕಾಮಬುದ್ಧಯೋ ಮಂತ್ರತತ್ತ್ವಾರ್ಥದರ್ಶಿನೋ ಮಂತ್ರಿಣಃ |34|
ಷಟ್ಕರ್ಣೋ ಮಂತ್ರಶ್ಛಿದ್ಯತೇ |32|
ತ್ರಯಾಣಾಮೈಕವಾಕ್ಯೇಽಸಂಪ್ರತ್ಯಯಃ |33|
ಆಪತ್ಸು ಸ್ನೇಹಯುಕ್ತಂ ಮಿತ್ರಂ |35|
ಮಿತ್ರಸಂಗ್ರಹಣೇ ಬಲಂ ಸಂಪದ್ಯತೇ |36|
ಬಲವಾನ್ ಅಲಬ್ಧಲಾಭೇ ಪ್ರಯತೇತ |37|
ಅಲಬ್ಧಲಾಭೋ ನಾಲಸಸ್ಯ |38|
ಅಲಸೇನ ಲಬ್ಧಮಪಿ ರಕ್ಷಿತುಂ ನ ಶಕ್ಯತೇ |39|
ನ ಚಾಲಸಸ್ಯಯುಕ್ತಸ್ಯ ರಕ್ಷಿತಂ ವಿವರ್ಧತೇ |40|
ನ ಭೃತ್ಯಾನ್ ಪೋಷಯತಿ |41|
ನ ತೀರ್ಥಂ ಪ್ರತಿಪಾದಯತಿ |42|
ಅಲಬ್ಧಲಾಭಾದಿಚತುಷ್ಟಯಂ ರಾಜತಂತ್ರಂ |43|
ತಚ್ಚ ರಾಜ್ಯತಂತ್ರಮಾಯತಂ ನೀತಿಶಾಸ್ತ್ರೇಷು |44|
ರಾಜ್ಯತಂತ್ರೇಷ್ವಾಯತ್ತೌ ಮಂತ್ರಾವಾಪೌ |45|
ಮಂತ್ರಂ ಸ್ವವಿಷಯೇ ಕೃತ್ಯೇಷ್ವಾಯತ್ತಂ |46|
ಆವಾಪೋ ಮಂಡಲೇ ಸನ್ನಿವಿಷ್ಟಃ |46|
ಸಂಧಿವಿಗ್ರಹಯೋರ್ಯೋನಿರ್ಮಂಡಲಂ |48|
ನೀತಿಶಾಸ್ತ್ರಾನುಗೋ ರಾಜಾ |49|
ಅನಂತರಪ್ರಕೃತಿಃ ಶತ್ರುಃ |50|
ಏಕಾಂತರಿತಂ ಮಿತ್ರಮಿಷ್ಯತೇ |51|
ಹೇತುತಃ ಶತ್ರುಮಿತ್ರೇ ಭವಿಷ್ಯತಃ |52|
ಹೀಯಮಾನೇನ ಸ ಸಂಧಿಂ ಕುರ್ವೀತ |53|
ತೇಜೋ ಹಿ ಸಂಧಾನಹೇತುಸ್ತದರ್ಥಿನಾಂ |54|
ನಾತಪ್ತಲೋಹಂ ಲೋಹೇನ ಸಂಧತ್ತೇ |55|
ಬಲವಾನ್ ಹೀನೇನ ವಿಗೃಹ್ಣೀಯಾತ್ |56|
ನ ಜ್ಯಾಯಸಾ ಸಮೇನ ವಾ |57|
ಹಸ್ತಿನಃ ಪಾದಯುದ್ಧಮಿವ ಬಲವದ್ವಿಗ್ರಹಃ ||58 |
ಆಮಪಾತ್ರಮಾಪೇನ ಸಹ ವಿನಶ್ಯತಿ |59|
ಅರಿಪ್ರಯತ್ನಮಭಿಸಮೀಕ್ಷ್ಯಾತ್ಮರಕ್ಷಯಾ ವಸೇತ್ |60|
ಸಂಧಾಯೈಕತೋ ವಾ ಯಾಯಾತ್ |58|
ಅಮಿತ್ರವಿರೋಧಾದಾತ್ಮರಕ್ಷಾಮಾವಸೇತ್ |59|
ಶಕ್ತಿಹೀನೋ ಬಲವಂತಮಾಶ್ರಯೇತ್ |61|
ದುರ್ಬಲಾಶ್ರಯೋ ದುಃಖಮಾವಹತಿ |62|
ಅಗ್ನಿವದ್ ರಾಜಾನಮಾಶ್ರಯೇತ್ |63|
ರಾಜ್ಞಾಃ ಪ್ರತಿಕೂಲಂ ನಾಚರೇತ್ |64|
ನೋದ್ಧತವೇಶಧರಃ ಸ್ಯಾತ್ |66|
ನ ದೇವಚರಿತಂ ಚರೇತ್ |65|
ದ್ವಯೋರಪೀಷ್ರ್ಯತೋದ್ರ್ವೈಧೀಭಾವಕುರ್ವೀತ |67|
ನ ವ್ಯಸನಪರಸ್ಯ ಕಾರ್ಯಾವಾಪ್ತಿಃ |68|
ಇಂದ್ರಿಯವಶವರ್ತಿನೋ ನಾಸ್ತಿ ಕಾರ್ಯಾವಾಪ್ತಿಃ |69|
ನಾಸ್ತಿ ಕಾರ್ಯಂ ದ್ಯುತಪ್ರವೃತ್ತಸ್ಯ |70|
ಮೃಗಯಾಪರಸ್ಯ ಧರ್ಮಾರ್ಥೌ ವಿನಶ್ಯತಃ |71|
ನ ಕಾಮಾಸಕ್ತಸ್ಯ ಕಾರ್ಯಾನುಷ್ಠಾನಂ |73|
ಅರ್ಥೇಷು ಪಾನವ್ಯಸನೀ ನ ಗಣ್ಯತೇ |72|
ಅರ್ಥದೂಷಕಂ ಶ್ರೀಃ ಪರಿತ್ಯಜತಿ |76|
ಅಗ್ನಿದಹಾದಪಿ ವಿಶಿಷ್ತಂ ವಾಕ್ಪಾರುಷ್ಯಂ |74|
ದಂಡಪಾರುಷ್ಯಾತ್ ಸರ್ವಜನದ್ವೇಷ್ಯೋ ಭವತಿ |75|
ಅಮಿತ್ರೋ ದಂಡನೀತ್ಯಾಮಾಯತ್ತಃ |77|
ದಂಡನೀತಿಮನುಷ್ಠನ್ ಪ್ರಜಾಃ ಸಂರಕ್ಷತಿ |78|
ದಂಡಃ ಸರ್ವಸಂಪದಾ ಯೋಜಯತಿ |79|
ದಂಡಾಭಾವೇ ತ್ರಿವರ್ಗಾಭಾವಃ |80|
ದಂಡಭಯಾದಕಾರ್ಯಾಣಿ ನ ಕುರ್ವಂತಿ |81|
ದಂಡನೀತ್ಯಾಮಾಯತ್ತಮಾತ್ಮರಕ್ಷಣಂ |82|
ಆತ್ಮನಿ ರಕ್ಷಿತೇ ಸರ್ವಂ ರಕ್ಷಿತಂ ಭವತಿ |83|
ಆತ್ಮಾಯತ್ತೌ ವೃದ್ಧಿವಿನಾಶೌ |84|
ದಂಡನೀತ್ಯಾದಿ ವಿಜ್ಞಾನೇ ಪ್ರಣೀಯತೇ |85|
ದುರ್ಬಲೋಽಪಿ ರಾಜಾ ನಾವಮಂತವ್ಯಃ |86|
ನಾಸ್ತ್ಯಗ್ನೇರ್ದೌರ್ಬಲ್ಯಂ |87|
ದಂಡೇನ ಪ್ರಣೀಯತೇ ವೃತ್ತಿಃ |88|
ವೃತ್ತಿಮೂಲೋಽರ್ಥಲಾಭಃ |89|
ಅರ್ಥಮೂಲೌ ಧರ್ಮಕಾಮೌ |90|
ಅರ್ಥಮೂಲಂ ಕಾರ್ಯಂ |91|
ಯತ್ನಪ್ರಯತ್ನಾತ್ ಕಾರ್ಯಸಿದ್ಧಿರ್ಭವತಿ ಸ ಉಪಾಯಃ |92|
ಉಪಾಯಪೂರ್ವಂ ಕಾರ್ಯಂ ನ ದುಷ್ಕರಂ ಸ್ಯಾತ್ |93|
ಅನುಪಾಯಪೂರ್ವಂ ಕಾರ್ಯಂ ಕೃತಮಪಿ ವಿನಶ್ಯತಿ |94|
ಕಾರ್ಯಾರ್ಥಿನಾಮುಪಾಯ ಏವ ಸಹಾಯಃ |95|
ಕಾರ್ಯಂ ಪುರುಷಕಾರೇಣ ಲಕ್ಷ್ಯಂ ಸಂಪದ್ಯತೇ |96|
ಪುರುಷಕಾರಮನುವರ್ತತೇ ದೈವಂ |97|
ದೈವಂ ವಿನಾ ಅತಿಪ್ರಯತ್ನಂ ಕರೋತಿ ಯತ್ತದ್ ವಿಫಲಂ |98|
ಅನೀಹಮಾನಸ್ಯ ವೃತ್ತಿರ್ನ ಸಂಪದ್ಯತೇ |99|
ಪೂರ್ವಂ ನಿಶ್ಚಿತ್ಯ ಪಶ್ಚಾತ್ಕಾರ್ಯಮಾರಭೇತ |100|
ಇತಿ ಚಾಣಕ್ಯಸೂತ್ರೇ ಪ್ರಥಮೋಽಧ್ಯಾಯಃ ||

ದ್ವಿತೀಯೋಽಧ್ಯಾಯಃ ||
ಅರ್ಥಮೂಲಂ ಸರ್ವಕಾರ್ಯಂ ಯದಲ್ಪಪ್ರಯತ್ನಾತ್ ಕಾರ್ಯಸಿದ್ಧಿರ್ಭವತಿ |1|
ಉಪಾಯಪೂರ್ವಂ ಕಾರ್ಯಂ ನ ದುಷ್ಕರಂ ಸ್ಯಾತ್ |2|
ಅನುಪಾಯಪೂರ್ವಕಾರ್ಯಂ ಕೃತಮಪಿ ವಿನಶ್ಯತಿ |3|
ಕಾರ್ಯಾರ್ಥಿನಾಮುಪಾಯ ಏವ ಸಹಾಯಃ |4|
ಕಾರ್ಯಂ ಪುರುಷಕಾರೇಣ ಲಕ್ಷ್ಯಂ ಸಂಪದ್ಯತೇ |5|
ಪುರುಷಕಾರಮನುವರ್ತತೇ ದೈವಂ |6|
ದೈವಂ ವಿನಾಽತಿಪ್ರಯತ್ನಂ ಯತ್ಕರೋತಿ ತದ್ವಿಫಲಂ |7|
ಅಸಮಾಹಿತಸ್ಯ ಕಾರ್ಯಂ ನ ವಿದ್ಯತೇ |8|
ಪೂರ್ವಂ ನಿಶ್ಚಿತ್ಯ ಪಶ್ಚಾತ್ಕಾರ್ಯಮಾರಭೇತ |9|
ಕಾರ್ಯಾಂತರೇ ದೀರ್ಘಸೂತ್ರತಾ ನ ಕರ್ತವ್ಯಾ |10|
ನ ಚಲಚಿತ್ತಸ್ಯ ಕಾರ್ಯಾವಾಪ್ತಿಃ |11|
ಹಸ್ತಗತಾವಮಾನನಾತ್ ಕಾರ್ಯವ್ಯತಿಕ್ರಮೋ ಭವತಿ |12|
ದೋಷವರ್ಜಿತಾನಿ ಕಾರ್ಯಾಣಿ ದುರ್ಲಭಾನಿ |13|
ದುರನಿಬಂಧಂ ಕಾರ್ಯಂ ನ ಆರಭೇತ |14|
ಕಾಲವಿತ್ ಕಾರ್ಯಂ ಸಾಧಯೇತ್ |15|
ಕಾಲಾತಿಕ್ರಮಾತ್ ಕಾಲ ಏವ ಫಲಂ ಪಿಬತಿ |16|
ಕ್ಷಣಂ ಪ್ರತಿ ಕಾಲವಿಕ್ಷೇಪಂ ನ ಕುರ್ಯಾತ್ ಸರ್ವಕೃತೇಷು |17|
ದೇಶಕಾಲವಿಭಾಗೌ ಜ್ಞಾತ್ವಾ ಕಾರ್ಯಮಾರಭೇತ |18|
ದೈವಹೀನಂ ಕಾರ್ಯಂ ಸುಸಾಧಮಪಿ ದುಸ್ಸಾಧಂ ಭವತಿ |19|
ನೀತಿಜ್ಞೋ ದೇಶಕಾಲೌ ಪರೀಕ್ಷೇತ |20|
ಪರೀಕ್ಷ್ಯಕಾರಿಣಿ ಶ್ರೀಶ್ಚಿರಂ ತಿಷ್ಠತಿ |21|
ಸರ್ವಾಶ್ಚ ಸಂಪದಃ ಸರ್ವೋಪಾಯೇನ ಪರಿಗೃಹ್ಣೀಯಾತ್ |22|
ಭಾಗ್ಯವಂತಮಪ್ಯಪರೀಕ್ಷ್ಯಕಾರಿಣಂ ಶ್ರೀಃ ಪರಿತ್ಯಜತಿ |23|
ಜ್ಞಾತ್ವಾಽನುಮಾನೈಶ್ಚ ಪರೀಕ್ಷಾ ಕರ್ತವ್ಯಾ |24|
ಯೋ ಯಸ್ಮಿನ್ಕರ್ಮಣಿ ಕುಶಲಸ್ತಂ ತಸ್ಮಿನ್ನೇವ ಯೋಜಯೇತ್ |25|
ದುಸ್ಸಾಧಮಪಿ ಸುಸಾಧಂ ಕರೋತ್ಯುಪಾಯಜ್ಞಃ |26|
ಅಜ್ಞಾನಿನಾ ಕೃತಮಪಿ ನ ಬಹುಮಂತವ್ಯಂ ಯಾದೃಚ್ಛಿಕತ್ವಾತ್ |27|
ಕೃಮಯೋಽಪೋ ಹಿ ಕದಾಚಿತ್ ರೂಪಾಂತರಾಣಿ ಕುರ್ವಂತಿ |28|
ಸಿದ್ಧಸ್ಯೈವ ಕಾರ್ಯಸ್ಯ ಪ್ರಕಾಶನಂ ಕರ್ತವ್ಯಂ |29|
ಜ್ಞಾನವತಾಮಪಿ ದೈವಮಾನುಷದೋಷಾತ್ ಕಾರ್ಯಾಣಿ ದುಷ್ಯಂತಿ |30|
ದೈವಂ ದೋಷಂ ಶಾಂತಿಕರ್ಮಣಾ ಪ್ರತಿಷೇಧಯೇತ್ |31|
ಮಾನುಷೀಂ ಕಾರ್ಯವಿಪತ್ತಿಂ ಕೌಶಲೇನ ವಿನಿವಾರಯೇತ್ |32|
ಕಾರ್ಯವಿಪತ್ತೌ ದೋಷಾನ್ ವರ್ಣಯಂತಿ ಬಾಲಿಶಾಃ |33|
ಕಾರ್ಯಾರ್ಥಿನಾ ದಾಕ್ಷಿಣ್ಯಂ ನ ಕರ್ತವ್ಯಂ |34|
ಕ್ಷೀರಾರ್ಥಿ ವತ್ಸೋ ಮಾತುರೂಧಃ ಪ್ರತಿಹಂತಿ |35|
ಅಪ್ರಯತ್ನಾತ್ ಕಾರ್ಯವಿಪತ್ತಿರ್ಭವೇತ್ |36|
ನ ದೈವಮಾತ್ರಪ್ರಮಾಣಾನಾಂ ಕಾರ್ಯಸಿದ್ಧಿಃ |37|
ಕಾರ್ಯಬಾಹ್ಯೋ ನ ಪೋಷಯತ್ಯಾಶ್ರಿತಾನ್ |38|
ಯಃ ಕಾರ್ಯಂ ನ ಪಶ್ಯತಿ ಸೋಽನ್ಧಃ |39|
ಪ್ರತ್ಯಕ್ಷಪರೋಕ್ಷನುಮಾನೈಃ ಕಾರ್ಯಾಣಿ ಪರೀಕ್ಷೇತ |40|
ಅಪರೀಕ್ಷಕಾರಿಣಂ ಶ್ರೀಃ ಪರಿತ್ಯಜತಿ |41|
ಪರೀಕ್ಷ್ಯ ತಾರ್ಯಾ ವಿಪತ್ತಿಃ |42|
ಸ್ವಶಕ್ತಿಂ ಜ್ಞಾತ್ವಾ ಕಾರ್ಯಮಾರಭೇತ |43|
ಸ್ವಜನಂ ತರ್ಪಯಯಿತ್ವಾ ಯಶ್ಶೇಷಭೋಜೀ ಸೋಽಮೃತಭೋಜೀ |44|
ಸಮ್ಯಗನುಷ್ಠಾನಾದಾಯಮುಖಾನಿ ವರ್ಧಂತೇ |45|
ನಾಸ್ತಿ ಭೀರೋಃ ಕಾರ್ಯಚಿಂತಾ |46|
ಸ್ವಾಮಿನಃ ಶೀಲಂ ಜ್ಞಾತ್ವಾ ಕಾರ್ಯಾರ್ಥೀ ಕಾರ್ಯಂ ಸಾಧಯೇತ್ |47|
ಧೇನೋಶ್ಶೀಲಜ್ಞೋ ಹಿ ಕ್ಷೀರಂ ಭುಂಕ್ತೇ |48|
ಕ್ಷುದ್ರೇ ಗುಹ್ಯಪ್ರಕಾಶನಮಾತ್ಮವಾನ್ ನ ಕುರ್ಯಾತ್ |49|
ಆಶ್ರಿತೈರಪ್ಯವಮನ್ಯತೇ ಮೃದುಸ್ವಭಾವಃ |50|
ತೀಕ್ಷ್ಣದಂಡಃ ಸರ್ವೇಷಾಮುದ್ವೇಜನೀಯೋ ಭವತಿ |51|
ಯಥಾರ್ಹದಂಡಕಾರೀ ಸ್ಯಾತ್ |52|
ಅಲ್ಪಸಾರಂ ಶ್ರುತವಂತಮಪಿ ನ ಬಹುಮನ್ಯತೇ ಲೋಕಃ |53|
ಅತಿಭಾರಃ ಪುರುಷಮವಸಾದಯತಿ |54|
ಯಃ ಸಂಸದಿ ಪರದೋಷಂ ಶಂಸತಿ ಸ ಸ್ವದೋಷಬಹುತ್ವಮೇವ ಪ್ರಖ್ಯಾಪಯತಿ |55|
ಆತ್ಮಾನಮೇವ ನಾಶಯತ್ಯನಾತ್ಮವತಾಂ ಕೋಪಃ |56|
ನಾಸ್ತ್ಯಪ್ರಾಪ್ಯಂ ಸತ್ಯವತಾಂ |57|
ನ ಕೇವಲೇನ ಸಾಹಸೇನ ಕಾರ್ಯಸಿದ್ಧಿರ್ಭವತಿ ||58
ವ್ಯಸನಾರ್ತೋ ವಿಸ್ಮರತ್ಯವಶ್ಯಕರ್ತವ್ಯಾನ್ |59|
ನಾಸ್ತ್ಯನಂತರಾಯಃ ಕಾಲವಿಕ್ಷೇಪೇ |60|
ಅಸಂಶಯವಿನಾಶಾತ್ ಸಂಶಯವಿನಾಶಃ ಶ್ರೇಯಾ |61|
ಕೇವಲಂ ಧನಾನಿ ನಿಕ್ಷೇಪ್ತುಂ ನ ಸ್ವಾರ್ಥಂ ನ ದಾನಂ ನ ಧರ್ಮಃ |62|
ನಾರ್ಯಾ ಆಗತೋಽರ್ಥಃ ತದ್ವಿಪರೀತಮನರ್ಥಭಾವಂ ಭಜತೇ |63|
ಯೋ ಧರ್ಮಾರ್ಥೌ ನ ವ್ಯರ್ಥಯತಿ ಸ ಕಾಮಃ ತದ್ವಿಪರೀತೋಽನರ್ಥಸೇವೀ |64|
ಋಜುಸ್ವಭಾವಪ್ರೋ ಜನೋ ದುರ್ಲಭಃ |65|
ಅವಮಾನೇನಾಗತಮೈಶ್ವರ್ಯಮವಮನ್ಯತ ಏವ ಸಾಧುಃ |66|
ಬಹೂನಪಿ ಹಿ ಗುಣಾನೇಕದೋಷೋ ಗ್ರಸತಿ |67|
ಮಹಾತ್ಮನಾ ಪರಂ ಸಾಹಸಂ ನ ಕರ್ತವ್ಯಂ |68|
ಕದಾಚಿದಪಿ ಚಾರಿತ್ರಂ ನ ಲಂಘಯೇತ್ |69|
ಕ್ಷುಧಾಽಽರ್ತೋ ನ ತೃಣಂ ಚರತಿ ಸಿಂಹಃ |70|
ಪ್ರಾಣಾದಪಿ ಪ್ರತ್ಯಯೋ ರಕ್ಷಿತವ್ಯಃ |71|
ಪಿಶುನೋ ನೇತಾ ಪುತ್ರದಾರೈರಪಿ ತ್ಯಜ್ಯತೇ |72|
ಇತಿ ದ್ವಿತೀಯೋಽಧ್ಯಾಯಃ ||

ತೃತೀಯೋಽಧ್ಯಾಯಃ ||
ಬಾಲಾದಪಿ ಯುಕ್ತಮರ್ಥ ಶೃಣುಯಾತ್ |1|
ಸತ್ಯಮಪ್ಯಶ್ರದ್ಧೇಯಂ ನ ವದೇತ್ |2|
ನಾಲ್ಪದೋಷಾದ್ಬಹುಗುಣಾಸ್ತ್ಯಜ್ಯಂತೇ |3|
ವಿಪಶ್ಚಿತ್ಸ್ವಪಿ ಸುಲಭಾ ದೋಷಾಃ |4|
ನಾಸ್ತಿ ರತ್ನಮಖಂಡಿತಂ |5|
ಮರ್ಯಾದಾತೀತಂ ನ ಕದಾಚಿದಪಿ ವಿಶ್ವಸೇತ್ |6|
ಅಪ್ರಿಯೇ ಕೃತೇ ಪ್ರಿಯಮಪಿ ದ್ವೇಷ್ಯಂ ಭವತಿ |7|
ನಮಂತ್ಯಪಿ ಹಿ ತುಲಾಕೋಟಿಃ ಕೃಪೋದಕಕ್ಷಯಂ ಕರೋತಿ |8|
ಸತಾಂ ಮತಂ ನಾತಿಕ್ರಾಮೇತ್ |9|
ಗುಣವದಾಶ್ರಯಾನ್ ನಿರ್ಗುಣೋಽಪಿ ಗುಣೀ ಭವತಿ |10|
ಕ್ಷೀರಾಶ್ರಿತಂ ಜಲಂ ಕ್ಷೀರಮಿವ ಭವತಿ |11|
ಮೃತ್ಪಿಂಡೇಽಪಿ ಪಾಟಲಿಪುಷ್ಪಂ ಸ್ವಗಂಧಮುತ್ಪಾದಯತಿ |12|
ರಜತಂ ಕನಕಸಂಗಾತ್ ಕನಕಂ ಭವತಿ |13|
ಉಪಕರ್ತರ್ಯಪಕರ್ತುಮಿಚ್ಛತ್ಯಬುಧಃ |14|
ನ ಪಾಪ ಕರ್ಮಣಾಮಾಕ್ರೋಶಭಯಂ |15|
ಉತ್ಸಾಹವತಾಂ ಶತ್ರವೋಽಪಿ ವಶೀಭವಂತಿ |16|
ವಿಕ್ರಮಧನಾ ಹಿ ರಾಜಾನಃ |17|
ನಾಸ್ತ್ಯಾಲಸ್ಯೈಹಿಕಮಾಮುಷ್ಮಿಕಂ ವಾ |18|
ನಿರುತ್ಸಾಹಾದ್ದೈವಂ ಪತತಿ |19|
ಮತ್ಸ್ಯಾರ್ಥೀವ ಜಾಲಮುಪಯುಜ್ಯಾರ್ಥಂ ಗೃಹ್ಣೀಯಾತ್ |20|
ಅವಿಶ್ವಸ್ತೇಷು ವಿಶ್ವಸೋ ನ ಕರ್ತವ್ಯಃ |21|
ವಿಷಂ ವಿಷಮೇವ ಸಾರ್ವಕಾಲಂ |22|
ಅರ್ಥಸಮಾದಾನೇ ವೈರಿಣಾಂ ಸಂಗ ಏವ ನ ಕರ್ತವ್ಯಃ |23|
ಅರ್ಥಸಿದ್ಧೌ ವೈರಿಣಂ ನ ವಿಶ್ವಸೇತ್ |24|
ಅರ್ಥಾಧೀನ ಏವ ನಿಯತಸಂಬಂಧಃ |25|
ಶತ್ರೋರಪಿ ಸುತಃ ಸಖಾ ರಕ್ಷಿತವ್ಯಃ |26|
ಯಾವಚ್ಛತ್ರೋಶ್ಚ್ಛಿದ್ರಂ ಪಶ್ಯತಿ ತಾವದ್ಧಸ್ತೇನ ವಾ ಸ್ಕಂಧೇನ ವಾ ಸಂವಾಹ್ಯಃ ಛಿದ್ರೇ ತು ಪ್ರಹರೇತ್ |27|
ಆತ್ಮಛಿದ್ರಂ ನ ಪ್ರಕಾಶಯೇತ್ |28|
ಛಿದ್ರಪ್ರಹಾರಿಣಃ ಶತ್ರವೋಽಪಿ |29|
ಹಸ್ತಗತಮಪಿ ಶತ್ರುಂ ನ ವಿಶ್ವಸೇತ್| 30||
ಸ್ವಜನಸ್ಯ ದುರ್ವೃತಂ ನಿವಾರಯೇತ್ |31|
ಸ್ವಜನಾವಮಾನೋಽಪಿ ಮನಸ್ವಿನಾಂ ದುಃಖಮಾವಹತಿ |32|
ಏಕಾಂಗದೋಷಃ ಪುರುಷಮವಸಾದಯತಿ |33|
ಶತ್ರುಂ ಜಯತಿ ಸುವೃತ್ತತಾ |34|
ನಿಕೃತಿಪ್ರಿಯಾ ನೀಚಾಃ |35|
ನೀಚಸ್ಯ ಮತಿರ್ನ ದಾತವ್ಯಾ |36|
ನೀಚೇಷು ವಿಶ್ವಾಸೋ ನ ಕರ್ತವ್ಯಃ |37|
ಸುಪೂಜಿತೋಽಪಿ ದುರ್ಜನಃ ಪೀಡಯತ್ಯೇವ |38|
ಚಂದನದೀನಪಿ ದಾವೋಽಗ್ನಿರ್ದಹತ್ಯೇವ |39|
ಕದಾಽಪಿ ಕಮಪಿ ಪುರುಷಂ ನಾವಮನ್ಯೇತ |40|
ಕ್ಷಂತವ್ಯಮಿತಿ ಪುರುಷಂ ನ ಬಾಧೇತ |41|
ಭತ್ರ್ರಾಽಧಿಕಂ ರಹಸ್ಯುಕ್ತಂ ವಕ್ತುಮಚ್ಛಿಂತ್ಯಬುಧಃ |42|
ಅನುರಾಗ್ಸ್ತು ಫಲೇನ (ಹಿತೇನ) ಸೂಚ್ಯತೇ |43|
ಆಜ್ಞಾಫಲಮೈಶ್ವರ್ಯಂ |44|
ದಾತವ್ಯಮಪಿ ಬಾಲಿಶಃ ಪರಿಕ್ಲೇಶೇನ ದಾಸ್ಯತಿ |45|
ಮಹದೈಶ್ವರ್ಯಂ ಪ್ರಾಪ್ಯಾಪಿ ಅಧೃತಿಮಾನ್ ವಿನಶ್ಯತಿ |46|
ನಾಸ್ತ್ಯಧೃತೇರೈಹಿಕಮಾಮುಷ್ಮಿಕಂ ವಾ |47|
ನ ದುರ್ಜನೈಃ ಸಹ ಸಂಸರ್ಗಃ ಕರ್ತವ್ಯಃ |48|
ಶೌಂಡಹಸ್ತಗತಂ ಪಯೋಽಪ್ಯವಮನ್ಯತೇ ಜನಃ |49|
ಕಾರ್ಯಸಂಕಟೇಷ್ವರ್ಥವ್ಯವಸಾಯಿನೀ ಬುದ್ಧಿಃ |50|
ಮಿತಭೋಜನಂ ಸ್ವಾಸ್ಥ್ಯಂ |51|
ಪಥ್ಯಮಪ್ಯಪಥ್ಯಾಜೀರ್ಣೇ ನಾಶ್ನೀಯಾತ್ |52|
ಜೀರ್ಣಭೋಜಿನಂ ವ್ಯಾಧಿರ್ನೋಪಸರ್ಪತಿ |53|
ಜೀರ್ಣಶರೀರೇ ವರ್ಧಮಾನಂ ವ್ಯಾಧಿಂ ನೋಪೇಕ್ಷೇತ |54|
ಅಜೀರ್ಣೇ ಭೋಜನಂ ದುಃಖಂ |55|
ಶತ್ರೋರಪಿ ವಿಶಿಷ್ಯತೇ ವ್ಯಾಧಿಃ |56|
ದಾನಂ ನಿಧಾನಮನುಗಾಮಿ |57|
ಪಟುತರೇಽಪಿ ತೃಷ್ಣಾಪರೇ ಸುಲಭಮತಿಸಂಧಾನಂ |58|
ತೃಷ್ಣಯಾ ಮತಿಶ್ಛಾದ್ಯತೇ |59|
ಕಾರ್ಯಬಹುತ್ವೇ ಬಹುಫಲಮಾಯತಿಕಂ ಕುರ್ಯಾತ್ |60|
ಸ್ವಯಮೇವಾವಸ್ಕನ್ನಂ ಕಾರ್ಯಂ ನಿರೀಕ್ಷೇತ |61|
ಮೂರ್ಖೇಷು ಸಾಹಸಂ ನಿಯತಂ |62|
ಮೂರ್ಖೇಷು ವಿವಾದೋ ನ ಕರ್ತವ್ಯಃ |63|
ಮೂರ್ಖೇಷು ಮೂರ್ಖವದೇವ ಕಥಯೇತ್ |64|
ಆಯಸೈರಾಯಸಂ ಛೇದ್ಯಂ |65|
ನಾಸ್ತ್ಯಧೀಮತಸ್ಸಖಾ |66|
ಇತಿ ತೃತೀಯೋಽಧ್ಯಾಯಃ ||

ಚತುರ್ಥೋಽಧ್ಯಾಯಃ ||
ಧರ್ಮೇಣ ಧಾರ್ಯತೇ ಲೋಕಃ |1|
ಪ್ರೇತಮಪಿ ಧರ್ಮಾಧರ್ಮಾವನುಗಚ್ಛತಃ |2|
ದಯಾ ಧರ್ಮಸ್ಯ ಜನ್ಮಭೂಮಿಃ |3|
ಧರ್ಮಮೂಲೇ ಸತ್ಯದಾನೇ |4|
ಧರ್ಮೇಣ ಜಯತಿ ಲೋಕಾನ್ |5|
ಮೃತ್ಯರಪಿ ಧರ್ಮಿಷ್ಠಂ ರಕ್ಷತಿ |6|
ಧರ್ಮಾದ್ವಿಪರೀತಂ ಪಾಪಂ ಯತ್ರ ಯತ್ರ ಪ್ರಸಜ್ಯತೇ ತತ್ರ ತತ್ರ ಧರ್ಮಾವಮತಿರೇವ ಮಹತೀ ಪ್ರಸಜ್ಯತೇ |7|
ಉಪಸ್ಥಿತವಿನಾಶಾನಾಂ ಪ್ರಕೃತಿಃ ಆಕಾರೇಣ ಕಾರ್ಯೇಣ ಚ ಲಕ್ಷ್ಯತೇ |8|
ಆತ್ಮವಿನಾಶಂ ಸೂಚಯತ್ಯಧರ್ಮಬುದ್ಧಿಃ |9|
ಪಿಶುನವಾದಿನೋ ರಹಸ್ಯಂ ಕುತಃ  |10|
ಪರರಹಸ್ಯಂ ನೈವ ಶ್ರೋತವ್ಯಂ |11|
ವಲ್ಲಭಸ್ಯ ಸ್ವಾರ್ಥಪರತ್ವಮಧರ್ಮಯುಕ್ತಂ |12|
ಸ್ವಜನೇಷ್ವಪ್ಯತಿಕ್ರಮೋ ನ ಕರ್ತವ್ಯಃ |13|
ಮತಾಽಪಿ ದುಷ್ಟಾ ತ್ಯಾಜ್ಯಾ |14|
ಸ್ವಹಸ್ತೋಽಪಿ ವಿಷದಿಗ್ಧಶ್ಛೇದ್ಯಃ |15|
ಪರೋಽಪಿ ಚ ಹಿತೋ ಬಂಧುಃ |16|
ಕಕ್ಷಾದಪ್ಯೌಷಧಂ ಗೃಹ್ಯತೇ |17|
ನಾಸ್ತಿ ಚೋರೇಷು ವಿಶ್ವಾಸಃ |18|
ಅಪ್ರತೀಕಾರೇಷ್ವನಾದರೋ ನ ಕರ್ತವ್ಯಃ |19|
ವ್ಯಸನಂ ಮನಾಗಪಿ ಬಾಧತೇ |20|
ಅಮರವದರ್ಥಜಾತಮಾರ್ಜಯೇತ್ |21|
ಅರ್ಥವಾನ್ ಸರ್ವಲೋಕಸ್ಯ ಬಹುಮತಃ |22|
ಮಹೇಂದ್ರಮಪ್ಯರ್ಥಹೀನಂ ನ ಬಹುಮನ್ಯತೇ ಲೋಕಃ |23|
ದಾರಿದ್ರ್ಯಂ ಖಲು ಪುರುಷಸ್ಯ ಸಜೀವಿತಂ ಮರಣಂ |24|
ವಿರೂಪೋಽಪ್ಯರ್ಥವಾನ್ ಸುರೂಪಃ |25|
ಅದಾತಾರಮಪ್ಯರ್ಥವಂತಮರ್ಥಿನೋ ನ ತ್ಯಜಂತಿ |26|
ಅಕುಲೀನೋಽಪಿ ಧನವಾನ್ ಕುಲೀನಾದ್ವಿಶಿಷ್ಟಃ |27|
ನಾಸ್ತ್ಯವಮಾನಭಯಮನಾರ್ಯಸ್ಯ |28|
ನೋದ್ಯೋಗವತಾಂ ವೃತ್ತಿಭಯಂ |29|
ನ ಜಿತೇಂದ್ರಿಯಾಣಾಂ ವಿಷಯಭಯಂ |30|
ನ ಕೃತಾರ್ಥಾನಾಂ ಮರಣಭಯಂ |31|
ಕಸ್ಯಚಿದರ್ಥಂ ಸ್ವಮಿವ ಮನ್ಯತೇ ಸಾಧುಃ |32|
ಪರವಿಭವೇಷ್ವಾದರೋ ನ ಕರ್ತವ್ಯಃ |33|
ಪರವಿಭವೇಷ್ವಾದರೋಽಪಿ ನಾಶಮೂಲಂ |34|
ಪಲಾಲಮಪಿ ಪರದ್ರವ್ಯಂ ನ ಹರ್ತವ್ಯಂ |35|
ಪರದ್ರವ್ಯಾಪಹರಣಮಾತ್ಮದ್ರವ್ಯನಾಶಹೇತುಃ |36|
ನ ಚೌರ್ಯಾತ್ ಪರಂ ಮೃತ್ಯುಪಾಶಃ |37|
ಯವಾಗೂರಪಿ ಪ್ರಾಣಧಾರಣಂ ಕರೋತಿ ಕಾಲೇ |38|
ನ ಮೃತಸ್ಯೌಷಧಂ ಪ್ರಯೋಜನಂ |39|
ಸಮಕಾಲೇ ಪ್ರಭುತ್ವಸ್ಯಪ್ರಯೋಜನಂ ಭವತಿ |40|
ನೀಚಸ್ಯ ವಿದ್ಯಾಃ ಪಾಪಕರ್ಮಣ್ಯೇವ ತಂ ಯೋಜಯಂತಿ |41|
ಪಯಃಪಾನಮಪಿ ವಿಷವರ್ಧನಂ ಭುಜಂಗಸ್ಯ ನ ತ್ವಮೃತಂ ಸ್ಯಾತ್ |42|
ನ ಧಾನ್ಯಸಮೋ ಹ್ಯರ್ಥಃ |43|
ನ ಕ್ಷುಧಾಸಮಃ ಶತ್ರುಃ |44|
ಅಕೃತೇರ್ನಿಯತಾ ಕ್ಷುತ್ |45|
ನಾಸ್ತ್ಯಭಕ್ಷಂ ಕ್ಷುಧಿತಸ್ಯ |46|
ಇಂದ್ರಿಯಾಣಿ ಪ್ರತಿಪದಂ ನರಾನ ಜರಾವಶಾನ್ ಕುರ್ವಂತಿ |47|
ಸಾನುಕ್ರೋಶಂ ಭರ್ತಾರಮಾಜೀವೇತ್ |48|
ಲುಬ್ಧಸೇವೀ ಪಾವಕೇಚ್ಛಯಾ ಖದ್ಯೋತಂ ಧಮತಿ |49|
ವಿಶೇಷಜ್ಞಂ ಸ್ವಾಮಿನಮಾಶ್ರಯೇತ್ |50|
ಪುರುಷಸ್ಯ ಮೈಥುನಂ ಜರಾ |51|
ಸ್ತ್ರೀಣಾಮಮೈಥುನಂ ಜರಾ |52|
ನ ನೀಚೋತ್ತಮಯೋತ್ವೈವಾಹಃ |53|
ಅಗಮಯಾಗಮನಾದಾಯುರ್ಯಶಃ ಪುಣ್ಯಾನಿ ಕ್ಷೀಯಂತೇ |54|
ನಾಸ್ತ್ಯಹಂಕಾರಸಮಃ ಶತ್ರುಃ |55|
ಸಂಸದಿ ಶತ್ರು ನ ಪರಿಕ್ರೋಶೇತ |56|
ಶತ್ರುವ್ಯಸನಂ ಶ್ರವಣಸುಖಂ |57|
ಅಧನಸ್ಯ ಬುದ್ಧಿರ್ನ ವಿದ್ಯತೇ |58|
ಹಿತಮಪ್ಯಧನಸ್ಯ ವಾಕ್ಯಂ ನ ಗೃಹ್ಯತೇ |59|
ಅಧನಃ ಸ್ವಭಾರ್ಯಯಾಽಪ್ಯವಮನ್ಯತೇ |60|
ಪುಷ್ಪಹೀನಂ ಸಹಕಾರಮಪಿ ನೋಪಾಸತೇ ಭ್ರಮರಾಃ |61|
ಇತಿ ಚತುರ್ಥೋಽಧ್ಯಾಯಃ ||

ಪಂಚಮೋಽಧ್ಯಾಯಃ ||
ವಿದ್ಯಾ ಧನಮಧನಾನಾಂ |1|
ವಿದ್ಯಾ ಚೋರೈರಪಿ ನ ಗ್ರಾಹ್ಯಾ |2|
ವಿದ್ಯಾ ಸುಲಭಾ ಖ್ಯಾತಿಃ |3|
ಯಶಃ ಶರೀರಂ ನ ವಿನಶ್ಯತಿ |4|
ಯಃ ಪರಾಥಮನ್ಯಮುಪಸರ್ಪತಿ ಸ ಸತ್ಪುರುಷಃ |5|
ಇಂದ್ರಿಯಾಣಾಂ ಪ್ರಶಮಂ ಶಾಸ್ತ್ರಂ |6|
ಅಕಾರ್ಯಪ್ರವೃತ್ತೇಃ ಶಾಸ್ತ್ರಾಂಕುಶಂ ನಿವಾರಯತಿ |7|
ನೀಚಸ್ಯ ವಿದ್ಯಾ ನೋಪೇತವ್ಯಾ |8|
ಮ್ಲೇಚ್ಛಭಾಷಣಂ ನ ಶಿಕ್ಷೇತ |9|
ಮ್ಲೇಚ್ಛಾನಾಮಪಿ ಸುವೃತ್ತಂ ಗ್ರಾಹ್ಯಂ |10|
ಗುಣೇ ನ ಮತ್ಸರಃ ಕರ್ತವ್ಯಃ |11|
ಶತ್ರೋರಪಿ ಸುಗುಣೋ ಗ್ರಾಹ್ಯಃ |12|
ವಿಷಾದಪ್ಯಮೃತಂ ಗ್ರಾಹ್ಯಂ |13|
ಅವಸ್ಥಯಾ ಪುರುಷಃ ಸಂಮಾನ್ಯತೇ |14|
ಸ್ಥಾನ ಏವ ನರಾಃ ಪೂಜ್ಯಂತೇ |15|
ಆರ್ಯವೃತಮನುತಿಷ್ಠೇತ್ |16|
ಕದಾಪಿ ಮರ್ಯಾದಾಂ ನಾತಿಕ್ರಾಮೇತ್ |17|
ನಾಸ್ತ್ಯರ್ಘಃ ಓಉರುಷರತ್ನಸ್ಯ |18|
ನ ಸ್ತ್ರೀರತ್ನಸಮಂ ರತ್ನಂ |19|
ಸುದುರ್ಲಭಂ ಹಿ ರತ್ನಂ |20|
ಅಯಶೋ ಭಯಂ ಭಯೇಷು |21|
ನಾಸ್ತ್ಯಲಸಸ್ಯ ಶಾಸ್ತ್ರಾಧಿಗಮಃ |22|
ನ ಸ್ತ್ರೈಣಸ್ಯ ಸ್ವರ್ಗಾಪ್ತಿರ್ಧರ್ಮಕೃತ್ಯಂ ಚ |23|
ಸ್ತ್ರಿಯೋಽಪಿ ಸ್ತ್ರೈಣಮವಮನ್ಯಂತೇ |24|
ನ ಪುಷ್ಪಾರ್ಥೀ ಸಿಂಚತಿ ಶುಷ್ಕತರುಂ |25|
ಅದ್ರವ್ಯಪ್ರಯತ್ನೋ ವಾಲಿಕಾಕ್ವಾಥನಾದನನ್ಯಃ |26|
ನ ಮಹಾಜನಹಾಸಃ ಕರ್ತವ್ಯಃ |27|
ಕಾರ್ಯಸಂಪದಂ ನಿಮಿತ್ತಾನಿ ಸೂಚಯಂತಿ |28|
ನಕ್ಷತ್ರಾದಪಿ ನಿಮಿತ್ತಾನಿ ವಿಶೇಷಯಂತಿ |29|
ನ ತ್ವರಿತಸ್ಯ ನಕ್ಷತ್ರಪರೀಕ್ಷಾ |30|
ಪರಿಚಯೇ ದೋಷಾ ನ ಛದ್ಯಂತೇ ||31 |
ಸ್ವಯಮಶುದ್ಧಃ ಪರಾನಾಶಂಕತೇ |32|
ಸ್ವಭಾವೋ ದುರತಿಕ್ರಮಃ |33|
ಅಪರಾಧಾನುರೂಪೋ ದಂಡಃ |34|
ಪ್ರಶ್ನಾನುರೂಪಂ ಪ್ರತಿವಚನಂ |35|
ವಿಭವಾನುರೂಪಮಾಭರಣಂ |36|
ಕುಲಾನುರೂಪಂ ವೃತಂ |37|
ಕಾರ್ಯಾನುರೂಪಃ ಪ್ರಯತ್ನಃ |38|
ಪಾತ್ರಾನುರೂಪಂ ದಾನಂ |39|
ವಯೋಽನುರೂಪಓ ವೇಷಃ |40|
ಸ್ವಾಮ್ಯನುಕೂಲೋ ಭೃತ್ಯಃ |41|
ಭತೃವಶವರ್ತಿನೀ ಭಾರ್ಯಾ |42|
ಗುರುವಶಾನುವರ್ತೀ ಶಿಷ್ಯಃ |43|
ಪಿತೃವಶಾನುವರ್ತೀ ಪುತ್ರಃ |44|
ಅತ್ಯುಪಚಾರಃ ಶಂಕಿತವ್ಯಃ |45|
ಸ್ವಾಮಿನಿ ಕುಪಿತೇ ಸ್ವಾಮಿನಮೇವಾನುವರ್ತೇತ |46|
ಮಾತೃತಾಡಿತೋ ವತ್ಸೋ ಮಾತರಮೇವಾನುರೋದಿತಿ |47|
ಸ್ನೇಹವತಃ ಸ್ವಲ್ಪೋ ಹಿ ರೋಷಃ |48|
ಬಾಲಿಶಃ ಆತ್ಮಛಿದ್ರಂ ನ ಪಶ್ಯತಿ ಅಪಿ ತು ಪರಚ್ಛಿದ್ರಮೇವ ಪಶ್ಯತಿ |49|
ಸದೋಪಚಾರಃ ಕಿತವಃ |50|
ಕಾಮ್ಯೈರ್ವಿಶೇಷೈರೂಪಚಾರಮುಪಚಾರಃ |51|
ಚಿರಪರಿಚಿತಾನಾಮತ್ಯುಪಚಾರಃ ಶಂಕಿತವ್ಯಃ |52|
ಶ್ವಸಹಸ್ರಾದೇಕಾಕಿನೀ ಗೌಃ ಶ್ರೇಯಸೀ |53|
ಶ್ವೋ ಮಯೂರಾದದ್ಯ ಕಪೋತೋ ವರಃ |54|
ಅತಿಸಂಗೋ ದೋಷಮುತ್ಪಾದಯತಿ |55|
ಸರ್ವಂ ಜಯತ್ಯಕ್ರೋಧಃ |56|
ಯದ್ಯಪಕಾರಿಣೀ ಕೋಪಃ ಕರ್ತವ್ಯಃ ತರ್ಹಿ ಸ್ವಕೋಪೇ ಏವಕೋಪಃ ಕರ್ತವ್ಯಃ |57|
ಮತಿಮತ್ಸು ಮೂರ್ಖಮಿತ್ರಗುರುವಲ್ಲಭೇಷು ವಿವಾದೋ ನ ಕರ್ತವ್ಯಃ |58|
ನಾಸ್ತ್ಯಪಿಶಾಚಮೈಶ್ವರ್ಯಂ |59|
ನಾಸ್ತಿ ಧನವತಾಂ ಸುಕರ್ಮಸು ಶ್ರಮಃ |60|
ನಾಸ್ತಿ ಗತಿಶ್ರಮೋ ಯಾನವತಾಂ |61|
ಅಲೋಹಮಯಂ ನಿಗಡಂ ಕಲತ್ರಂ |62|
ಯೋ ಯಸ್ಮಿನ್ ಕರ್ಮಣಿ ಕುಶಲಃ ಸ ತಸ್ಮಿನ್ ಯೋಕ್ತವ್ಯಃ |63|
ದುಷ್ಕಲತ್ರಂ ಮನಸ್ವಿನಾಂ ಶರೀರಕರ್ಶನಂ |64|
ಅಪ್ರಮತ್ತೋ ದಾರಾನ್ ನಿರೀಕ್ಷೇತ |65|
ಸ್ತ್ರೀಷು ಕಿಂಚಿದಪಿ ನ ವಿಶ್ವಸೇತ್ |66|
ನ ಸಮಾಧಿಃ ಸ್ತ್ರೀಷು ಲೋಕಜ್ಞತಾ ಚ |67|
ಗುರೂಣಾಂ ಮಾತಾ ಗರೀಯಸೀ |68|
ಸರ್ವಾವಸ್ಥಾಸು ಮಾತಾ ಭರ್ತವ್ಯಾ |69|
ವೈರೂಪ್ಯಮಲಂಕಾರೇಣಾಚ್ಛಾದ್ಯತೇ |70|
ಸ್ತ್ರೀಣಾಂ ಭೂಷಣಂ ಲಜ್ಜಾ |71|
ವಿಪ್ರಾಣಾಂ ಭೂಷಣಂ ವೇದಃ |72|
ಸರ್ವೇಷಾಂ ಭೂಷಣಂ ಧರ್ಮಃ |73|
ಭೂಷಣಾನಾಂ ಭೂಷಣಂ ಸವಿನಯ ವಿದ್ಯಾ |74|
ಇತಿ ಪಂಚಮೋಽಧ್ಯಾಯಃ ||

ಷಷ್ಠೋಽಧ್ಯಾಯಃ ||
ಅನುಪದ್ರವಂ ದೇಶಮಾವಸೇತ್ |1|
ಸಾಧುಜನಬಹುಲೋ ದೇಶಃ ಆಶ್ರಯಣೀಯಃ |2|
ರಾಜ್ಞೋ ಭೇತವ್ಯಂ ಸಾರ್ವಕಾಲಂ |3|
ನ ರಾಜ್ಞಃ ಪರಂ ದೈವತಂ |4|
ಸುದೂರಮಪಿ ದಹತಿ ರಾಜವಹ್ನಿಃ |5|
ರಿಕ್ತಹಸ್ತೋ ನ ಹಾಜಾನಮಭಿಗಚ್ಛೇತ್ ಗುರುಂ ದೈವೇ ಚ |6|
ಕುಟಿಂಬಿನೋ ಭೇತವ್ಯಂ |7|
ಗಂತವ್ಯಂ ಚ ಸದಾ ರಾಜಕುಲಂ |8|
ರಾಜಪುರುಷೈಃ ಸಂಬಂಧಂ ಕುರ್ಯಾತ್ |9|
ಹಾಜದಾಶೀ ನ ಸೇವಿತವ್ಯಾ |10|
ನ ಚಕ್ಷುಷಾಽಪಿ ರಾಜಾನಂ ನಿರೀಕ್ಷೇತ |11|
ಪುತ್ರೇ ಗುಣವತಿ ಕುಟುಂಬಿನಃ ಸ್ವರ್ಗಃ |12|
ಪುತ್ರಾ ವಿದ್ಯಾನಾಂ ಪಾರಂ ಗಮಚಿತವ್ಯಾಃ |13|
ಜನಪದಾರ್ಥಂ ಗ್ರಾಮಂ ತ್ಯಜೇತ್ |14|
ಗ್ರಾಮಾರ್ಥಂ ಕುಟುಂಬಸ್ತ್ಯಜತೇ |15|
ಅತಿಲಾಭಃ ಪುತ್ರಲಾಭಃ |16|
ದುರ್ಗತೇರಯಃ ಪಿತರೌ ರಕ್ಷತಿ ಸ ಪುತ್ರಃ |17|
ಯಃ ಕುಲಂ ಪ್ರಖ್ಯಾಪಯತಿ ಸ ಪುತ್ರಃ |18|
ನಾನಪತ್ಯಸ್ಯ ಸ್ವರ್ಗಃ |19|
ಯಾ ಪ್ರಸೂತೇ ಸಾ ಭಾರ್ಯಾ |20|
ತೀರ್ಥಸಮವಾಯೇ ಪುತ್ರವತೀಮನುಗಚ್ಛೇತ್ |21|
ನ ತೀರ್ಥಾಭಿಗಮನಾದ್ಬ್ರಹ್ಮಚರ್ಯಂ ನಶ್ಯತಿ |22|
ನ ಪರಕ್ಷೇತ್ರೇ ಬೀಜಂ ವಿನಿಕ್ಷಿಪೇತ್ |23|
ಪುತ್ರಾರ್ಥಾ ಹಿ ಸ್ತ್ರಿಯಃ |24|
ಸ್ವದಾಸೀಪರಿಗ್ರಹೋ ಹಿ ಸ್ವಸ್ಯೈವ ದಾಸಭಾವಾಪಾದನಂ |25|
ಉಪಸ್ಥಿತವಿನಾಶಃ ಪಥ್ಯವಾಕ್ಯಂ ನ ಶೃಣೋತಿ |26|
ನಾಸ್ತಿ ದೇಹಿನಾಂ ಸುಖದುಃಖಾಭಾವಃ |27|
ಮಾತರಮಿವ ವತ್ಸಾಃ ಸುಖದುಃಖಾನಿ ಕರ್ತಾರಮೇವಾನುಗಚ್ಛಂತಿ |28|
ತಿಲಮಾತ್ರಮಪ್ಯುಪಕಾರಂ ಶೈಲಮಾತ್ರಂ ಮನ್ಯತೇ ಸಾಧುಃ |29|
ಉಪಕಾರೋಽನಾರ್ಥಷ್ವಕರ್ತವ್ಯಃ |30|
ಪ್ರತ್ಯುಪಕಾರಭಯಾದನಾರ್ಯಃ ಶತ್ರುರ್ಭವತಿ |31|
ಸ್ವಲ್ಪೋಪಕಾರಕೃತೇಽಪಿ ಪ್ರತ್ಯುಪಕಾರಂ ಕರ್ತುಮಾರ್ಯೋ ಜಾಗರ್ತಿ |32|
ನ ಕದಾಽಪಿ ದೇವತಾ ಮಂತವ್ಯಾ |33|
ನ ಚಕ್ಷುಷಃ ಸಮಂ ಜ್ಯೋತಿರಸ್ತಿ |34|
ಚಕ್ಷುರ್ಹಿ ಶರೀರಿಣಾಂ ನೇತಾ |35|
ಅಪಚಕ್ಷುಷಃ ಕಿಂ ಶರೀರೇಣ |36|
ನಾಪ್ಸು ಮೂತ್ರಂ ಕುರ್ಯಾತ್ |37|
ನ ನಗ್ನೋ ಜಲಂ ಪ್ರವಿಶೇತ್ |38|
ಯಥಾ ಶರೀರಂ ತಥಾ ಜ್ಞಾನಂ |39|
ಯಥಾ ಬುದ್ಧಿಸ್ತಥಾ ವಿಭವಃ |40|
ಅಗ್ವಾವಕ್ನಿಂ ನ ನಿಙಿಪೇತ್ |41|
ತಪಸ್ವಿನಃ ಪೂಜನೀಯಾಃ |42|
ಪರದಾರಾನ್ ನ ಗಚ್ಛೇತ್ |43|
ಅನ್ನದಾನಂ ಭ್ರೂಣಹತ್ಯಾಮಪಿ ಮಾಷ್ರ್ಟಿ |44|
ನ ವೇದಬಾಹ್ಯೋ ಧರ್ಮಃ |45|
ಕಥಂಚಿದಪಿ ಧರ್ಮಂ ನಿಷೇವೇತ |46|
ಸ್ವರ್ಗಂ ನಯತಿ ಸೂನೃತಂ |47|
ನಾಸ್ತಿ ಸತ್ಯಾತ್ ಪರಂ ತಪಃ |48|
ಸತ್ಯಂ ಸ್ವರ್ಗಸ್ಯ ಸಾಧನಂ |49|
ಸತ್ಯೇನ ಧಾರ್ಯತೇ ಲೋಕಃ |50|
ಸತ್ಯಾದ್ದೇವೋ ವರ್ಷತಿ |51|
ನಾನೃತಾತ್ ಪಾತಕಂ ಪರಂ |52|
ನ ಮೀಮಾಂಸ್ಯಾ ಗುರವಃ |53|
ಖಲತ್ವಂ ನೋಪೇಯಾತ್ |54|
ನಾಸ್ತಿ ಖಲಸ್ಯ ಮಿತ್ರಂ |55|
ಲೋಕಯಾತ್ರಾ ದರಿದ್ರಂ ಬಾಧತೇ |56|
ಅತಿಶೂರೋ ದಾನಶೂರಃ |57|
ಗುರುದೇವಬ್ರಾಹ್ಮಣೇಷು ಭಕ್ತಿರ್ಭೂಷಣಂ |58|
ಸರ್ವಸ್ಯ ಭೂಷಣಂ ವಿನಯಃ |59|
ಅಕುಲೀನೋಽಪಿ ವಿನೀತಃ ಕುಲೀನಾತ್ವಿಶಿಷ್ಟಃ |60|
ಆಚಾರಾದಾಯುರ್ವರ್ಧತೇ ಕೀರ್ತಿಃ ಶ್ರೇಯಶ್ಚ |61|
ಪ್ರಿಯಮಪ್ಯಹಿತಂ ನ ವಕ್ತವ್ಯಂ |62|
ಬಹುಜನವಿರುದ್ಧಮೇಕಂ ನಾನುವರ್ತೇತ |63|
ನ ಕೃತಾರ್ಥಸ್ಯ ನೀಚೇಷು ಸಂಬಂಧಃ |64|
ಋಣಶತ್ರುವ್ಯಾಧಯೋ ನಿಃಶೇಷಾಃ ಕರ್ತವ್ಯಾಃ |65|
ಭೂತ್ಯನುವರ್ತನಂ ಪುರುಷಸ್ಯ ರಸಾಯನಂ |66|
ನಾರ್ಥಿಷ್ವವಜ್ಞಾ ನರಕಾನ್ನಿವರ್ತನಂ |67|
ದುಷ್ಕರಂ ಕರ್ಮ ಕಾರಯಿತ್ವಾ ಕರ್ತಾರಮವಮನ್ಯತೇ ನೀಚಃ |68|
ನಾಕೃತಜ್ಞಸ್ಯ ನರಕಾನ್ನಿವರ್ತನಂ |69|
ಜಿಹ್ವಾಯತ್ತೌ ವೃದ್ಧಿವನಾಶೌ |70|
ವಿಷಯಾಮೃತಯೋರಾಕರೀ ಜಿಹ್ವಾ |71|
ಪ್ರಿಯವಾದಿನೋ ನ ಶತ್ರುಃ |72|
ಸ್ತುತಾ ಅಪಿ ದೇವತಾಸ್ತುಷ್ಯಂತಿ |73|
ಅನಂತಮಪಿ ದುರ್ವಚನಂ ಚಿರಂ ತಿಷ್ಠತಿ |74|
ರಾಜದ್ವಿಷ್ಟಂ ನ ವಕ್ತವ್ಯಂ |75|
ಶ್ರುತಿಸುಖಾತ್ ಕೋಕಿಲಾನಾಪಾದಪಿ ತುಷ್ಯಂತಿ ಜನಾಃ |76|
ಸ್ವಧರ್ಮಹೇತುಃ ಸತ್ಪುರುಷಃ |77|
ನಾಸ್ತ್ಯರ್ಥಿನೋ ಗೌರವಂ |78|
ಸ್ತ್ರೀಣಾಂ ಭೂಷಣಂ ಸೌಭಾಗ್ಯಂ |79|
ಶತ್ರೋರಪಿ ನ ಪಾತನೀಯ ವೃತ್ತಿಃ |80|
ಅಪ್ರಯತ್ನೋದಕಂ ಕ್ಷೇತ್ರಂ |81|
ಏರಂಡಮವಲಂಬ್ಯ ಕುಂಜರಂ ನ ಕೋಪಯೇತ್ |82|
ಅತಿಪ್ರವೃದ್ಧಾಪಿ ಶಾಲ್ಮಲೀ ವಾರಣಸ್ತಂಭೋ ನ ಭವತಿ |83|
ಅತಿದೀರ್ಘೋಽಪಿ ಕರ್ಣಿಕಾರೋ ನ ಮುಸಲೀ ಭವತಿ |84|
ಅತಿದೀಪ್ತೋಽಪಿ ಖದ್ಯೋತೋ ನ ಪಾವಕಃ |85|
ನ ಪ್ರವೃದ್ಧತ್ವಂ ಗುಣಹೇತುಃ |86|
ಸುಜೀರ್ಣೋಽಪಿ ಪುಚುಮಲ್ದೋ ನ ಶಂಕುಲಾಯತೇ |87|
ಯಥಾ ಬೀಜಂ ತಥಾ ನಿಷ್ಪತ್ತಿಃ |88|
ಯಥಾ ಶ್ರುತಂ ತಥಾ ಬುದ್ಧಿಃ |89|
ಯಥಾ ಕುಲಂ ತಥಾಽಽಚಾರಂ |90|
ಸಂಸ್ಕೃತಃ ಪಿಚುಮಂದೋ ನ ಸಹಕಾರೋ ಭವತಿ |91|
ನ ಚಾಗತಂ ಸುಖಂ ತ್ಯಜೇತ್ |92|
ಸ್ವಯಮೇವ ದುಃಖಮಧಿಗಚ್ಛತಿ |93|
ರಾತ್ರಿಚಾರಣಂ ನ ಕುರ್ಯಾತ್ |94|
ನ ಚಾರ್ಧರಾತ್ರಂ ಸ್ವಪೇತ್ |95|
ತದ್ವಿದ್ವಿದ್ಭಿಃ ಪರೀಕ್ಷೇತ |96|
ಪರಗೃಹಮಕಾರಣತೋ ನ ಪ್ರವಿಶೇತ್ |97|
ಜ್ಞಾತ್ವಾಽಪಿ ದೋಷಮೇವ ಕರೋತಿ ಲೋಕಃ |98|
ಶಾಸ್ತ್ರಪ್ರಧಾನಾ ಲೋಕವೃತ್ತಿಃ |99|
ಶಾಸ್ತ್ರಾಭಾವೇ ಶಿಷ್ಟಾಚಾರಮನುಗಚ್ಛೇತ್ |100|
ನಾಚರಿತಾಚ್ಛಸ್ತ್ರಂ ಗರೀಯಃ |101|
ದೂರಸ್ಥಮಪಿ ಚಾಚಕ್ಷುಃ ಪಶ್ಯತಿ ರಾಜಾ |102|
ಗತಾನುಗತಿಕೋ ಲೋಕಃ |103|
ಯಮನುಜೀವೇತ್ ತಂ ನಾಪವದೇತ್ |104|
ಇತಿ ಷಷ್ಠೋಽಧ್ಯಾಯಃ ||

ಸಪ್ತಮೋಽಧ್ಯಾಯಃ ||
ತಪಸ್ಸಾರ ಇಂದ್ರಿಯನಿಗ್ರಹಃ |1|
ದುರ್ಲಭಃ ಸ್ತ್ರೀಬಂಧನಾನ್ಮೋಕ್ಷಃ |2|
ಸ್ತ್ರೀನಾಮ ಸರ್ವಾಶುಭಾನಾಂ ಕ್ಷೇತ್ರಂ |3|
ನ ಚ ಸ್ತ್ರೀಣಾಂ ಪುರುಷಪರೀಕ್ಷಾ |4|
ಸ್ತ್ರೀಣಾಂ ಮನಃ ಕ್ಷಣಿಕಂ |5|
ಅಶುಭದ್ವೇಷಿಣಃ ಸ್ತ್ರೀಷು ನ ಪ್ರಸಕ್ತಾ ಭವೇಯುಃ |6|
ಯಜ್ಞಫಲಜ್ಞಾಸ್ತ್ರಿವೇದವಿದಃ |7|
ಸ್ವರ್ಗಸ್ಥಾನಂ ನ ಶಾಶ್ವತಂ ಅಪಿ ತು ಯಾನತ್ಪುಣ್ಯಫಲಂ |8|
ನ ಚ ಸ್ವರ್ಗಪತನಾತ್ ಪರಂ ದುಃಖಂ |9|
ದೇಹೀ ದೇಹಂ ತ್ಯಕ್ತ್ವಾ ಐಂದ್ರಪದಂ ನ ವಾಂಛತಿ |10|
ದುಃಖಾನಾಮೌಷಧಂ ನಿರ್ವಾಣಂ |11|
ಅನಾರ್ಯಸಂಬಂಧಾದ್ವರಮಾರ್ಯಶತ್ರುತಾ |12|
ನಿಹಂತಿ ದುರ್ವಚನಂ ಕುಲಂ |13|
ನ ಪುತ್ರಸಂಸ್ಪರ್ಶಾತ್ ಪರಂ ಸುಖಂ |14|
ವಿವಾದೇ ಧರ್ಮಮನುಸ್ಮರೇತ್ |15|
ನಿಶಾಂತೇ ಕಾರ್ಯಂ ಚಿಂತಯೇತ್ |16|
ಪ್ರದೋಷೇ ನ ಸಂಯೋಗಃ ಕರ್ತವ್ಯಃ |17|
ಉಪಸ್ಥಿತವಿನಾಶಃ ದುರ್ನಯಂ ಶುಭಂ ಮನ್ಯತೇ |18|
ಕ್ಷೀರಾರ್ಥಿನಃ ಕಿಂ ಕರಿಣ್ಯಾ |19|
ನ ದಾನಸಮಂ ವಶ್ಯಂ |20|
ಪರಾಯತ್ತೇಷೂತ್ಕಂಠಾಂ ನ ಕುರ್ಯಾತ್ |21|
ಅತ್ಸಮೃದ್ಧಿರಸದ್ಭರೇವ ಭುಜ್ಯತೇ |22|
ನಿಂಬಫಲಂ ಕಾಕೈರ್ಹಿ ಭುಜ್ಯತೇ |23|
ನಾಂಭೋಧಿಸ್ತೃಷ್ಣಾಮಪೋಹತಿ |24|
ವಾಲಿಕಾ ಅಪಿ ಸ್ವಗುಣಮಾಶ್ರಯಂತೇ |25|
ಸಂತೋಽಸತ್ಸು ನ ರಮಂತೇ |26|
ಹಂಸಃ ಪ್ರತವನೇ ನ ರಮತೇ |27|
ಅರ್ಥಾರ್ಥಂ ಪ್ರವರ್ತತೇ ಲೋಕಃ |28|
ಆಶಯಾ ಬಧ್ಯತೇ ಲೋಕಃ |29|
ನ ಚ ಆಶಾಪರೈಃ ಶ್ರೀ ಸಹ ತಿಷ್ಠತಿ |30|
ಆಶಾಪರೇ ನ ಧೈರ್ಯಂ |31|
ದೈನ್ಯಾನ್ಮರಣಮುತ್ತಮಂ |32|
ಆಶಾಲಜ್ಜಾಂ ವ್ಯಪೋಹತಿ |33|
ನ ಮಾತ್ರಾ ಸಹ ವಾಶಃ ಕರ್ತವ್ಯಃ |34|
ಆತ್ಮಾ ನ ಸ್ತೋತವ್ಯಃ |35|
ನ ದಿವಾ ಸ್ವಪ್ನಂ ಕುರ್ಯಾತ್ |36|
ನ ಚಾಸನ್ನಪಿ ಪಶ್ಯತ್ಯೈಶ್ವರ್ಯಾಂಧಃ ನಾಪಿ ಶ್ರುಣೋತೀಷ್ಟಂ ವಾಕ್ಯಂ |37|
ಸ್ತ್ರೀಣಾಂ ನ ಭರಿತುಃ ಪರಂ ದೈವತಂ ತದನುವರ್ತನಂ ತಾಸಾಮುಭಯಸೌಖ್ಯಂ |38|
ಅತಿಥಿಅಭ್ಯಾಗತಂ ಚ ಪೂಜಯೇದ್ಯಥಾವಿಧಿ |39|
ನಾಸ್ತಿ ಹವ್ಯಸ್ಯ ವ್ಯಾಘಾತಃ |40|
ಶತ್ರುರ್ಮಿತ್ರವತ್ ಪ್ರತಿಭಾತಿ |41|
ಮೃಗತೃಷ್ಣಾ ಜಲವದ್ಭಾತಿ ಹಿ |42|
ಇತಿ ಸಪ್ತಮೋಽಧ್ಯಾಯಃ ||

ಅಥಾಷ್ಟಮೋಽಧ್ಯಾಯಃ ||
ಧುರಮೇಧಸೋಽಸಚ್ಛಾಸ್ತ್ರಂ ಮೋಹಯತಿ |1|
ಸತ್ಸಂಗಃ ಸ್ವರ್ಗವಾಸಃ |2|
ಆರ್ಯಾಃ ಸ್ವಮಿವ ಪರಂ ಮನ್ವತೇ |3|
ರೂಪಾನುವರ್ತೀ ಗುಣಃ |4|
ಯತ್ರ ಸುಖೇನ ವರ್ತತೇ ತದೇವ ಸ್ಥಾನಂ |5|
ವಿಶ್ವಾಸಘಾತಿನೋ ನ ನಿಷ್ಕೃತಿಃ |6|
ದೈವಾಯತ್ತಂ ನ ಶೋಚೇತ್ |7|
ಆಶ್ರಿತದುಃಖಮಾತ್ಮನ ಇವ ಮನ್ಯತೇ ಸಾಧುಃ |8|
ಹೃದ್ಗತಮಾಚ್ಛಾದ್ಯಾನ್ಯದ್ವದತ್ಯನಾರ್ಯಃ |9|
ಬುದ್ಧಿಹೀನಃ ಪಿಶಾಚತುಲ್ಯಃ |10|
ಅಸಹಾಯಃ ಪಥಿ ನ ಗಚ್ಛೇತ್ |11|
ಪುತ್ರೋ ನ ಸ್ತೋತವ್ಯಃ |12|
ಸ್ವಾಮೀ ಸ್ತೋತವ್ಯೋಽನುಜೀವಿಭಿಃ |13|
ಧರ್ಮಕೃತ್ಯಾನಿ ಸರ್ವಾಣಿ ಸ್ವಾಮಿನ ಇತ್ಯೇವ ಘೋಷಯೇತ್ |14|
ರಾಜಾಜ್ಞಾಂ ನಾತಿಲಂಘಯೇತ್ |15|
ಯಥಾಽಽಜ್ಞಪ್ತಂ ತಥಾ ಕುರ್ಯಾತ್ |16|
ನಾಸ್ತಿ ಬುದ್ಧಿಮತಾಂ ಶತ್ರುಃ |17|
ಆತ್ಮಛಿದ್ರಂ ನ ಪ್ರಕಾಶಯೇತ್ |18|
ಕ್ಷಮಾವಾನೇವ ಸರ್ವಂ ಸಾಧಯತಿ |19|
ಆಪದರ್ಥಂ ಧನಂ ರಕ್ಷೇತ್ |20|
ಸಾಹಸವತಾಂ ಪ್ರಾಯಂ ಕರ್ತವ್ಯಂ |21|
ಶ್ವಃ ಕಾರ್ಯಮದ್ಯ ಕುರ್ವೀತ |22|
ಆಪರಾಹ್ಣಿಕಂ ಪೂರ್ವಾಹ್ಣ ಏವ ಕರ್ತವ್ಯಂ |23|
ವ್ಯವಹಾರಾನುಲೋಮೋ ಧರ್ಮಃ |24|
ಸರ್ವಜ್ಞತಾ ಲೋಕಜ್ಞತಾ |25|
ಶಾಸ್ತ್ರಜ್ಞೋಽಪ್ಯಲೋಕಜ್ಞೋ ಮೂರ್ಖತುಲ್ಯಃ |26|
ಶಾತ್ರಪ್ರಯೋಜನಂ ತತ್ತ್ವದರ್ಶನಂ |27|
ತತ್ತ್ವಜ್ಞಾನಂ ಕಾರ್ಯಮೇವ ಪ್ರಕಾಶಯತಿ |28|
ವ್ಯವಹಾರೇ ಪಕ್ಷಪಾತೋ ನ ಕರ್ತವ್ಯಃ |29|
ಧರ್ಮಾದಪಿ ವ್ಯವಹಾರೋ ಗರೀಯಾನ್ |30|
ಆತ್ಮಾ ಹಿ ವ್ಯವಹಾರಸ್ಯ ಸಾಕ್ಷೀ |31|
ಸರ್ವಸಾಕ್ಷೀ ಹ್ಯಾತ್ಮಾ |32|
ನ ಸ್ಯಾತ್ ಕೂಟಸಾಕ್ಷೀ |33|
ಕೂಟಸಾಕ್ಷಿಣೋ ನರಕೇ ಪತಂತಿ |34|
ಪ್ರಚ್ಛನ್ನಪಾಪಾನಾಂ ಸಾಕ್ಷಿಣೋ ಮಹಾಭೂತಾನಿ |35|
ಆತ್ಮನಃ ಪಾಪಮಾತ್ಮೈವ ಪ್ರಕಾಶಯತಿ |36|
ವ್ಯವಹಾರೋಽನ್ತರ್ಗತಮಾಕಾರಃ ಸೂಚಯತಿ |37|
ಆಕಾರಸಂವರಣಂ ದೇವಾನಾಮಪ್ಯಶಕ್ಯಂ |38|
ಚೋರರಾಜಪುರುಷೇಭ್ಯೋ ವಿತ್ತಂ ರಕ್ಷೇತ್ |39|
ದುರ್ದರ್ಶನಾ ಹಿ ರಾಜಾನಃ ಪ್ರಜಾ ರಕ್ಷಂತಿ |40|
ಸುದರ್ಶನಾ ರಾಜಾನಃ ಪ್ರಜಾ ರಕ್ಷಂತಿ |41|
ನ್ಯಾಯಯುಕ್ತಂ ರಾಜಾನಂ ಮಾತರಂ ಮನ್ಯಂತೇ ಪ್ರಜಾಃ |42|
ತಾದೃಶಃ ಸ ರಾಜಾ ಇಹ ಸುಖಂ ತತಃ ಸ್ವರ್ಗಣ್ ಚಾಪ್ನೋತಿ |43|
ಅಹಿಂಸಾಲಕ್ಷಣೋ ಧರ್ಮಃ |44|
ಸ್ವಶರೀರಮಪಿ ಪರಶರೀರಂ ಮನ್ಯತೇ ಸಾಧುಃ |45|
ಮಾಂಸಭಕ್ಷಣಮಪ್ಯುಕ್ತಂ ಸರ್ವೇಷಾಂ |46|
ನ ಸಂಸಾರಭಯಂ ಜ್ಞಾನವತಾಂ |47|
ವಿಜ್ಞಾನದೀಪೇನ ಸಂಸಾರಭಯಂ ನಿವರ್ತಯತಿ |48|
ಸರ್ವಮವಿತ್ಯಂ ಭವತಿ |49|
ಕೃಮಿಶನ್ಕೃನ್ಮೂತ್ರಭಾಚನಂ ಶರೀರಂ ಪುಣ್ಯತಯಪಾಪಜನ್ಮಹೇತುಃ |50|
ಜನ್ಮಮರಣಾದಿಷು ತು ದುಃಖಮೇವ |51|
ತಪಸಾ ಸ್ವರ್ಗಮಾಪ್ನೋತಿ |52|
