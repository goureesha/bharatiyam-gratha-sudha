|| ಉತ್ತರ ನೀರಾಜನ/ ಮಂಗಳಾರತಿ/ ಆರತಿ ಮಂತ್ರಗಳು ||

ವೇದಾ॒ಹಮೇ॒ತಂ ಪುರು॑ಷಂ ಮ॒ಹಾಂತಮ್᳚ |
ಆ॒ದಿ॒ತ್ಯವ॑ರ್ಣಂ॒ ತಮ॑ಸ॒ಸ್ತು ಪಾ॒ರೇ |
ಸರ್ವಾ॑ಣಿ ರೂ॒ಪಾಣಿ॑ ವಿ॒ಚಿತ್ಯ॒ ಧೀರಃ॑ |
ನಾಮಾ॑ನಿ ಕೃ॒ತ್ವಾಽಭಿ॒ವದ॒ನ್॒ ಯದಾಸ್ತೇ᳚ ||1||

ಈಶಾನಸ್ಸರ್ವ॑ವಿದ್ಯಾ॒ನಾ॒ಮೀಶ್ವರಸ್ಸರ್ವ॑ಭೂತಾ॒ನಾಂ॒ ಬ್ರಹ್ಮಾಧಿ॑ಪತಿ॒ಬ್ರ್ರಹ್ಮ॒ಣೋಽಧಿ॑ಪತಿ॒ಬ್ರ್ರಹ್ಮಾ॑ ಶಿ॒ವೋ ಮೇ॑ ಅಸ್ತು ಸದಾಶಿ॒ವೋಮ್ ||2||

ಶ್ರಿ॒ಯೇ ಜಾ॒ತಃ ಶ್ರಿ॒ಯ ಆನಿರಿ॑ಯಾಯ॒ ಶ್ರಿಯಂ॒ ವಯೋ᳚ ಜರಿ॒ತೃಭ್ಯೋ᳚ ದಾಧಾತಿ |
ಶ್ರಿಯಂ॒ ವಸಾ᳚ನಾ ಅಮೃತ॒ತ್ವಮಾ᳚ಯ॒ನ್ ಭವಂ᳚ತಿ ಸ॒ತ್ಯಾ ಸ॑ಮಿ॒ಥಾ ಮಿ॒ತದ್ರೌ᳚ ||3||

ಯಸ್ತೇ॑ ದೇವವರುಣ ಗಾಯ॒ತ್ರ ಛಂ॑ದಾಃ॒ ಪಾಶಃ॑ |
ತಂ ತ॑ ಏ॒ತೇ ನಾವ॑ಯಜೇ |
ಯಸ್ತೇ॑ ದೇವವರುಣ ತ್ರಿ॒ಷ್ಟುಪ್ ಛಂ॑ದಾಃ॒ ಪಾಶಃ॑ |
ತಂ ತ॑ ಏ॒ತೇ ನಾವ॑ಯಜೇ |
ಯಸ್ತೇ॑ ದೇವವರುಣ॒ ಜಗ॑ತೀಛಂದಾಃ॒ ಪಾಶಃ॑ |
ತಂ ತ॑ ಏ॒ತೇ ನಾವ॑ಯಜೇ |
ಸೋಮೋ॒ ವಾ ಏ॒ತಸ್ಯ॑ ರಾ॒ಜ್ಯಮಾದ॑ತ್ತೇ |
ಯೋ ರಾಜಾ॒ ಸನ್ರಾ॒ಜ್ಯೋವಾ॒ ಸೋಮೇ॑ನ॒ ಯಜ॑ತೇ |
ದೇ॒ವ॒ಸು॒ವಾಮೇ॒ತಾನಿ॑ ಹ॒ವೀಗ್‍ಂಷಿ॑ ಭವಂತಿ |
ಏ॒ತಾವಂ॑ತೋ॒ ವೈ ದೇ॒ವಾನಾಗ್॑‍ಂ ಸ॒ವಾಃ |
ತ ಏ॒ವಾಸ್ಮೈ॑ ಸ॒ವಾನ್‍ಪ್ರಯ॑ಚ್ಛಂತಿ |
ತ ಏ॑ನಂ॒ ಪುನ॑ಸ್ಸುವಂತೇ ರಾ॒ಜ್ಯಾಯ॑ |
ದೇ॒ವ॒ಸೂ ರಾಜಾ॑ ಭವತಿ ||4||

ಸಾಂ॒ಗ್ರ॒ಹ॒ಣ್ಯೇಷ್ಟ್ಯಾ॑ ಯಜತೇ |
ಇ॒ಮಾಂ ಜ॒ನತಾ॒ಗ್॒‍ಂ ಸಂಗೃ॑ಹ್ಣಾ॒ನೀತಿ॑ |
ದ್ವಾದ॑ಶಾರತ್ನೀ ರಶ॒ನಾ ಭ॑ವತಿ |
ದ್ವಾದ॑ಶ॒ ಮಾಸಾ᳚ಸ್ಸಂವತ್ಸ॒ರಃ |
ಸಂ॒ವ॒ತ್ಸ॒ರಮೇ॒ವಾವ॑ರುಂದೇ |
ಮೌಂ॒ಜೀ ಭ॑ವತಿ |
ಊಗ್ರ್ವೈ ಮುಂಜಾಃ᳚ |
ಊರ್ಜ॑ಮೇ॒ವಾವ॑ರುಂಧೇ |
ಚಿ॒ತ್ರಾ ನಕ್ಷ॑ತ್ರಂ ಭವತಿ |
ಚಿ॒ತ್ರಂ ವಾ ಏ॒ತತ್ಕರ್ಮ॑ |
ಯದ॑ಶ್ವಮೇ॒ಧಸ್ಸಮೃ॑ದ್ಧ್ಯೈ ||5||

ವಿ॒ರಾಜ॑ಮೇ॒ವ ತೈರಾ॒ಪ್ತ್ವಾ ಯಜ॑ಮಾ॒ನೋಽವ॑ರುಂಧೇ |
ಏಕಾ॑ದಶ ದ॒ಶತ॒ ಆಲ॑ಭ್ಯಂತೇ |
ಏಕಾ॑ದಶಾಕ್ಷರಾ ತ್ರಿ॒ಷ್ಟುಪ್ |
ತ್ರೈಷ್ಟು॑ಭಾಃ ಪ॒ಶವಃ॑ |
ಪ॒ಶೂನೇ॒ವಾವ॑ರುಂದೇ |
ವೈ॒ಶ್ವ॒ದೇ॒ವೋ ವಾ ಅಶ್ವಃ॑ |
ನಾ॒ನಾ॒ದೇ॒ವ॒ತ್ಯಾಃ᳚ ಪ॒ಶವೋ॑ ಭವಂತಿ |
ಅಷ್ವ॑ಸ್ಯ ಸರ್ವ॒ತ್ವಾಯ॑ ನಾನಾ॑ರೂಪಾ ಭವಂತಿ |
ತಸ್ಮಾ॒ನ್ನಾನಾ॑ರೂಪಾಃ ಪ॒ಶವಃ॑ |
ಬ॒ಹು॒ರೂ॒ಪಾ ಭ॑ವಂತಿ |
ತಸ್ಮಾ᳚ದ್ ಬಹುರೂ॒ಪಾಃ ಪ॒ಶವ॒ಸ್ಸಮೃ॑ದ್ಧ್ಯೈ ||6||

ಋ॒ತಂ ತ್ವಾ॑ ಸ॒ತ್ಯೇನ॒ ಪರಿ॑ಷಿಂಚಾ॒ಮೀತಿ॑ ಸಾ॒ಯಂ ಪರಿ॑ಷಿಂಚತಿ |
ಸ॒ತ್ಯಂ ತ್ವ॒ರ್ತೇನ॒ ಪರಿ॑ಷಿಂಚಾ॒ಮೀತಿ॑ ಪ್ರಾ॒ತಃ |
ಅ॒ಗ್ನಿರ್ವಾ ಋ॒ತಮ್ |
ಅ॒ಸಾವಾ॑ದಿ॒ತ್ಯಸ್ಸ॒ತ್ಯಮ್ |
ಅ॒ಗ್ನಿಮೇ॒ವ ತದಾ॑ದಿ॒ತ್ಯೇನ॑ ಸಾ॒ಯಂ ಪರಿ॑ಷಿಂಚತಿ |
ಅ॒ಗ್ನಿನಾ॑ದಿ॒ತ್ಯಂ ಪ್ರಾ॒ತಸ್ಸಹಃ |
ಯಾವ॑ದಹೋರಾ॒ತ್ರೇ ಭವ॑ತಃ |
ತಾವ॑ದಸ್ಯ ಲೋ॒ಕಸ್ಯ॑ |
ನಾರ್ತಿ॒ರ್ನ ರಿಷ್ಟಿಃ॑ |
ನಾಂತೋ॒ ನ ಪ॑ರ್ಯಂ॒ತೋ᳚ಽಸ್ತಿ |
ಯಸ್ಯೈ॒ವಂ ವಿ॒ದುಷೋ᳚ಽಗ್ನಿಹೋ॒ತ್ರಂ ಜುಹ್ವ॑ತಿ |
ಯ ಉ॑ ಚೈನದೇ॒ವಂ ವೇದ॑ ||7||

ನರ್ಯ॑ ಪ್ರ॒ಜಾಂ ಮೇ॑ ಗೋಪಾಯ |
ಅ॒ಮೃ॒ತ॒ತ್ವಾಯ॑ ಜೀ॒ವಸೇ᳚ |
ಜಾ॒ತಾಂ ಜ॑ನಿ॒ಷ್ಯಮಾ॑ಣಾಂ ಚ |
ಅ॒ಮೃತೇ॑ ಸ॒ತ್ಯೇ ಪ್ರತಿ॑ಷ್ಠಿತಾಮ್ |
ಅಥ॑ರ್ವ ಪಿ॒ತುಂ ಮೇ॑ ಗೋಪಾಯ |
ರಸ॒ಮನ್ನ॑ಮಿ॒ಹಾಯು॑ಷೇ |
ಅದ॑ಬ್ಧಾ॒ಯೋಽಶೀ॑ತತನೋ |
ಅವಿ॑ಷಂ ನಃ ಪಿ॒ತುಂ ಕೃ॑ಣು |
ಶಗ್ಗ್ಸ್ಯ॑ ಪ॒ಶೂನ್ಮೇ॑ ಗೋಪಾಯ |
ದ್ವಿ॒ಪಾದೋ॒ ಯೇ ಚತು॑ಷ್ಪದಃ |
ಅ॒ಷ್ಟಾಶ॑ಫಾಶ್ಚ॒ ಯ ಇ॒ಹಾಗ್ನೇ᳚ |
ಯೇ ಚೈಕ॑ಶಫಾ ಆಶು॒ಗಾಃ |
ಸಪ್ರ॑ಥ ಸ॒ಭಾಂ ಮೇ॑ ಗೋಪಾಯ |
ಯೇ ಚ॒ ಸಭ್ಯಾ᳚ಸ್ಸಭಾ॒ಸದಃ॑ |
ತಾನಿಂ॑ದ್ರಿ॒ಯಾವ॑ತಃ ಕುರು |
ಸರ್ವ॒ಮಾಯು॒ರುಪಾ॑ಸತಾಮ್ |
ಅಹೇ॑ ಬುಧ್ನಿಯ॒ ಮಂತ್ರಂ॑ ಮೇ ಗೋಪಾಯ |
ಯಮೃಷ॑ಯಸ್ತ್ರೈವಿ॒ದಾ ವಿ॒ದುಃ |
ಋಚ॒ಸ್ಸಾಮಾ॑ನಿ॒ ಯಜೂಗ್॑‍ಂಷಿ |
ಸಾ ಹಿ ಶ್ರೀರ॒ಮೃತಾ॑ ಸ॒ತಾಮ್ ||8||

ತಸ್ಮಾ᳚ದ್ವಾಜಪೇಯಯಾ॒ಜೀ ಪೂ॒ತೋ ಮೇಧ್ಯೋ॑ ದಕ್ಷಿ॒ಣ್ಯಃ॑ |
ಪ್ರಾಙುದ್ದ್ರ॑ವತಿ ಸೋಮಗ್ರ॒ಹೈಃ |
ಅ॒ಮುಮೇ॒ವ ತ್ರೈರ್ಲೋ॒ಕಮ॒ಭಿ ಜ॑ಯತಿ |
ಪ್ರತಿ॑ಷ್ಠಂತಿ ಸೋಮಗ್ರ॒ಹೈಃ |
ಯಾವ॑ದೇ॒ವ ಸ॒ತ್ಯಮ್ |
ತೇನ॑ ಸೂಯತೇ |
ವಾ॒ಜ॒ಸೃದ್ಭ್ಯ॑ಸ್ಸುರಾಗ್ರ॒ಹಾನ್ ಹ॑ರಂತಿ |
ಅನೃ॑ತೇನೈ॒ವ ವಿಶ॒ಗ್॒‍ಂ ಸಗ್‍ಂ ಸೃ॑ಜತಿ |
ಹಿ॒ರ॒ಣ್ಯ॒ಪಾ॒ತ್ರಂ ಮಧೋ᳚ ಪೂ॒ರ್ಣಂ ದ॑ದಾತಿ |
ಮ॒ಧ॒ವ್ಯೋ॑ಽಸಾ॒ನೀತಿ॑ |
ಏ॒ಕ॒ಧಾ ಬ್ರ॒ಹ್ಮಣ॒ ಉಪ॑ ಹರತಿ |
ಏ॒ಕ॒ಧೈವ ಯಜ॑ಮಾನ॒ ಆಯು॒ಸ್ತೇಜೋ॑ ದಧಾತಿ ||9||

ಆಶಾ᳚ಸ್ತೇ॒ಽಯಂ ಯಜ॑ಮಾನೋ॒ಽಸೌ |
ಆಯು॒ರಾಶಾ᳚ಸ್ತೇ |
ಸು॒ಪ್ರ॒ಜಾ॒ಸ್ತ್ವಮಾಶಾ᳚ಸ್ತೇ |
ಸ॒ಜಾ॒ತ॒ವ॒ನ॒ಸ್ಯಾಮಾಶಾ᳚ಸ್ತೇ |
ಉತ್ತ॑ರಾಂ ದೇವಯ॒ಜ್ಯಾಮಾಶಾ᳚ಸ್ತೇ |
ಭೂಯೋ॑ ಹವಿ॒ಷ್ಕರ॑ಣ॒ಮಾಶಾ᳚ಸ್ತೇ |
ದಿ॒ವ್ಯಂ ಧಾಮಾಶಾ᳚ಸ್ತೇ |
ವಿಶ್ವಂ॑ ಪ್ರಿ॒ಯಮಾಶಾ᳚ಸ್ತೇ |
ಯದ॒ನೇನ॑ ಹ॒ವಿಷಾಶಾ᳚ಸ್ತೇ |
ತದ॑ಶ್ಯಾ॒ತ್ತದೃ॑ದ್ಧ್ಯಾತ್ |
ತದ॑ಸ್ಮೈ ದೇ॒ವಾ ರಾ॑ಸಂತಾಮ್ |
ತದ॒ಗ್ನಿರ್ದೇ॒ವೋ ದೇ॒ವೇಭ್ಯೋ॒ ವನ॑ತೇ |
ವ॒ಯಮ॒ಗ್ನೇರ್ಮಾನು॑ಷಾಃ |
ಇ॒ಷ್ಟಂ ಚ॑ ವೀ॒ತಂ ಚ॑ |
ಉ॒ಭೇ ಚ॑ ನೋ॒ ದ್ಯಾವಾ॑ಪೃಥಿ॒ವೀ ಅಗ್‍ಂಹ॑ಸಸ್ಪಾತಾಮ್ |
ಇ॒ಹ ಗತಿ॑ರ್ವಾ॒ಮಸ್ಯೇ॒ದಂಚ॑ |
ನಮೋ॑ ದೇ॒ವೇಭ್ಯಃ॑ ||10||

ತೇನ॒ಸೋ᳚ಽಸ್ಯಾ॒ಭೀಷ್ಟಃ॑ ಪ್ರೀ॒ತಃ |
ಯಥ್ಸ॒ಭಾಯಾಂ᳚ ವಿ॒ಜಯಂ॑ತೇ |
ತೇನ॒ ಸೋ᳚ಽಸ್ಯಾ॒ಭೀಷ್ಟಃ॑ ಪ್ರೀ॒ತಃ |
ಯದಾ॑ವಸ॒ಥೇಽನ್ನ॒ಗ್॒‍ಂ ಹರಂ॑ತಿ |
ತೇನ॒ ಸೋ᳚ಽಸ್ಯಾ॒ಭೀಷ್ಟಃ॑ ಪ್ರೀ॒ತಃ |
ತಥಾ᳚ಽಸ್ಯ॒ ಸರ್ವೇ᳚ ಪ್ರೀ॒ತಾ ಅ॒ಭೀಷ್ಟಾ॒ ಅಧೀ॑ಯಂತೇ |
ಪ್ರ॒ವ॒ಸ॒ಥಮೇ॒ಷ್ಯನ್ನೇ॒ವಮುಪ॑ತಿಷ್ಠೇ॒ತೈಕ॑ಮೇಕಮ್ |
ಯಥಾ᳚ ಬ್ರಾಹ್ಮ॒ಣಾಯ॑ ಗೃಹೇವಾ॒ಸಿನೇ॑ ಪರಿ॒ದಾಯ॑ ಗೃ॒ಹಾನೇತಿ॑ |
ತಾ॒ದೃಗೇ॒ವ ತತ್ |
ಪುನ॑ರಾ॒ಗತ್ಯೋಪ॑ತಿಷ್ಠತೇ |
ಸಾ ಭಾ॑ಗೇಯಮೇ॒ ವೈಷಾಂ॒ ತತ್ |
ಸಾ ತತ॑ ಊ॒ಧ್ರ್ವಾ ರೋ॑ಹತ್ |
ಸಾ ರೋ॑ಹಿ॒ಣ್ಯ॑ಭವತ್ |
ತದ್ರೋ॑ಹಿ॒ಣ್ಯೈ ರೋ॑ಹಿಣಿ॒ತ್ವಮ್ |
ರೋ॒ಹಿ॒ಣ್ಯಾಮ॒ಗ್ನಿಮಾದ॑ಧೀತ |
ಸ್ವ ಏ॒ವೈನಂ॒ ಯೋನೌ॒ ಪ್ರತಿ॑ಷ್ಠಿತ॒ಮಾಧ॑ತ್ತೇ |
ಋ॒ಧ್ನೋತ್ಯೇ॑ನೇನ ||11||

ತ್ರ್ಯಂ॑ಬಕಂ ಯಜಾಮಹ॒ ಇತ್ಯಾ॑ಹ |
ಮೃ॒ತ್ಯೋರ್ಮು॑ಕ್ಷೀಯ॒ ಮಾಽಮೃತಾ॒ದಿತಿ॒ ವಾವೈತದಾ॑ಹ |
ಉತ್ಕಿ॑ರಂತಿ |
ಭಗ॑ಸ್ಯ ಲೀಪ್ಸಂತೇ |
ಮೂತೇ॑ ಕೃ॒ತ್ವಾಽಽಸ॑ಜಂತಿ |
ಯಥಾ॒ ಜನಂ॑ ಯ॒ತೇ॑ಽವ॒ಸಂ ಕ॒ರೋತಿ॑ |
ತಾ॒ದೃಗೇ॒ವ ತತ್ |
ಏ॒ಷ ತೇ॑ ರುದ್ರಭಾ॒ಗ ಇತ್ಯಾ॑ಹ ನಿ॒ರವ॑ತ್ತ್ಯೈ |
ಅಪ್ರ॑ತಿಖ್ಷ॒ಮಾ ಯಂ॑ತಿ |
ಅ॒ಪಃ ಪರಿ॑ಷಿಂಚಂತಿ |
ರು॒ದ್ರಸ್ಯಾಂ॒ತರ್ಹಿ॑ತ್ಯೈ |
ಪ್ರ ವಾ ಏ॒ತೇ᳚ಽಸ್ಮಾಲ್ಲೋ॒ಕಾಚ್ಚ್ಯ॑ವಂತೇ |
ಯೇ ತ್ರ್ಯಂ॑ಬಕೈ॒ಶ್ಚರಂ॑ತಿ |
ಆ॒ದಿ॒ತ್ಯಂ ಚ॒ರುಂ ಪುನ॒ರೇತ್ಯ॒ ನಿರ್ವ॑ಪತಿ |
ಇ॒ಯಂ ವಾ ಅದಿ॑ತಿಃ |
ಅ॒ಸ್ಯಾಮೇ॒ವ ಪ್ರತಿ॑ ತಿಷ್ಠಂತಿ ||12||

ಐಂ॒ದ್ರಾ॒ಗ್ನಮುಪ॑ ಸಾದಿತಮ್ |
ಸರ್ವಾ᳚ಭ್ಯೋ॒ ವಾ ಏ॒ಷ ದೇ॒ವತಾ᳚ಭ್ಯೋ ಜುಹೋತಿ |
ಯೋ᳚ಽಗ್ನಿಹೋ॒ತ್ರಂ ಜು॒ಹೋತಿ॑ |
ಯಥಾ॒ ಖಲು॒ ವೈ ಧೇ॒ನುಂತೀ॒ರ್ಥೇ ತ॒ರ್ಪಯ॑ತಿ |
ಏ॒ವಮ॑ಗ್ನಿಹೋ॒ತ್ರೀ ಯಜ॑ಮಾನಂ ತರ್ಪಯತಿ |
ತೃಪ್ಯ॑ತಿ ಪ್ರ॒ಜಯಾ॑ ಪ॒ಶುಭಿಃ॑ |
ಪ್ರಸು॑ವ॒ರ್ಗಲ್ಲೋ॒ಕಂಜಾ॑ನಾತಿ |
ಪಶ್ಯ॑ತಿ ಪು॒ತ್ರಮ್ |
ಪಶ್ಯ॑ತಿ॒ ಪೌತ್ರ᳚ಮ್ |
ಪ್ರ ಪ್ರ॒ಜಯಾ॑ ಪ॒ಶುಭಿ॑ರ್ಮಿಥು॒ನೈರ್ಜಾ॑ಯತೇ |
ಯಸ್ಯೈ॒ವಂ ವಿ॒ದುಷೋ᳚ಽಗ್ನಿಹೋ॒ತ್ರಂ ಜುಹ್ವ॑ತಿ |
ಯ ಉ॑ ಚೈನದೇ॒ವಂ ವೇದ॑ ||13||

ಬ್ರ॒ಹ್ಮವಾ॒ದಿನೋ॑ ವದಂತಿ |
ಕಸ್ಮಾ᳚ಥ್ಸ॒ತ್ಯಾತ್ |
ಅ॒ಸ್ಮಿಲ್ಲೋ॒ಕೇ ಬ॒ಹವಃ॒ ಕಾಮಾ॒ ಇತಿ॑ |
ಯಥ್ಸ॑ಮಾ॒ನೀಭ್ಯೋ॑ ದೇ॒ವತಾ᳚ಭ್ಯೋ॒ಽನ್ಯೇ᳚ಽನ್ಯೇ ಪ॒ಶವ॑ ಆಲ॒ಭ್ಯಂತೇ᳚ |
ಅ॒ಸ್ಮಿನ್ನೇ॒ವ ತಲ್ಲೋ॒ಕೇ ಕಾಮಾಂ᳚ದಧಾತಿ |
ತಸ್ಮಾ॑ದ॒ಸ್ಮಿಲ್ಲೋ॒ಕೇ ಬ॒ಹವಃ॒ ಕಾಮಾಃ᳚ |
ತ್ರ॒ಯಾ॒ಣಾಂ ತ್ರ॑ಯಾಣಾಗ್‍ಂ ಸ॒ಹ ವ॒ಪಾ ಜು॑ಹೋತಿ |
ತ್ರ್ಯಾ॑ವೃತೋ॒ ವೈ ದೇ॒ವಾಃ |
ತ್ರ್ಯಾ॑ವೃತ ಇ॒ಮೇ ಲೋ॒ಕಾಃ |
ಏ॒ಷಾಂ ಲೋ॒ಕಾನಾ॒ಮಾಪ್ತ್ಯೈ᳚ |
ಏ॒ಷಾಂ ಲೋ॒ಕಾನಾಂ॒ ಕ್ಲೃಪ್ತ್ಯೈ᳚ |
ಪರ್ಯ॑ಗ್ನಿಕೃತಾನಾರ॒ಣ್ಯಾನುಥ್ಸೃ॑ಜಂ॒ತ್ಯಹಿಗ್॑‍ಂಸಾಯೈ ||14||

ಯದಿ॒ದಂ ಕಿಂಚ॑ |
ಯ ಏ॒ವಂ ವೇದ॑ |
ಕಲ್ಪ॑ತೇಽಸ್ಮೈ |
ಸ ವಾ ಅ॒ಯಂ ಮ॑ನು॒ಷ್ಯೇ॑ಷು ಯ॒ಜ್ಞಸ್ಸ॒ಪ್ತಹೋ॑ತಾ |
ಅ॒ಮುತ್ರ॑ ಸ॒ದ್ಭ್ಯೋ ದೇ॒ವೇಭ್ಯೋ॑ ಹ॒ವ್ಯಂ ವ॑ಹತಿ |
ಯ ಏ॒ವಂ ವೇದ॑ |
ಉಪೈ॑ನಂ ಯ॒ಜ್ಞೋ ನ॑ಮತಿ |
ಯೋ ವೈ ಚತು॑ರ್ಹೋತೃಣಾಂನಿ॒ದಾನಂ॒ ವೇದ॑ |
ನಿ॒ದಾನ॑ವಾನ್ಭವತಿ |
ಅ॒ಗ್ನಿ॒ಹೋ॒ತ್ರಂ ವೈ ದಶ॑ಹೋತುರ್ನಿ॒ದಾನ᳚ಮ್ |
ದ॒ರ್ಶ॒ಪೂ॒ರ್ಣ॒ಮಾ॒ಸೌ ಚತು॑ರ್‍ಹೋತುಃ |
ಚಾ॒ತು॒ರ್ಮಾ॒ಸ್ಯಾನಿ॒ ಪಂಚ॑ಹೋತುಃ |
ಪ॒ಶು॒ಬಂ॒ಧಷ್ಷಡ್ಡೋ॑ತುಃ |
ಸೌ॒ಮ್ಯೋ᳚ಽಧ್ವ॒ರಸ್ಸ॒ಪ್ತಹೋ॑ತುಃ |
ಏ॒ತದ್ವೈ ಚತು॑ರ್‍ಹೋತೃಣಾಂನಿ॒ದಾನ᳚ಮ್ |
ಯ ಏ॒ವಂ ವೇದ॑ |
ನಿ॒ದಾನ॑ವಾನ್‍ಭವತಿ ||15||

ಪ್ರತ್ಯೇ॒ವ ತಿ॑ಷ್ಠತಿ |
ಬ್ರ॒ಹ್ಮ॒ವಾ॒ದಿನೋ॑ ವದಂತಿ |
ದಶ॑ಹೋತಾ॒ ಚತು॑ರ್ ಹೋತಾ |
ಪಂಚ॑ಹೋತಾ॒ ಷಡ್ಡೋ॑ತಾ ಸ॒ಪ್ತಹೋ॑ತಾ |
ಅಥ॒ ಕಸ್ಮಾ॒ಚ್ಚತು॑ರ್‍ಹೋತಾರ ಉಚ್ಯಂತ॒ ಇತಿ॑ |
ಇಂದ್ರೋ॒ ವೈ ಚತು॑ರ್‍ಹೋತಾ |
ಇಂದ್ರಃ॒ ಖಲು॒ ವೈ ಶ್ರೇಷ್ಠೋ॑ ದೇ॒ವತಾ॑ನಾಮುಪ॒ದೇಶ॑ನಾತ್ |
ಯ ಏ॒ವಮಿಂದ್ರ॒ಗ್ಗ್‍ಶ್ರೇ॒ಷ್ಠಂ॑ ದೇ॒ವತಾ॑ನಾಮುಪ॒ದೇಶ॑ನಾ॒ದ್ವೇದ॑ |
ವಸಿ॑ಷ್ಠಸ್ಸಮಾ॒ನಾನಾಂ᳚ ಭವತಿ |
ತಸ್ಮಾ॒ಚ್ಛ್ರೇಷ್ಠ॑ಮಾ॒ಯಂತಂ॑ ಪ್ರಥ॒ಮೇನೈ॒ವಾನು॑ ಬುಧ್ಯಂತೇ |
ಅ॒ಯಮಾಗನ್॑ |
ಅ॒ಯಮವಾ॑ಸಾ॒ದಿತಿ॑ |
ಕೀ॒ರ್ತಿರ॑ಸ್ಯ॒ ಪೂರ್ವಾಗ॑ಚ್ಛತಿ ಜ॒ನತಾ॑ಮಾಯ॒ತಃ |
ಅಥೋ॑ ಏನಂ ಪ್ರಥ॒ಮೇನೈ॒ವಾನು॑ ಬುಧ್ಯಂತೇ |
ಅ॒ಯಮಾಗನ್॑ |
ಅ॒ಯಮವಾ॑ಸಾ॒ದಿತಿ॑ ||16||

ಯದ್ವಾ ಇ॒ದಂ ಕಿಂಚ॑ |
ತಥ್ಸರ್ವಂ॒ ಚತು॑ರ್ ಹೋತಾರಃ |
ಚತು॑ರ್‍ಹೋತೃ॒ಭ್ಯೋಧಿ॑ ಯ॒ಜ್ಞೋ ನಿರ್ಮಿ॑ತಃ |
ಸ ಯ ಏ॒ವಂ ವಿ॒ದ್ವಾನ್ ವಿ॒ವದೇ॑ತ |
ಅ॒ಹಮೇ॒ವ ಭೂಯೋ॑ ವೇದ |
ಯಶ್ಚತು॑ರ್‍ಹೋತ್॒ಚನ್॒ ವೇದೇತಿ॑ |
ಸ ಹ್ಯೇ॑ವ ಭೂಯೋ॒ ವೇದ॑ |
ಯಶ್ಚತು॑ರ್‍ಹೋತ್॒ಚನ್॒ ವೇದ॑ |
ಯೋ ವೈ ಚತು॑ರ್‍ಹೋತೃಣಾ॒ಗ್॒‍ಂ ಹೋತ್॒ಚನ್॒ ವೇದ॑ |
ಸರ್ವಾ॑ಸು ಪ್ರ॒ಜಾಸ್ವನ್ನ॑ಮತ್ತಿ ||
ಸರ್ವಾ॒ ದಿಶೋ॒ಽಭಿ ಜ॑ಯತಿ ||17||

ಸರ್ವಾ॒ ದಿಶೋ॒ಽಭಿ ಜ॑ಯತಿ |
ಪ್ರಜಾಪ॑ತಿ॒ರ್ವೈ ದಶ॑ಹೋತೃಣಾ॒ಗ್॒‍ಂ ಹೋತಾ᳚ |
ಸೋಮ॒ಶ್ಚತು॑ರ್‍ಹೋತೃಣಾ॒ಗ್॒‍ಂ ಹೋತಾ᳚ |
ಅ॒ಗ್ನಿಃ ಪಂಚ॑ಹೋತೃಣಾ॒ಗ್॒‍ಂ ಹೋತಾ᳚ |
ಧಾ॒ತಾ ಷಡ್ಡೋ॑ತೃಣಾ॒ಗ್॒‍ಂ ಹೋತಾ᳚ |
ಅ॒ರ್ಯ॒ಮಾ ಸ॒ಪ್ತಹೋ॑ತೃಣಾ॒ಗ್॒‍ಂ ಹೋತಾ᳚ |
ಏ॒ತೇ ವೈ ಚತು॑ರ್‍ಹೋತೃಣಾ॒ಗ್॒‍ಂ ಹೋತಾ॑ರಃ |
ತಾನ್ ಯ ಏ॒ವಂ ವೇದ॑ |
ಸರ್ವಾ॑ಸು ಪ್ರ॒ಜಾಸ್ವನ್ನ॑ಮತ್ತಿ |
ಸರ್ವಾ॒ ದಿಶೋ॒ಽಭಿ ಜ॑ಯತಿ ||18||

ವಾಗ್ದೇ॒ವೀ ಜು॑ಷತಾಮಿ॒ದಗ್‍ಂ ಹ॒ವಿಃ |
ಚಖ್ಷು॑ರ್ದೇ॒ವಾನಾಂ॒ ಜ್ಯೋತಿ॑ರ॒ಮೃತೇ॒ ನ್ಯ॑ಕ್ತಮ್ |
ಅ॒ಸ್ಯ ವಿ॒ಜ್ಞಾನಾ॑ಯ ಬಹು॒ಧಾ ನಿಧೀ॑ಯತೇ |
ತಸ್ಯ॑ ಸು॒ಮ್ನಮ॑ಶೀಮಹಿ |
ಮಾ ನೋ॑ ಹಾಸೀದ್ವಿಚಖ್ಷ॒ಣಮ್ |
ಆಯು॒ರಿನ್ನಃ॒ ಪ್ರತೀ᳚ರ್ಯತಾಮ್ |
ಅನಂ॑ಧಾ॒ಶ್ಚಖ್ಷು॑ಷಾ ವ॒ಯಮ್ |
ಜೀ॒ವಾ ಜ್ಯೋತಿ॑ರಶೀಮಹಿ |
ಸುವ॒ಜ್ರ್ಯೋತಿ॑ರು॒ತಾಮೃತ᳚ಮ್ |
ಶ್ರೋತ್ರೇ॑ಣ ಭ॒ದ್ರಮು॒ತ ಶೃ॑ಣ್ವಂತಿ ಸ॒ತ್ಯಮ್ |
ಶ್ರೋತ್ರೇ॑ಣ॒ ವಾಚಂ॑ ಬಹು॒ಧೊದ್ಯಮಾ॑ನಾಮ್ |
ಶ್ರೋತ್ರೇ॑ಣ॒ ಮೋದ॑ಶ್ಚ॒ ಮಹ॑ಶ್ಚ ಶ್ರೂಯತೇ |
ಶ್ರೋತ್ರೇ॑ಣ॒ ಸರ್ವಾ॒ ದಿಶ॒ ಆಶೃ॑ಣೋಮಿ |
ಯೇನ॒ ಪ್ರಾಚ್ಯಾ॑ ಉ॒ತ ದ॑ಖ್ಷಿ॒ಣಾ |
ಪ್ರ॒ತೀಚ್ಚೈ॑ ದಿ॒ಶಶ್ಶೃ॒ಣ್ವಂತ್ಯು॑ತ್ತ॒ರಾತ್ |
ತದಿಚ್ಛ್ರೋತ್ರಂ॑ ಬಹು॒ಧೋದ್ಯಮಾ॑ನಮ್ |
ಅ॒ರಾನ್ನ ನೇ॒ಮಿಃ ಪರಿ॒ ಸರ್ವಂ॑ ಬಭೂವ ||19||

ತಂ ಕಾ॒ಲೇಕಾ॑ಲ॒ ಆಗ॑ತೇ ಯಜತೇ |
ಬ್ರ॒ಹ್ಮ॒ವಾ॒ದಿನೋ॑ ವದಂತಿ |
ಸತ್ವಾ ಅ॑ಧ್ವ॒ರ್ಯುಸ್ಸ್ಯಾ᳚ತ್ |
ಯೋ ಯತೋ॑ ಯ॒ಜ್ಞಂ ಪ್ರ॑ಯುಂ॒ಕ್ತೇ |
ತದೇ॑ನಂ ಪ್ರತಿಷ್ಠಾ॒ಪಯ॒ತೀತಿ॑ |
ವಾತಾ॒ದ್ವಾ ಅ॑ಧ್ವ॒ರ್ಯುರ್ಯ॒ಜ್ಞಂ ಪ್ರಯುಂ॑ಕ್ತೇ |
ದೇವಾ॑ಗಾತುವಿದೋ ಗಾ॒ತುಂ ವಿ॒ತ್ವಾಗಾ॒ತುಮಿತೇತ್ಯಾ॑ಹ |
ಯತ॑ ಏ॒ವ ಯ॒ಜ್ಞಂ ಪ್ರ॑ಯುಂ॒ಕ್ತೇ |
ತದೇ॑ನಂ॒ ಪ್ರತಿ॑ಷ್ಠಾಪಯತಿ |
ಪ್ರತಿ॑ತಿಷ್ಠತಿ ಪ್ರ॒ಜಯಾ॑ ಪ॒ಶುಭಿ॑ರ್ಯಜ॑ಮಾನಃ ||20||

ಸರ್ವೇ᳚ಭ್ಯಸ್ತ್ವಾ ದೇ॒ವೇಭ್ಯ॒ ಇತ್ಯು॒ಪರಿ॑ಷ್ಟಾತ್ |
ಸರ್ವೇ॒ ವೈ ದೇ॒ವಾಸ್ತ್ವಿಷಿ॑ಮಂತೋ ಹರ॒ಸ್ವಿನಃ॑ |
ತ್ವಿಷಿ॑ಮೇ॒ವಾಸ್ಮ್॒ನ್॒ ಹರೋ॑ ದಧಾತಿ |
ತಸ್ಮಾ॒ದಶ್ವಃ॑ ಪಶೂ॒ನಾಂ ತ್ವಿಷಿ॑ಮಾನ್ ಹರ॒ಸ್ವಿತ॑ಮಃ |
ದಿ॒ವೇ ತ್ವಾ॒ಽಂತರಿ॑ಖ್ಷಾಯ ತ್ವಾ ಪೃಥಿ॒ವ್ಯೈ ತ್ವೇತ್ಯಾ॑ಹ |
ಏ॒ಭ್ಯ ಏ॒ವೈನಂ॑ ಲೋ॒ಕೇಭ್ಯಃ॒ ಪ್ರೋಖ್ಷ॑ತಿ |
ಸ॒ತೇ ತ್ವಾಽಸ॑ತೇ ತ್ವಾ॒ಽದ್ಭ್ಯಸ್ತ್ವೌಷ॑ಧೀಭ್ಯಸ್ತ್ವಾ॒ ವಿಶ್ವೇ᳚ಭ್ಯಸ್ತ್ವಾ ಭೂ॒ತೇಭ್ಯ॒ ಇತ್ಯಾ॑ಹ |
ತಸ್ಮಾ॑ದಶ್ವಮೇಧಯಾ॒ಜಿನ॒ಗ್॒‍ಂ ಸರ್ವಾ॑ಣಿ ಭೂ॒ತಾನ್ಯುಪ॑ಜಿವಂತಿ |
ಬ್ರ॒ಹ್ಮ॒ವಾ॒ದಿನೋ॑ ವದಂತಿ |
ಯತ್ಪ್ರಾ॑ಜಾಪ॒ತ್ಯೋಶ್ವಃ॑ |
ಅಥ॒ ಕಸ್ಮಾ॑ದೇನಮ॒ನ್ಯಾಭ್ಯೋ॑ ದೇ॒ವತಾ॒ಭ್ಯೋಪಿ॒ ಪ್ರೋಖ್ಷ॒ತೀತಿ॑ |
ಅಶ್ವೇ॒ ವೈ ಸರ್ವಾ॑ ದೇ॒ವತಾ॑ ಅ॒ನ್ವಾಯ॑ತ್ತಾಃ |
ತಂ ಯದ್ವಿಶ್ವೇ᳚ಭ್ಯಸ್ತ್ವಾ ಭೂ॒ತೇಭ್ಯ॒ ಇತಿ॑ ಪ್ರೋ॒ಖ್ಷತಿ॑ |
ದೇ॒ವತಾ॑ ಏ॒ವಾಸ್ಮಿ॑ನ್ನ॒ನ್ವಾಯಾ॑ತಯತಿ |
ತಸ್ಮಾ॒ದಶ್ವೇ॒ ಸರ್ವಾ॑ ದೇ॒ವತಾ॑ ಅ॒ನ್ವಾಯ॑ತ್ತಾಃ ||21||

ಪ್ರ॒ಜಾಪ॑ತಿರಕಾಮಯತ ಮ॒ಹಾನ॑ನ್ನಾ॒ದಸ್ಸ್ಯಾ॒ಮಿತಿ॑ |
ಸ ಏ॒ತಾವ॑ಶ್ವಮೇ॒ಧೇ ಮ॑ಹಿ॒ಮಾನಾ॑ವಪಶ್ಯತ್ |
ತಾವ॑ಗೃಹ್ಣೀತ |
ತತೋ॒ ವೈ ಸ ಮ॒ಹಾನ॑ನ್ನಾ॒ದೋ॑ಽಭವತ್ |
ಯಃ ಕಾ॒ಮಯೇ॑ತ ಮ॒ಹಾನ॑ನ್ನಾ॒ದಸ್ಸ್ಯಾ॒ಮಿತಿ॑ |
ಸ ಏ॒ತಾವ॑ಶ್ವಮೇ॒ದೇ ಮ॑ಹಿ॒ಮಾನೌ॑ ಗೃಹ್ಣೀತ |
ಮ॒ಹಾನೇ॒ವಾನ್ನಾ॒ದೋ ಭ॑ವತಿ |
ಯ॒ಜ॒ಮಾ॒ನ॒ದೇ॒ವ॒ತ್ಯಾ॑ ವೈ ವ॒ಪಾ |
ರಾಜಾ॑ ಮಹಿ॒ಮಾ |
ಯದ್ವ॒ಪಾಂ ಮ॑ಹಿ॒ಮ್ನೋಭ॒ಯತಃ॑ ಪರಿ॒ಯಜ॑ತಿ |
ಯಜ॑ಮಾನಮೇ॒ವ ರಾ॒ಜ್ಯೇನೋ॑ಭ॒ಯತಃ॒ ಪರಿ॑ಗೃಹ್ಣಾತಿ |
ಪು॒ರಸ್ತಾ᳚ಥ್ಸ್ವಾಹಾಕಾರಾ॒ ವಾ ಅ॒ನ್ಯೇ ದೇ॒ವಾಃ |
ಉ॒ಪರಿ॑ಷ್ಟಾಥ್ಸ್ವಾಹಾಕಾರಾ ಅ॒ನ್ಯೇ |
ತೇ ವಾ ಏ॒ತೇಽಶ್ವ॑ ಏ॒ವಮೇಧ್ಯ॑ ಉ॒ಭಯೇಽವ॑ರುಧ್ಯಂತೇ |
ಯದ್ವ॒ಪಾಂಮ॑ಹಿ॒ಮ್ನೋಭ॒ಯತಃ॑ ಪರಿ॒ಯಜ॑ತಿ |
ತಾನೇ॒ವೋಭಯಾ᳚ನ್‍ಪ್ರೀಣಾತಿ ||22||

ಪರ್ಯಾ᳚ಪ್ತ್ಯಾ॒ ಅನಂ॑ತರಾಯಾಯ॒ ಸರ್ವ॑ಸ್ತೋಮೋಽತಿ ರಾ॒ತ್ರ ಉ॑ತ್ತ॒ಮ ಮಹ॑ರ್ಭವತಿ॒ ಸರ್ವ॒ಸ್ಯಾಪ್ತ್ಯೈ॒ ಸರ್ವ॑ಸ್ಯ॒ ಜಿತ್ಯೈ॒ ಸರ್ವ॑ಮೇ॒ವ ತೇನಾ᳚ಪ್ನೋತಿ॒ ಸರ್ವಂ॑ ಜಯತಿ ||23||

ನಮ॑ಸ್ತೇ ಅಸ್ತು ಭಗವನ್ವಿಶ್ವೇಶ್ವ॒ರಾಯ॑ ಮಹಾದೇ॒ವಾಯ॑ ತ್ರ್ಯಂಬ॒ಕಾಯ॑ ತ್ರಿಪುರಾಂತ॒ಕಾಯ॑ ತ್ರಿಕಾಗ್ನಿಕಾ॒ಲಾಯ॑ ಕಾಲಾಗ್ನಿರು॒ದ್ರಾಯ॑ ನೀಲಕಂ॒ಠಾಯ॑ ಮೃತ್ಯುಂಜ॒ಯಾಯ॑ ಸರ್ವೇಶ್ವ॒ರಾಯ॑ ಸದಾಶಿ॒ವಾಯ॑ ಶ್ರೀಮನ್ಮಹಾದೇ॒ವಾಯ॒ ನಮಃ॑ ||24||

ಕರ್ಪೂರಾದಿ ವೇದೋಕ್ತ ಮಂಗಲ ನಿರಾಜನಂ ದರ್ಶಯಾಮಿ ||
