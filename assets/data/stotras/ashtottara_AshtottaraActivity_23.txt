|| ಶ್ರೀ ರಾಘವೇಂದ್ರ ಅಷ್ಟೋತ್ತರ ಶತನಾಮಾವಲಿಃ ||

ಓಂ ಸ್ವವಾಗ್ದೇವತಾ ಸರಿಸದ್ಭಕ್ತವಿಮಲೀಕತ್ರ್ರೇ ನಮಃ |
ಓಂ ಶ್ರೀ ರಾಘವೇಂದ್ರಾಯ ನಮಃ |
ಓಂ ಸಕಲಪ್ರದಾತ್ರೇ ನಮಃ |
ಓಂ ಭಕ್ತಾಘಸಂಭೇದನ ವೃಷ್ಟಿವಜ್ರಾಯ ನಮಃ |
ಓಂ ಕ್ಷಮಾಸುರೇಂದ್ರಾಯ ನಮಃ |
ಓಂ ಹರಿಪಾದ ಕಂಜನಿಷೇವಣಾಲ್ಲಬ್ಧ ಸಮಸ್ತ ಸಂಪದೇ ನಮಃ |
ಓಂ ದೇವಸ್ವಭಾಯ ನಮಃ |
ಓಂ ದಿವಿಜದೃಮಾಯ ನಮಃ |
ಓಂ ಇಷ್ಟಪ್ರದಾತ್ರೇ ನಮಃ |
ಓಂ ಭವ್ಯಸ್ವರೂಪಾಯ ನಮಃ ||10||
ಓಂ ಭವದುಃಖತೂಲ ಸಂಘಾಗ್ನಿಚರ್ಯಾಯ ನಮಃ |
ಓಂ ಸುಖಧೈರ್ಯಶಾಲಿನೇ ನಮಃ |
ಓಂ ಸಮಸ್ತದುಷ್ಟಗ್ರಹ ನಿಗ್ರಹೇಷಾಯ ನಮಃ |
ಓಂ ದುರತ್ಯಯೋಪಪ್ಲವ ಸಿಂಧುಸೇತವೇ ನಮಃ |
ಓಂ ನಿರಸ್ತದೋಷಾಯ ನಮಃ |
ಓಂ ನಿರವದ್ಯವೇಶಾಯ ನಮಃ |
ಓಂ ಪ್ರತ್ಯರ್ಥಮೂಕತ್ವನಿದಾನಭಾಷಾಯ ನಮಃ |
ಓಂ ವಿದ್ವತ್ಪರಿಜ್ಞೇಯ ಮಹಾವಿಶೇಷಾಯ ನಮಃ |
ಓಂ ವಾಗ್ವೈಖರೀ ನಿರ್ಜಿತಭವ್ಯ ಶೇಷಾಯ ನಮಃ |
ಓಂ ಸಂತಾನಸಂಪತ್ಪರಿಶುದ್ಧ ಭಕ್ತಿ ವಿಜ್ಞಾನ ವಾಗ್ದೇಹ ಸುಪಾಟವಾದಿಧಾತ್ರೇ ನಮಃ ||20||
ಓಂ ಶರೀರೋತ್ಥ ಸಮಸ್ತ ದೋಷಹಂತ್ರೇ ನಮಃ |
ಓಂ ಶ್ರೀ ಗುರುರಾಘವೇಂದ್ರಾಯ ನಮಃ |
ಓಂ ತಿರಸ್ಕೃತ ಸುರನದೀ ಜಲಪಾದೋದಕ ಮಹಿಮವತೇ ನಮಃ |
ಓಂ ದುಸ್ತಾಪತ್ರಯ ನಾಶನಾಯ ನಮಃ |
ಓಂ ಮಹಾವಂದ್ಯಾಸುಪುತ್ರದಾಯಕಾಯ ನಮಃ |
ಓಂ ವ್ಯಂಗಸ್ವಂಗಸಮೃದ್ಧಿದಾಯ ನಮಃ |
ಓಂ ಗ್ರಹಮಹಾಪಾಪಾಪಹಾಯ ನಮಃ |
ಓಂ ದುರಿತಕಾನನದಾವಭೂತ ಸ್ವಭಕ್ತ ದರ್ಶನಾಯ ನಮಃ |
ಓಂ ಸ್ವತಂತ್ರಸ್ವತಂತ್ರಾಯ ನಮಃ |
ಓಂ ಶ್ರೀ ಮಧ್ವಮತ ವರ್ಧನಾಯ ನಮಃ ||30||
ಓಂ ವಿಜಯೇಂದ್ರ ಕರಾಬ್ಜೋತ್ಥ ಸುಧೀಂದ್ರ ವರಪುತ್ರಕಾಯ ನಮಃ |
ಓಂ ಯತಿರಾಜೇ ನಮಃ |
ಓಂ ಗುರವೇ ನಮಃ |
ಓಂ ಭಯಾಪಹಾಯ ನಮಃ |
ಓಂ ಜ್ಞಾನಭಕ್ತಿ ಸುಪುತ್ರಾಯುರ್ಯಶಃ ಶ್ರೀ ಪುಣ್ಯವರ್ಧನಾಯ ನಮಃ |
ಓಂ ಪ್ರತಿವಾದಿ ಜಯಸ್ವಾಂತಭೇದಚಿಹ್ನಾದರಾಯ ನಮಃ |
ಓಂ ಸರ್ವವಿದ್ಯಾಪ್ರವೀಣಾಯ ನಮಃ |
ಓಂ ಅಪರೋಕ್ಷೀಕೃತ ಶ್ರೀಶಾಯ ನಮಃ |
ಓಂ ಸಮುಪೇಕ್ಷಿಕೃತಭಾವಜಾಯ ನಮಃ |
ಓಂ ದಯಾದಾಕ್ಷಿಣ್ಯ ವೈರಾಗ್ಯ ವಾಕ್ಪಾಟವಮುಖಾಂಕಿತಾಯ ನಮಃ ||40||
ಓಂ ಶಾಪಾನುಗ್ರಹ ಶಕ್ತಾಯ ನಮಃ |
ಓಂ ಅಜ್ಞಾನ ವಿಸ್ಮೃತಿ ಭ್ರಾಂತಿ ಸಂಶಯಾಪಸ್ಮೃತಿಕ್ಷಯಾದಿ ದೋಷನಾಶಕಾಯ ನಮಃ |
ಓಂ ಅಷ್ಟಾಕ್ಷರ ಜಪೇಷ್ಟಾರ್ಥ ಪ್ರದಾತ್ರೇ ನಮಃ |
ಓಂ ಆತ್ಮಾತ್ಮೀಯ ಸಮುದ್ಭವ ಕಾಯಜದೋಷಹಂತ್ರೇ ನಮಃ |
ಓಂ ಸರ್ವಪುಮರ್ಥ ಪ್ರದಾತ್ರೇ ನಮಃ |
ಓಂ ಕಾಲತ್ರಯ ಪ್ರಾರ್ಥನಾ ಕರ್ತೈಹಿಕಾಮುಸ್ಮಿಕ ಸರ್ವೇಷ್ಟ ಪ್ರದಾತ್ರೇ ನಮಃ |
ಓಂ ಅಗಮ್ಯ ಮಹಿಮ್ನೇ ನಮಃ |
ಓಂ ಮಹಾಯಶಸೇ ನಮಃ |
ಓಂ ಶ್ರೀ ಮಾಧ್ವಮತ ದುಗ್ದಾಬ್ಧಿ ಚಂದ್ರಾಯ ನಮಃ |
ಓಂ ಅನಘಾಯ ನಮಃ ||50||
ಓಂ ಯಥಾಶಕ್ತಿ ಪ್ರದಕ್ಷಿಣಕತ್ರ್ರೇ ಸರ್ವಯಾತ್ರಾ ಫಲದಾತ್ರೇ ನಮಃ |
ಓಂ ಶಿರೋಧಾರಣ ಸರ್ವತೀರ್ಥ ಸ್ನಾನ ಫಲದಾತೃಸ್ವವೃಂದಾವನ ಗತಜಲಾಯ ನಮಃ |
ಓಂ ಕರಣಸರ್ವಾಭೀಷ್ಟದಾತ್ರೇ ನಮಃ |
ಓಂ ಸಂಕೀರ್ತನೇನ ವೇದಾದ್ಯರ್ಥ ಜ್ಞಾನದಾತ್ರೇ ನಮಃ |
ಓಂ ಸಂಸಾರಮಗ್ನ ಜನೋದ್ಧಾರ ಕತ್ರ್ರೇ ನಮಃ |
ಓಂ ಕುಷ್ಟಾದಿರೋಗ ನಿವರ್ತಕಾಯ ನಮಃ |
ಓಂ ಅಂಧದಿವ್ಯ ದೃಷ್ಟಿದಾತ್ರೇ ನಮಃ |
ಓಂ ಏಡಮೂಕವಾಕ್ಪಟಿತ್ವ ಪ್ರದಾತ್ರೇ ನಮಃ |
ಓಂ ಪೂರ್ಣಾಯುಃ ಪ್ರದಾತ್ರೇ ನಮಃ |
ಓಂ ಪೂರ್ಣಸಂಪತ್ತಿದಾತ್ರೇ ನಮಃ ||60||
ಓಂ ಕುಕ್ಷಿಗತ ಸರ್ವದೋಷಘ್ನೇ ನಮಃ |
ಓಂ ಪಂಗುಖಂಜ ಸಮೀಚಿನಾವಯುವದಾತ್ರೇ ನಮಃ |
ಓಂ ಭೂತಪ್ರೇತಪಿಶಾಚಾದಿ ಪೀಡಾಘ್ನೇ ನಮಃ |
ಓಂ ದೀಪಸಂಯೋಜನಾದ್ ಜ್ಞಾನಪುತ್ರದಾತ್ರೇ ನಮಃ |
ಓಂ ದಿವ್ಯಜ್ಞಾನ ಭಕ್ತ್ಯಾದಿ ವರ್ಧನಾಯ ನಮಃ |
ಓಂ ಸರ್ವಾಭೀಷ್ಟದಾಯ ನಮಃ |
ಓಂ ರಾಜಚೋರಮಹಾವ್ಯಾಘ್ರ ಸರ್ಪನಕ್ರಾದಿ ಪೀಡಾಘ್ನೇ ನಮಃ |
ಓಂ ಸ್ವಸ್ತೋತ್ರ ಪಠನೇಷ್ಟಾರ್ಥ ಸಮೃದ್ಧಿದಾಯ ನಮಃ |
ಓಂ ಉದ್ಯತ್ಪ್ರದ್ಯೋತನದ್ಯೋತ ಪ್ರತಾಪಾಯ ನಮಃ |
ಓಂ ಧರ್ಮಕೂರ್ಮಾಸನ ಸ್ಥಿತಾಯ ನಮಃ ||70||
ಓಂ ಖದ್ಯಖದ್ಯೋತನ ದ್ಯೋತ ಪ್ರತಾಪಾಯ ನಮಃ |
ಓಂ ಶ್ರೀರಾಮ ಮಾನಸಾಯ ನಮಃ |
ಓಂ ಧೃತಕಾಷಾಯವಸನಾಯ ನಮಃ |
ಓಂ ತುಲಸೀಹಾರ ವಕ್ಷಸೇ ನಮಃ |
ಓಂ ದೋರ್ದಂಡ ವಿಲಸದ್ಧಂಡ ಕಮಂಡಲವಿರಾಜಿತಾಯ ನಮಃ |
ಓಂ ಅಭಯಜ್ಞಾನ ಮುದ್ರಾಕ್ಷಮಾಲಾ ಶೀಲಕರಾಂಬುಜಾಯ ನಮಃ |
ಓಂ ಯೋಗೇಂದ್ರಮಧ್ಯ ಪಾದಾಬ್ಜಾಯ ನಮಃ |
ಓಂ ಪಾಪಾದ್ರಿ ಪಾಟನ ವಜ್ರಾಯ ನಮಃ |
ಓಂ ಕ್ಷಮಾಸುರ ಗಣಾಧೀಶಾಯ ನಮಃ |
ಓಂ ಹರಿಸೇವಾಲಬ್ಧ ಸರ್ವಸಂಪದೇ ನಮಃ ||80||
ಓಂ ತತ್ತ್ವ ಪ್ರದರ್ಶಕಾಯ ನಮಃ |
ಓಂ ಇಷ್ಟಪ್ರದಾನ ಕಲ್ಪಧ್ರುಮಾಯ ನಮಃ |
ಓಂ ಶ್ರುತ್ಯರ್ಥ ಬೋಧಕಾಯ ನಮಃ |
ಓಂ ಭವ್ಯಕೃತೇ ನಮಃ |
ಓಂ ಬಹುವಾದಿ ವಿಜಯಿನೇ ನಮಃ |
ಓಂ ಪುಣ್ಯವರ್ಧನ ಪಾದಾಬ್ಜಾಭಿಷೇಕ ಜಲಸಂಚಯಾಯ ನಮಃ |
ಓಂ ದ್ಯುನದೀತುಲ್ಯ ಸದ್ಗುಣಾಯ ನಮಃ |
ಓಂ ಭಕ್ತಾಘವಿಧ್ವಂಸಕರ ನಿಜಮೂರ್ತಿಪ್ರದರ್ಶಕಾಯ ನಮಃ |
ಓಂ ಜಗದ್ಗುರವೇ ನಮಃ |
ಓಂ ಕೃಪಾನಿಧಯೇ ನಮಃ ||90||
ಓಂ ಸರ್ವಶಾಸ್ತ್ರ ವಿಶಾರದಾಯ ನಮಃ |
ಓಂ ನಿಖಿಲೇಂದ್ರಿಯ ದೋಷಘ್ನೇ ನಮಃ |
ಓಂ ಅಷ್ಟಾಕ್ಷರ ಮನೂದಿತಾಯ ನಮಃ |
ಓಂ ಸರ್ವಸೌಖ್ಯ ಕೃತೇ ನಮಃ |
ಓಂ ಮೃತಪೋತ ಪ್ರಾಣಧಾತ್ರೇ ನಮಃ |
ಓಂ ವೇದಿಸ್ಥ ಪುರುಷೋಜ್ಜೀವಿನೇ ನಮಃ |
ಓಂ ವಹ್ನಿಸ್ಥ ಮಾಲಿಕೋದ್ಧತ್ರೇ ನಮಃ |
ಓಂ ಸಮಗ್ರ ಟೀಕಾ ವ್ಯಾಖ್ಯಾತ್ರೇ ನಮಃ |
ಓಂ ಭಾಟ್ಟ ಸಂಗ್ರಹಕೃತೇ ನಮಃ |
ಓಂ ಸುಧಾಪರಿಮಲೋದ್ಧತ್ರ್ರೇ ನಮಃ ||100||
ಓಂ ಅಪಸ್ಮಾರಾಪ ಹತ್ರ್ರೇ ನಮಃ |
ಓಂ ಉಪನಿಷತ್ಖಂಡಾರ್ಥ ಕೃತೇ ನಮಃ |
ಓಂ ಋಗ್ವ್ಯಾಖ್ಯಾನ ಕೃದಾಚಾರ್ಯಾಯ ನಮಃ |
ಓಂ ಮಂತ್ರಾಲಯ ನಿವಾಸಿನೇ ನಮಃ |
ಓಂ ನ್ಯಾಯ ಮುಕ್ತಾವಲೀ ಕತ್ರ್ರೇ ನಮಃ |
ಓಂ ಚಂದ್ರಿಕಾ ವ್ಯಾಖ್ಯಾ ಕತ್ರ್ರೇ ನಮಃ |
ಓಂ ಸುತಂತ್ರದೀಪಿಕಾ ಕತ್ರ್ರೇ ನಮಃ |
ಓಂ ಗೀತಾರ್ಥ ಸಂಗ್ರಹಕೃತೇ ನಮಃ ||108||
