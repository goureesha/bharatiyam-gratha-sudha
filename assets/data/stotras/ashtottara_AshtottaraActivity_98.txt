|| ಶ್ರೀಕಾಮಾಕ್ಷೀ ಅಷ್ಟೋತ್ತರ ಶತನಾಮಾವಲಿಃ ||

ಓಂ ಶ್ರೀ ಕಾಲಕಂಠ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ತ್ರಿಪುರಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಬಾಲಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಮಾಯಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ತ್ರಿಪುರಸುನ್ದರ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಸುನ್ದರ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಸೌಭಾಗ್ಯವತ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಕ್ಲೀಂಕಾರ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಸರ್ವಮಂಗಲಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಐಂಕಾರ್ಯೈ ನಮಃ | 10
ಓಂ ಶ್ರೀ ಸ್ಕನ್ದಜನನ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಪರಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಪಂಚದಶಾಕ್ಷರ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ತ್ರೈಲೋಕ್ಯಮೋಹನಾಧೀಶಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಸರ್ವಾಶಾಪೂರವಲ್ಲಭಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಸರ್ವಸಂಕ್ಷೋಭಣಾಧೀಶಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಸರ್ವಸೌಭಾಗ್ಯವಲ್ಲಭಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಸರ್ವಾರ್ಥಸಾಧಕಾಧೀಶಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಸರ್ವರಕ್ಷಾಕರಾಧಿಪಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಸರ್ವರೋಗಹರಾಧೀಶಾಯೈ ನಮಃ | 20
ಓಂ ಶ್ರೀ ಸರ್ವಸಿದ್ಧಿಪ್ರದಾಧಿಪಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಸರ್ವಾನನ್ದಮಯಾಧೀಶಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಯೋಗಿನೀಚಕ್ರನಾಯಿಕಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಭಕ್ತಾನುರಕ್ತಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ರಕ್ತಾಂಗ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಶಂಕರಾರ್ಧಶರೀರಿಣ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಪುಷ್ಪಬಾಣೇಕ್ಷುಕೋದಂಡಪಾಶಾಂಕುಶಕರಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಉಜ್ವಲಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಸಚ್ಚಿದಾನನ್ದಲಹರ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಶ್ರೀವಿದ್ಯಾಯೈ ನಮಃ | 30
ಓಂ ಶ್ರೀ ಪರಮೇಶ್ವರ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಅನಂಗಕುಸುಮೋದ್ಯಾನಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಚಕ್ರೇಶ್ವರ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಭುವನೇಶ್ವರ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಗುಪ್ತಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಗುಪ್ತತರಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ನಿತ್ಯಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ನಿತ್ಯಕ್ಲಿನ್ನಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಮದದ್ರವಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಮೋಹಿಣ್ಯೈ ನಮಃ | 40
ಓಂ ಶ್ರೀ ಪರಮಾನನ್ದಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಕಾಮೇಶ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ತರುಣೀಕಲಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಶ್ರೀಕಲಾವತ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಭಗವತ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಪದ್ಮರಾಗಕಿರೀಟಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ರಕ್ತವಸ್ತ್ರಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ರಕ್ತಭೂಷಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ರಕ್ತಗನ್ಧಾನುಲೇಪನಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಸೌಗನ್ಧಿಕಲಸದ್ವೇಣ್ಯೈ ನಮಃ | 50
ಓಂ ಶ್ರೀ ಮನ್ತ್ರಿಣ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ತನ್ತ್ರರೂಪಿಣ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ತತ್ವಮಯ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಸಿದ್ಧಾನ್ತಪುರವಾಸಿನ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಶ್ರೀಮತ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಚಿನ್ಮಯ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ದೇವ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಕೌಲಿನ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಪರದೇವತಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಕೈವಲ್ಯರೇಖಾಯೈ ನಮಃ | 60
ಓಂ ಶ್ರೀ ವಶಿನ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಸರ್ವೇಶ್ವರ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಸರ್ವಮಾತೃಕಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ವಿಷ್ಣುಸ್ವಸ್ರೇ ನಮಃ |
ಓಂ ಶ್ರೀ ವೇದಮಯ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಸರ್ವಸಮ್ಪತ್ಪ್ರದಾಯಿನ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಕಿಂಕರೀಭೂತಗೀರ್ವಾಣ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಸುತವಾಪಿವಿನೋದಿನ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಮಣಿಪೂರಸಮಾಸೀನಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಅನಾಹತಾಬ್ಜವಾಸಿನ್ಯೈ ನಮಃ | 70
ಓಂ ಶ್ರೀ ವಿಶುದ್ಧಿಚಕ್ರನಿಲಯಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಆಜ್ಞಾಪದ್ಮನಿವಾಸಿನ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಅಷ್ಟತ್ರಿಂಶತ್ಕಲಾಮೂತ್ರ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಸುಷುಮ್ನಾದ್ವಾರಮಧ್ಯಕಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಯೋಗೀಶ್ವರಮನೋಧ್ಯೇಯಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಪರಬ್ರಹ್ಮಸ್ವರೂಪಿಣ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಚತುರ್ಭುಜಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಚನ್ದ್ರಚೂಡಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಪುರಾಣಾಗಮರೂಪಿಣ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಓಂಕಾರ್ಯೈ ನಮಃ | 80
ಓಂ ಶ್ರೀ ವಿಮಲಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ವಿದ್ಯಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಪಂಚಪ್ರಣವರೂಪಿಣ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಭೂತೇಶ್ವರ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಭೂತಮಯ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಪಂಚಾಶತ್ಪೀಠರೂಪಿಣ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಷೋಡಾನ್ಯಾಸಮಹಾರೂಪಿಣ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಕಾಮಾಕ್ಷ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ದಶಮಾತೃಕಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಆಧಾರಶಕ್ತ್ಯೈ ನಮಃ | 90
ಓಂ ಶ್ರೀ ಅರುಣಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಲಕ್ಷ್ಮ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ತ್ರಿಪುರಭೈರವ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ರಹಃಪೂಜಾಸಮಾಲೋಲಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ರಹೋಯನ್ತ್ರಸ್ವರೂಪಿಣ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ತ್ರಿಕೋಣಮಧ್ಯನಿಲಯಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಬಿನ್ದುಮಂಡಲವಾಸಿನ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ವಸುಕೋಣಪುರಾವಾಸಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ದಶಾರದ್ವಯವಾಸಿನ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಚತುರ್ದಶಾರಚಕ್ರಸ್ಥಾಯೈ ನಮಃ | 100
ಓಂ ಶ್ರೀ ವಸುಪದ್ಮನಿವಾಸಿನ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಸ್ವರಾಬ್ಜಪತ್ರನಿಲಯಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ವೃತ್ತತ್ರಯವಾಸಿನ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಚತುರಸ್ರಸ್ವರೂಪಾಸ್ಯಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ನವಚಕ್ರಸ್ವರೂಪಿಣ್ಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಮಹಾನಿತ್ಯಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ವಿಜಯಾಯೈ ನಮಃ |
ಓಂ ಶ್ರೀ ಶ್ರೀರಾಜರಾಜೇಶ್ವರ್ಯೈ ನಮಃ || 108

|| ಇತಿ ಶ್ರೀ ಕಾಮಾಕ್ಷ್ಯಷ್ಟೋತ್ತರಶತ ನಾಮಾವಲಿಃ ಸಮಾಪ್ತಾ ||
