ಯೋಗಕುಂಡಲ್ಯುಪನಿಷದ್ಯೋಗಸಿದ್ಧಿಹೃದಾಸನಂ ।
ನಿರ್ವಿಶೇಷಬ್ರಹ್ಮತತ್ತ್ವಂ ಸ್ವಮಾತ್ರಮಿತಿ ಚಿಂತಯೇ  ॥
ಓಂ ಸಹ ನಾವವತು ಸಹ ನೌ ಭುನಕ್ತು ಸಹ ವೀರ್ಯಂ ಕರವಾವಹೈ ।
ತೇಜಸ್ವಿನಾವಧೀತಮಸ್ತು ಮಾ ವಿದ್ವಿಷಾವಹೈ  ॥
ಓಂ ಶಾಂತಿಃ ಶಾಂತಿಃ ಶಾಂತಿಃ  ॥
ಹರಿಃ ಓಂ  ॥ ಹೇತುದ್ವಯಂ ಹಿ ಚಿತ್ತಸ್ಯ ವಾಸನಾ ಚ ಸಮೀರಣಃ ।
ತಯೋರ್ವಿನಷ್ಟ ಏಕಸ್ಮಿಂಸ್ತದ್ದ್ವಾವಪಿ ವಿನಶ್ಯತಃ  ॥ ೧ ॥
ತಯೋರಾದೌ ಸಮೀರಸ್ಯ ಜಯಂ ಕುರ್ಯಾನ್ನರಃ ಸದಾ ।
ಮಿತಾಹಾರಶ್ಚಾಸನಂ ಚ ಶಕ್ತಿಶ್ಚಾಲಸ್ತೃತೀಯಕಃ  ॥ ೨ ॥
ಏತೇಷಾಂ ಲಕ್ಷಣಂ ವಕ್ಷ್ಯೇ ಶೃಣು ಗೌತಮ ಸಾದರಂ ।
ಸುಸ್ನಿಗ್ಧಮಧುರಾಹಾರಶ್ಚತುರ್ಥಾಂಶವಿವರ್ಜಿತಃ  ॥ ೩ ॥
ಭುಜ್ಯತೇ ಶಿವಸಂಪ್ರೀತ್ಯೈ ಮಿತಾಹಾರಃ ಸ ಉಚ್ಯತೇ ।
ಆಸನಂ ದ್ವಿವಿಧಂ ಪ್ರೋಕ್ತಂ ಪದ್ಮಂ ವಜ್ರಾಸನಂ ತಥಾ  ॥ ೪ ॥
ಊರ್ವೋರುಪರಿ ಚೇದ್ಧತ್ತೇ ಉಭೇ ಪಾದತಲೇ ಯಥಾ ।
ಪದ್ಮಾಸನಂ ಭವೇದೇತತ್ಸರ್ವಪಾಪಪ್ರಣಾಶನಂ  ॥ ೫ ॥
ವಾಮಾಂಘ್ರಿಮೂಲಕಂದಾಧೋ ಹ್ಯನ್ಯಂ ತದುಪರಿ ಕ್ಷಿಪೇತ್ ।
ಸಮಗ್ರೀವಶಿರಃಕಾಯೋ ವಜ್ರಾಸನಮಿತೀರಿತಂ  ॥ ೬ ॥
ಕುಂಡಲ್ಯೇವ ಭವೇಚ್ಛಕ್ತಿಸ್ತಾಂ ತು ಸಂಚಾಲಯೇದ್ಬುಧಃ ।
ಸ್ವಸ್ಥಾನಾದಾಭ್ರುವೋರ್ಮಧ್ಯಂ ಶಕ್ತಿಚಾಲನಮುಚ್ಯತೇ  ॥ ೭ ॥
ತತ್ಸಾಧನೇ ದ್ವಯಂ ಮುಖ್ಯಂ ಸರಸ್ವತ್ಯಾಸ್ತು ಚಾಲನಂ ।
ಪ್ರಾಣರೋಧಮಥಾಭ್ಯಾಸಾದೃಜ್ವೀ ಕುಂಡಲಿನೀ ಭವೇತ್  ॥ ೮ ॥
ತಯೋರಾದೌ ಸರಸ್ವತ್ಯಾಶ್ಚಾಲನಂ ಕಥಯಾಮಿ ತೇ ।
ಅರುಂಧತ್ಯೇವ ಕಥಿತಾ ಪುರಾವಿದ್ಭಿಃ ಸರಸ್ವತೀ  ॥ ೯ ॥
ಯಸ್ಯಾಃ ಸಂಚಾಲನೇನೈವ ಸ್ವಯಂ ಚಲತಿ ಕುಂಡಲೀ ।
ಇಡಾಯಾಂ ವಹತಿ ಪ್ರಾಣೇ ಬದ್ಧ್ವಾ ಪದ್ಮಾಸನಂ ದೃಢಂ  ॥ ೧೦ ॥
ದ್ವಾದಶಾಂಗುಲದೈರ್ಘ್ಯಂ ಚ ಅಂಬರಂ ಚತುರಂಗುಲಂ ।
ವಿಸ್ತೀರ್ಯ ತೇನ ತನ್ನಾಡೀಂ ವೇಷ್ಟಯಿತ್ವಾ ತತಃ ಸುಧೀಃ  ॥ ೧೧ ॥
ಅಂಗುಷ್ಠತರ್ಜನೀಭ್ಯಾಂ ತು ಹಸ್ತಾಭ್ಯಾಂ ಧಾರಯೇದ್ಧೃಢಂ ।
ಸ್ವಶಕ್ತ್ಯಾ ಚಾಲಯೇದ್ವಾಮೇ ದಕ್ಷಿಣೇನ ಪುನಃಪುನಃ  ॥ ೧೨ ॥
ಮುಹೂರ್ತದ್ವಯಪರ್ಯಂತಂ ನಿರ್ಭಯಾಚ್ಚಾಲಯೇತ್ಸುಧೀಃ ।
ಊರ್ಧ್ವಮಾಕರ್ಷಯೇತ್ಕಿಂಚಿತ್ಸುಷುಮ್ನಾಂ ಕುಂಡಲೀಗತಾಂ  ॥ ೧೩ ॥
ತೇನ ಕುಂಡಲಿನೀ ತಸ್ಯಾಃ ಸುಷುಮ್ನಾಯಾ ಮುಖಂ ವ್ರಜೇತ್ ।
ಜಹಾತಿ ತಸ್ಮಾತ್ಪ್ರಾಣೋಽಯಂ ಸುಷುಮ್ನಾಂ ವ್ರಜತಿ ಸ್ವತಃ  ॥ ೧೪ ॥
ತುಂದೇ ತು ತಾನಂ ಕುರ್ಯಾಚ್ಚ ಕಂಠಸಂಕೋಚನೇ ಕೃತೇ ।
ಸರಸ್ವತ್ಯಾಂ ಚಾಲನೇನ ವಕ್ಷಸಶ್ಚೋರ್ಧ್ವಗೋ ಮರುತ್  ॥ ೧೫ ॥
ಸೂರ್ಯೇಣ ರೇಚಯೇದ್ವಾಯುಂ ಸರಸ್ವತ್ಯಾಸ್ತು ಚಾಲನೇ ।
ಕಂಠಸಂಕೋಚನಂ ಕೃತ್ವಾ ವಕ್ಷಸಶ್ಚೋರ್ಧ್ವಗೋ ಮರುತ್  ॥ ೧೬ ॥
ತಸ್ಮಾತ್ಸಂಚಾಲಯೇನ್ನಿತ್ಯಂ ಶಬ್ದಗರ್ಭಾಂ ಸರಸ್ವತೀಂ ।
ಯಸ್ಯಾಃ ಸಂಚಾಲನೇನೈವ ಯೋಗೀ ರೋಗೈಃ ಪ್ರಮುಚ್ಯತೇ  ॥ ೧೭ ॥
ಗುಲ್ಮಂ ಜಲೋದರಃ ಪ್ಲೀಹಾ ಯೇ ಚಾನ್ಯೇ ತುಂದಮಧ್ಯಗಾಃ ।
ಸರ್ವೇ ತೇ ಶಕ್ತಿಚಾಲೇನ ರೋಗಾ ನಶ್ಯಂತಿ ನಿಶ್ಚಯಂ  ॥ ೧೮ ॥
ಪ್ರಾಣರೋಧಮಥೇದಾನೀಂ ಪ್ರವಕ್ಷ್ಯಾಮಿ ಸಮಾಸತಃ ।
ಪ್ರಾಣಶ್ಚ ದಹನೋ ವಾಯುರಾಯಾಮಃ ಕುಂಭಕಃ ಸ್ಮೃತಃ  ॥ ೧೯ ॥
ಸ ಏವ ದ್ವಿವಿಧಃ ಪ್ರೋಕ್ತಃ ಸಹಿತಃ ಕೇವಲಸ್ತಥಾ ।
ಯಾವತ್ಕೇವಲಸಿದ್ಧಿಃ ಸ್ಯಾತ್ತಾವತ್ಸಹಿತಮಭ್ಯಸೇತ್  ॥ ೨೦ ॥
ಸೂರ್ಯೋಜ್ಜಾಯೀ ಶೀತಲೀ ಚ ಭಸ್ತ್ರೀ ಚೈವ ಚತುರ್ಥಿಕಾ ।
ಭೇದೈರೇವ ಸಮಂ ಕುಂಭೋ ಯಃ ಸ್ಯಾತ್ಸಹಿತಕುಂಭಕಃ  ॥ ೨೧ ॥
ಪವಿತ್ರೇ ನಿರ್ಜನೇ ದೇಶೇ ಶರ್ಕರಾದಿವಿವರ್ಜಿತೇ ।
ಧನುಃಪ್ರಮಾಣಪರ್ಯಂತೇ ಶೀತಾಗ್ನಿಜಲವರ್ಜಿತೇ  ॥ ೨೨ ॥
ಪವಿತ್ರೇ ನಾತ್ಯುಚ್ಚನೀಚೇ ಹ್ಯಾಸನೇ ಸುಖದೇ ಸುಖೇ ।
ಬದ್ಧಪದ್ಮಾಸನಂ ಕೃತ್ವಾ ಸರಸ್ವತ್ಯಾಸ್ತು ಚಾಲನಂ  ॥ ೨೩ ॥
ದಕ್ಷನಾಡ್ಯಾ ಸಮಾಕೃಷ್ಯ ಬಹಿಷ್ಠಂ ಪವನಂ ಶನೈಃ ।
ಯಥೇಷ್ಟಂ ಪೂರಯೇದ್ವಾಯುಂ ರೇಚಯೇದಿಡಯಾ ತತಃ  ॥ ೨೪ ॥
ಕಪಾಲಶೋಧನೇ ವಾಪಿ ರೇಚಯೇತ್ಪವನಂ ಶನೈಃ ।
ಚತುಷ್ಕಂ ವಾತದೋಷಂ ತು ಕೃಮಿದೋಷಂ ನಿಹಂತಿ ಚ  ॥ ೨೫ ॥
ಪುನಃ ಪುನರಿದಂ ಕಾರ್ಯಂ ಸೂರ್ಯಭೇದಮುದಾಹೃತಂ ।
ಮುಖಂ ಸಂಯಮ್ಯ ನಾಡಿಭ್ಯಾಮಾಕೃಷ್ಯ ಪವನಂ ಶನೈಃ  ॥ ೨೬ ॥
ಯಥಾ ಲಗತಿ ಕಂಠಾತ್ತು ಹೃದಯಾವಧಿ ಸಸ್ವನಂ ।
ಪೂರ್ವವತ್ಕುಂಭಯೇತ್ಪ್ರಾಣಂ ರೇಚಯೇದಿಡಯಾ ತತಃ  ॥ ೨೭ ॥
ಶೀರ್ಷೋದಿತಾನಲಹರಂ ಗಲಶ್ಲೇಷ್ಮಹರಂ ಪರಂ ।
ಸರ್ವರೋಗಹರಂ ಪುಣ್ಯಂ ದೇಹಾನಲವಿವರ್ಧನಂ  ॥ ೨೮ ॥
ನಾಡೀಜಲೋದರಂ ಧಾತುಗತದೋಷವಿನಾಶನಂ ।
ಗಚ್ಛತಸ್ತಿಷ್ಠತಃ ಕಾರ್ಯಮುಜ್ಜಾಯ್ಯಾಖ್ಯಂ ತು ಕುಂಭಕಂ  ॥ ೨೯ ॥
ಜಿಹ್ವಯಾ ವಾಯುಮಾಕೃಷ್ಯ ಪೂರ್ವವತ್ಕುಂಭಕಾದನು ।
ಶನೈಸ್ತು ಘ್ರಾಣರಂಧ್ರಾಭ್ಯಾಂ ರೇಚಯೇದನಿಲಂ ಸುಧೀಃ  ॥ ೩೦ ॥
ಗುಲ್ಮಪ್ಲೀಹಾದಿಕಾಂದೋಷಾನ್ಕ್ಷಯಂ ಪಿತ್ತಂ ಜ್ವರಂ ತೃಷಾಂ ।
ವಿಷಾಣಿ ಶೀತಲೀ ನಾಮ ಕುಂಭಕೋಽಯಂ ನಿಹಂತಿ ಚ  ॥ ೩೧ ॥
ತತಃ ಪದ್ಮಾಸನಂ ಬದ್ಧ್ವಾ ಸಮಗ್ರೀವೋದರಃ ಸುಧೀಃ ।
ಮುಖಂ ಸಂಯಮ್ಯ ಯತ್ನೇನ ಪ್ರಾಣಂ ಘ್ರಾಣೇನ ರೇಚಯೇತ್  ॥ ೩೨ ॥
ಯಥಾ ಲಗತಿ ಕಂಠಾತ್ತು ಕಪಾಲೇ ಸಸ್ವನಂ ತತಃ ।
ವೇಗೇನ ಪೂರಯೇತ್ಕಿಂಚಿಧೃತ್ಪದ್ಮಾವಧಿ ಮಾರುತಂ  ॥ ೩೩ ॥
ಪುನರ್ವಿರೇಚಯೇತ್ತದ್ವತ್ಪೂರಯೇಚ್ಚ ಪುನಃ ಪುನಃ ।
ಯಥೈವ ಲೋಹಕಾರಾಣಾಂ ಭಸ್ತ್ರಾ ವೇಗೇನ ಚಾಲ್ಯತೇ  ॥ ೩೪ ॥
ತಥೈವ ಸ್ವಶರೀರಸ್ಥಂ ಚಾಲಯೇತ್ಪವನಂ ಶನೈಃ ।
ಯಥಾ ಶ್ರಮೋ ಭವೇದ್ದೇಹೇ ತಥಾ ಸೂರ್ಯೇಣ ಪೂರಯೇತ್  ॥ ೩೫ ॥
ಯಥೋದರಂ ಭವೇತ್ಪೂರ್ಣಂ ಪವನೇನ ತಥಾ ಲಘು ।
ಧಾರಯನ್ನಾಸಿಕಾಮಧ್ಯಂ ತರ್ಜನೀಭ್ಯಾಂ ವಿನಾ ದೃಢಂ  ॥ ೩೬ ॥
ಕುಂಭಕಂ ಪೂರ್ವವತ್ಕೃತ್ವಾ ರೇಚೇಯೇದಿಡಯಾನಿಲಂ ।
ಕಂಠೋತ್ಥಿತಾನಲಹರಂ ಶರೀರಾಗ್ನಿವಿವರ್ಧನಂ  ॥ ೩೭ ॥
ಕುಂಡಲೀಬೋಧಕಂ ಪುಣ್ಯಂ ಪಾಪಘ್ನಂ ಶುಭದಂ ಸುಖಂ ।
ಬ್ರಹ್ಮನಾಡೀಮುಖಾಂತಸ್ಥಕಫಾದ್ಯರ್ಗಲನಾಶನಂ  ॥ ೩೮ ॥
ಗುಣತ್ರಯಸಮುದ್ಭೂತಗ್ರಂಥಿತ್ರಯವಿಭೇದಕಂ ।
ವಿಶೇಷೇಣೈವ ಕರ್ತವ್ಯಂ ಭಸ್ತ್ರಾಖ್ಯಂ ಕುಂಭಕಂ ತ್ವಿದಂ  ॥ ೩೯ ॥
ಚತುರ್ಣಾಮಪಿ ಭೇದಾನಾಂ ಕುಂಭಕೇ ಸಮುಪಸ್ಥಿತೇ ।
ಬಂಧತ್ರಯಮಿದಂ ಕಾರ್ಯಂ ಯೋಗಿಭಿರ್ವೀತಕಲ್ಮಷೈಃ  ॥ ೪೦ ॥
ಪ್ರಥಮೋ ಮೂಲಬಂಧಸ್ತು ದ್ವಿತೀಯೋಡ್ಡೀಯಣಾಭಿಧಃ ।
ಜಾಲಂಧರಸ್ತೃತೀಯಸ್ತು ತೇಷಾಂ ಲಕ್ಷಣಮುಚ್ಯತೇ  ॥ ೪೧ ॥
ಅಧೋಗತಿಮಪಾನಂ ವೈ ಊರ್ಧ್ವಗಂ ಕುರುತೇ ಬಲಾತ್ ।
ಆಕುಂಚನೇನ ತಂ ಪ್ರಾಹುರ್ಮೂಲಬಂಧೋಽಯಮುಚ್ಯತೇ  ॥ ೪೨ ॥
ಅಪಾನೇ ಚೋರ್ಧ್ವಗೇ ಯಾತೇ ಸಂಪ್ರಾಪ್ತೇ ವಹ್ನಿಮಂಡಲೇ ।
ತತೋಽನಲಶಿಖಾ ದೀರ್ಘಾ ವರ್ಧತೇ ವಾಯುನಾಽಽಹತಾ  ॥ ೪೩ ॥
ತತೋ ಯಾತೌ ವಹ್ನ್ಯಪಾನೌ ಪ್ರಾಣಮುಷ್ಣಸ್ವರೂಪಕಂ ।
ತೇನಾತ್ಯಂತಪ್ರದೀಪ್ತೇನ ಜ್ವಲನೋ ದೇಹಜಸ್ತಥಾ  ॥ ೪೪ ॥
ತೇನ ಕುಂಡಲಿನೀ ಸುಪ್ತಾ ಸಂತಪ್ತಾ ಸಂಪ್ರಬುಧ್ಯತೇ ।
ದಂಡಾಹತಭುಜಂಗೀವ ನಿಃಶ್ವಸ್ಯ ಋಜುತಾಂ ವ್ರಜೇತ್  ॥ ೪೫ ॥
ಬಿಲಪ್ರವೇಶತೋ ಯತ್ರ ಬ್ರಹ್ಮನಾಡ್ಯಂತರಂ ವ್ರಜೇತ್ ।
ತಸ್ಮಾನ್ನಿತ್ಯಂ ಮೂಲಬಂಧಃ ಕರ್ತವ್ಯೋ ಯೋಗಿಭಿಃ ಸದಾ  ॥ ೪೬ ॥
ಕುಂಭಕಾಂತೇ ರೇಚಕಾದೌ ಕರ್ತವ್ಯಸ್ತೂಡ್ಡಿಯಾಣಕಃ ।
ಬಂಧೋ ಯೇನ ಸುಷುಮ್ನಾಯಾಂ ಪ್ರಾಣಸ್ತೂಡ್ಡೀಯತೇ ಯತಃ  ॥ ೪೭ ॥
ತಸ್ಮಾದುಡ್ಡೀಯಣಾಖ್ಯೋಽಯಂ ಯೋಗಿಭಿಃ ಸಮುದಾಹೃತಃ ।
ಸತಿ ವಜ್ರಾಸನೇ ಪಾದೌ ಕರಾಭ್ಯಾಂ ಧಾರಯೇದ್ದೃಢಂ  ॥ ೪೮ ॥
ಗುಲ್ಫದೇಶಸಮೀಪೇ ಚ ಕಂದಂ ತತ್ರ ಪ್ರಪೀಡಯೇತ್ ।
ಪಶ್ಚಿಮಂ ತಾನಮುದರೇ ಧಾರಯೇದ್-ಹೃದಯೇ ಗಲೇ  ॥ ೪೯ ॥
ಶನೈಃ ಶನೈರ್ಯದಾ ಪ್ರಾಣಸ್ತುಂದಸಂಧಿಂ ನಿಗಚ್ಛತಿ ।
ತುಂದದೋಷಂ ವಿನಿರ್ಧೂಯ ಕರ್ತವ್ಯಂ ಸತತಂ ಶನೈಃ  ॥ ೫೦ ॥
ಪೂರಕಾಂತೇ ತು ಕರ್ತವ್ಯೋ ಬಂಧೋ ಜಾಲಂಧರಾಭಿಧಃ ।
ಕಂಠಸಂಕೋಚರೂಪೋಽಸೌ ವಾಯುಮಾರ್ಗನಿರೋಧಕಃ  ॥ ೫೧ ॥
ಅಧಸ್ತಾತ್ಕುಂಚನೇನಾಶು ಕಂಠಸಂಕೋಚನೇ ಕೃತೇ ।
ಮಧ್ಯೇ ಪಶ್ಚಿಮತಾನೇನ ಸ್ಯಾತ್ಪ್ರಾಣೋ ಬ್ರಹ್ಮನಾಡಿಗಃ  ॥ ೫೨ ॥
ಪೂರ್ವೋಕ್ತೇನ ಕ್ರಮೇಣೈವ ಸಮ್ಯಗಾಸನಮಾಸ್ಥಿತಃ ।
ಚಾಲನಂ ತು ಸರಸ್ವತ್ಯಾಃ ಕೃತ್ವಾ ಪ್ರಾಣಂ ನಿರೋಧಯೇತ್  ॥ ೫೩ ॥
ಪ್ರಥಮೇ ದಿವಸೇ ಕಾರ್ಯಂ ಕುಂಭಕಾನಾಂ ಚತುಷ್ಟಯಂ ।
ಪ್ರತ್ಯೇಕಂ ದಶಸಂಖ್ಯಾಕಂ ದ್ವಿತೀಯೇ ಪಂಚಭಿಸ್ತಥಾ  ॥ ೫೪ ॥
ವಿಶತ್ಯಲಂ ತೃತೀಯೇಽಹ್ನಿ ಪಂಚವೃದ್ಧ್ಯಾ ದಿನೇ ದಿನೇ ।
ಕರ್ತವ್ಯಃ ಕುಂಭಕೋ ನಿತ್ಯಂ ಬಂಧತ್ರಯಸಮನ್ವಿತಃ  ॥ ೫೫ ॥
ದಿವಾ ಸುಪ್ತಿರ್ನಿಶಾಯಾಂ ತು ಜಾಗರಾದತಿಮೈಥುನಾತ್ ।
ಬಹುಸಂಕ್ರಮಣಂ ನಿತ್ಯಂ ರೋಧಾನ್ಮೂತ್ರಪುರೀಷಯೋಃ  ॥ ೫೬ ॥
ವಿಷಮಾಶನದೋಷಾಚ್ಚ ಪ್ರಯಾಸಪ್ರಾಣಚಿಂತನಾತ್ ।
ಶೀಘ್ರಮುತ್ಪದ್ಯತೇ ರೋಗಃ ಸ್ತಂಭಯೇದ್ಯದಿ ಸಂಯಮೀ  ॥ ೫೭ ॥
ಯೋಗಾಭ್ಯಾಸೇನ ಮೇ ರೋಗ ಉತ್ಪನ್ನ ಇತಿ ಕಥ್ಯತೇ ।
ತತೋಽಭ್ಯಾಸಂ ತ್ಯಜೇದೇವಂ ಪ್ರಥಮಂ ವಿಘ್ನ ಉಚ್ಯತೇ  ॥ ೫೮ ॥
ದ್ವಿತೀಯಂ ಸಂಶಯಾಖ್ಯಂ ಚ ತೃತೀಯಂ ಚ ಪ್ರಮತ್ತತಾ ।
ಆಲಸ್ಯಾಖ್ಯಂ ಚತುರ್ಥಂ ಚ ನಿದ್ರಾರೂಪಂ ತು ಪಂಚಮಂ  ॥ ೫೯ ॥
ಷಷ್ಠಂ ತು ವಿರತಿರ್ಭ್ರಾಂತಿಃ ಸಪ್ತಮಂ ಪರಿಕೀರ್ತಿತಂ ।
ವಿಷಮಂ ಚಾಷ್ಟಮಂ ಚೈವ ಅನಾಖ್ಯಂ ನವಮಂ ಸ್ಮೃತಂ  ॥ ೬೦ ॥
ಅಲಬ್ಧಿರ್ಯೋಗತತ್ತ್ವಸ್ಯ ದಶಮಂ ಪ್ರೋಚ್ಯತೇ ಬುಧೈಃ ।
ಇತ್ಯೇತದ್ವಿಘ್ನದಶಕಂ ವಿಚಾರೇಣ ತ್ಯಜೇದ್ಬುಧಃ  ॥ ೬೧ ॥
ಪ್ರಾಣಾಭ್ಯಾಸಸ್ತತಃ ಕಾರ್ಯೋ ನಿತ್ಯಂ ಸತ್ತ್ವಸ್ಥಯಾ ಧಿಯಾ ।
ಸುಷುಮ್ನಾ ಲೀಯತೇ ಚಿತ್ತಂ ತಥಾ ವಾಯುಃ ಪ್ರಧಾವತಿ  ॥ ೬೨ ॥
ಶುಷ್ಕೇ ಮಲೇ ತು ಯೋಗೀ ಚ ಸ್ಯಾದ್ಗತಿಶ್ಚಲಿತಾ ತತಃ ।
ಅಧೋಗತಿಮಪಾನಂ ವೈ ಊರ್ಧ್ವಗಂ ಕುರುತೇ ಬಲಾತ್  ॥ ೬೩ ॥
ಆಕುಂಚನೇನ ತಂ ಪ್ರಾಹುರ್ಮೂಲಬಂಧೋಽಯಮುಚ್ಯತೇ ।
ಅಪಾನಶ್ಚೋರ್ಧ್ವಗೋ ಭೂತ್ವಾ ವಹ್ನಿನಾ ಸಹ ಗಚ್ಛತಿ  ॥ ೬೪ ॥
ಪ್ರಾಣಸ್ಥಾನಂ ತತೋ ವಹ್ನಿಃ ಪ್ರಾಣಾಪಾನೌ ಚ ಸತ್ವರಂ ।
ಮಿಲಿತ್ವಾ ಕುಂಡಲೀಂ ಯಾತಿ ಪ್ರಸುಪ್ತಾ ಕುಂಡಲಾಕೃತಿಃ  ॥ ೬೫ ॥
ತೇನಾಗ್ನಿನಾ ಚ ಸಂತಪ್ತಾ ಪವನೇನೈವ ಚಾಲಿತಾ ।
ಪ್ರಸಾರ್ಯ ಸ್ವಶರೀರಂ ತು ಸುಷುಮ್ನಾ ವದನಾಂತರೇ  ॥ ೬೬ ॥
ಬ್ರಹ್ಮಗ್ರಂಥಿಂ ತತೋ ಭಿತ್ತ್ವಾ ರಜೋಗುಣಸಮುದ್ಭವಂ ।
ಸುಷುಮ್ನಾ ವದನೇ ಶೀಘ್ರಂ ವಿದ್ಯುಲ್ಲೇಖೇವ ಸಂಸ್ಫುರೇತ್  ॥ ೬೭ ॥
ವಿಷ್ಣುಗ್ರಂಥಿಂ ಪ್ರಯಾತ್ಯುಚ್ಚೈಃ ಸತ್ವರಂ ಹೃದಿ ಸಂಸ್ಥಿತಾ ।
ಊರ್ಧ್ವಂ ಗಚ್ಛತಿ ಯಚ್ಚಾಸ್ತೇ ರುದ್ರಗ್ರಂಥಿಂ ತದುದ್ಭವಂ  ॥ ೬೮ ॥
ಭ್ರುವೋರ್ಮಧ್ಯೇ ತು ಸಂಭಿದ್ಯ ಯಾತಿ ಶೀತಾಂಶುಮಂಡಲಂ ।
ಅನಾಹತಾಖ್ಯಂ ಯಚ್ಚಕ್ರಂ ದಲೈಃ ಷೋಡಶಭಿರ್ಯುತಂ  ॥ ೬೯ ॥
ತತ್ರ ಶೀತಾಂಶುಸಂಜಾತಂ ದ್ರವಂ ಶೋಷಯತಿ ಸ್ವಯಂ ।
ಚಲಿತೇ ಪ್ರಾಣವೇಗೇನ ರಕ್ತಂ ಪೀತಂ ರವೇರ್ಗ್ರಹಾತ್  ॥ ೭೦ ॥
ಯಾತೇಂದುಚಕ್ರಂ ಯತ್ರಾಸ್ತೇ ಶುದ್ಧಶ್ಲೇಷ್ಮದ್ರವಾತ್ಮಕಂ ।
ತತ್ರ ಸಿಕ್ತಂ ಗ್ರಸತ್ಯುಷ್ಣಂ ಕಥಂ ಶೀತಸ್ವಭಾವಕಂ  ॥ ೭೧ ॥
ತಥೈವ ರಭಸಾ ಶುಕ್ಲಂ ಚಂದ್ರರೂಪಂ ಹಿ ತಪ್ಯತೇ ।
ಊರ್ಧ್ವಂ ಪ್ರವಹತಿ ಕ್ಷುಬ್ಧಾ ತದೈವಂ ಭ್ರಮತೇತರಾಂ  ॥ ೭೨ ॥
ತಸ್ಯಾಸ್ವಾದವಶಾಚ್ಚಿತ್ತಂ ಬಹಿಷ್ಠಂ ವಿಷಯೇಷು ಯತ್ ।
ತದೇವ ಪರಮಂ ಭುಕ್ತ್ವಾ ಸ್ವಸ್ಥಃ ಸ್ವಾತ್ಮರತೋ ಯುವಾ  ॥ ೭೩ ॥
ಪ್ರಕೃತ್ಯಷ್ಟಕರೂಪಂ ಚ ಸ್ಥಾನಂ ಗಚ್ಛತಿ ಕುಂಡಲೀ ।
ಕ್ರೋಡೀಕೃತ್ಯ ಶಿವಂ ಯಾತಿ ಕ್ರೋಡೀಕೃತ್ಯ ವಿಲೀಯತೇ  ॥ ೭೪ ॥
ಇತ್ಯಧೋರ್ಧ್ವರಜಃ ಶುಕ್ಲಂ ಶಿವೇ ತದನು ಮಾರುತಃ ।
ಪ್ರಾಣಾಪಾನೌ ಸಮೌ ಯಾತಿ ಸದಾ ಜಾತೌ ತಥೈವ ಚ  ॥ ೭೫ ॥
ಭೂತೇಽಲ್ಪೇ ಚಾಪ್ಯನಲ್ಪೇ ವಾ ವಾಚಕೇ ತ್ವತಿವರ್ಧತೇ ।
ಧವಯತ್ಯಖಿಲಾ ವಾತಾ ಅಗ್ನಿಮೂಷಾಹಿರಣ್ಯವತ್  ॥ ೭೬ ॥
ಆಧಿಭೌತಿಕದೇಹಂ ತು ಆಧಿದೈವಿಕವಿಗ್ರಹೇ ।
ದೇಹೋಽತಿವಿಮಲಂ ಯಾತಿ ಚಾತಿವಾಹಿಕತಾಮಿಯಾತ್  ॥ ೭೭ ॥
ಜಾಡ್ಯಭಾವವಿನಿರ್ಮುಕ್ತಮಮಲಂ ಚಿನ್ಮಯಾತ್ಮಕಂ ।
ತಸ್ಯಾತಿವಾಹಿಕಂ ಮುಖ್ಯಂ ಸರ್ವೇಷಾಂ ತು ಮದಾತ್ಮಕಂ  ॥ ೭೮ ॥
ಜಾಯಾಭವವಿನಿರ್ಮುಕ್ತಿಃ ಕಾಲರೂಪಸ್ಯ ವಿಭ್ರಮಃ ।
ಇತಿ ತಂ ಸ್ವಸ್ವರೂಪಾ ಹಿ ಮತೀ ರಜ್ಜುಭುಜಂಗವತ್  ॥ ೭೯ ॥
ಮೃಷೈವೋದೇತಿ ಸಕಲಂ ಮೃಷೈವ ಪ್ರವಿಲೀಯತೇ ।
ರೌಪ್ಯಬುದ್ಧಿಃ ಶುಕ್ತಿಕಾಯಾಂ ಸ್ತ್ರೀಪುಂಸೋರ್ಭ್ರಮತೋ ಯಥಾ  ॥ ೮೦ ॥
ಪಿಂಡಬ್ರಹ್ಮಾಂಡಯೋರೈಕ್ಯಂ ಲಿಂಗಸೂತ್ರಾತ್ಮನೋರಪಿ ।
ಸ್ವಾಪಾವ್ಯಾಕೃತಯೋರೈಕ್ಯಂ ಸ್ವಪ್ರಕಾಶಚಿದಾತ್ಮನೋಃ  ॥ ೮೧ ॥
ಶಕ್ತಿಃ ಕುಂಡಲಿನೀ ನಾಮ ಬಿಸತಂತುನಿಭಾ ಶುಭಾ ।
ಮೂಲಕಂದಂ ಫಣಾಗ್ರೇಣ ದೃಷ್ಟ್ವಾ ಕಮಲಕಂದವತ್  ॥ ೮೨ ॥
ಮುಖೇನ ಪುಚ್ಛಂ ಸ।ನ್ಗೃಹ್ಯ ಬ್ರಹ್ಮರಂಧ್ರಸಮನ್ವಿತಾ ।
ಪದ್ಮಾಸನಗತಃ ಸ್ವಸ್ಥೋ ಗುದಮಾಕುಂಚ್ಯ ಸಾಧಕಃ  ॥ ೮೩ ॥
ವಾಯುಮೂರ್ಧ್ವಗತಂ ಕುರ್ವನ್ಕುಂಭಕಾವಿಷ್ಟಮಾನಸಃ ।
ವಾಯ್ವಾಘಾತವಶಾದಗ್ನಿಃ ಸ್ವಾಧಿಷ್ಠಾನಗತೋ ಜ್ವಲನ್  ॥ ೮೪ ॥
ಜ್ವಲನಾಘಾತಪವನಾಘಾತೋರೂನ್ನಿದ್ರಿತೋಽಹಿರಾಟ್ ।
ಬ್ರಹ್ಮಗ್ರಂಥಿಂ ತತೋ ಭಿತ್ತ್ವಾ ವಿಷ್ಣುಗ್ರಂಥಿಂ ಭಿನತ್ತ್ಯತಃ  ॥ ೮೫ ॥
ರುದ್ರಗ್ರಂಥಿಂ ಚ ಭಿತ್ತ್ವೈವ ಕಮಲಾನಿ ಭಿನತ್ತಿ ಷಟ್ ।
ಸಹಸ್ರಕಮಲೇ ಶಕ್ತಿಃ ಶಿವೇನ ಸಹ ಮೋದತೇ  ॥ ೮೬ ॥
ಸೈವಾವಸ್ಥಾ ಪರಾ ಜ್ಞೇಯಾ ಸೈವ ನಿರ್ವೃತಿಕಾರಿಣೀ ಇತಿ  ॥
ಇತಿ ಪ್ರಥಮೋಽಧ್ಯಾಯಃ  ॥ ೧ ॥
ಅಥಾಹಂ ಸಂಪ್ರವಕ್ಷ್ಯಾಮಿ ವಿದ್ಯಾಂ ಖೇಚರಿಸ।ನ್ಜ್ಞಿಕಾಂ ।
ಯಥಾ ವಿಜ್ಞಾನವಾನಸ್ಯಾ ಲೋಕೇಽಸ್ಮಿನ್ನಜರೋಽಮರಃ  ॥ ೧ ॥
ಮೃತ್ಯುವ್ಯಾಧಿಜರಾಗ್ರಸ್ತೋ ದೃಷ್ಟ್ವಾ ವಿದ್ಯಾಮಿಮಾಂ ಮುನೇ ।
ಬುದ್ಧಿಂ ದೃಢತರಾಂ ಕೃತ್ವಾ ಖೇಚರೀಂ ತು ಸಮಭ್ಯಸೇತ್  ॥ ೨ ॥
ಜರಾಮೃತ್ಯುಗದಘ್ನೋ ಯಃ ಖೇಚರೀಂ ವೇತ್ತಿ ಭೂತಲೇ ।
ಗ್ರಂಥತಶ್ಚಾರ್ಥತಶ್ಚೈವ ತದಭ್ಯಾಸಪ್ರಯೋಗತಃ  ॥ ೩ ॥
ತಂ ಮುನೇ ಸರ್ವಭಾವೇನ ಗುರುಂ ಮತ್ವಾ ಸಮಾಶ್ರಯೇತ್ ।
ದುರ್ಲಭಾ ಖೇಚರೀ ವಿದ್ಯಾ ತದಭ್ಯಾಸೋಽಪಿ ದುರ್ಲಭಃ  ॥ ೪ ॥
ಅಭ್ಯಾಸಂ ಮೇಲನಂ ಚೈವ ಯುಗಪನ್ನೈವ ಸಿಧ್ಯತಿ ।
ಅಭ್ಯಾಸಮಾತ್ರನಿರತಾ ನ ವಿಂದಂತೇ ಹ ಮೇಲನಂ  ॥ ೫ ॥
ಅಭ್ಯಾಸಂ ಲಭತೇ ಬ್ರಹ್ಮಂಜನ್ಮಜನ್ಮಾಂತರೇ ಕ್ವಚಿತ್ ।
ಮೇಲನಂ ತು ಜನ್ಮನಾಂ ಶತಾಂತೇಽಪಿ ನ ಲಭ್ಯತೇ  ॥ ೬ ॥
ಅಭ್ಯಾಸಂ ಬಹುಜನ್ಮಾಂತೇ ಕೃತ್ವಾ ತದ್ಭಾವಸಾಧಿತಂ ।
ಮೇಲನಂ ಲಭತೇ ಕಶ್ಚಿದ್ಯೋಗೀ ಜನ್ಮಾಂತರೇ ಕ್ವಚಿತ್  ॥ ೭ ॥
ಯದಾ ತು ಮೇಲನಂ ಯೋಗೀ ಲಭತೇ ಗುರುವಕ್ತ್ರತಃ ।
ತದಾ ತತ್ಸಿದ್ಧಿಮಾಪ್ನೋತಿ ಯದುಕ್ತಾ ಶಾಸ್ತ್ರಸಂತತೌ  ॥ ೮ ॥
ಗ್ರಂಥತಶ್ಚಾರ್ಥತಶ್ಚೈವ ಮೇಲನಂ ಲಭತೇ ಯದಾ ।
ತದಾ ಶಿವತ್ವಮಾಪ್ನೋತಿ ನಿರ್ಮುಕ್ತಃ ಸರ್ವಸಂಸೃತೇಃ  ॥ ೯ ॥
ಶಾಸ್ತ್ರಂ ವಿನಾಪಿ ಸಂಬೋದ್ಧುಂ ಗುರವೋಽಪಿ ನ ಶಕ್ನುಯುಃ ।
ತಸ್ಮಾತ್ಸುದುರ್ಲಭತರಂ ಲಭ್ಯಂ ಶಾಸ್ತ್ರಮಿದಂ ಮುನೇ  ॥ ೧೦ ॥
ಯಾವನ್ನ ಲಭ್ಯತೇ ಶಾಸ್ತ್ರಂ ತಾವದ್ಗಾಂ ಪರ್ಯಟೇದ್ಯತಿಃ ।
ಯದಾ ಸಂಲಭ್ಯತೇ ಶಾಸ್ತ್ರಂ ತದಾ ಸಿದ್ಧಿಃ ಕರೇ ಸ್ಥಿತಾ  ॥ ೧೧ ॥
ನ ಶಾಸ್ತ್ರೇಣ ವಿನಾ ಸಿದ್ಧಿರ್ದೃಷ್ಟಾ ಚೈವ ಜಗತ್ತ್ರಯೇ ।
ತಸ್ಮಾನ್ಮೇಲನದಾತಾರಂ ಶಾಸ್ತ್ರದಾತಾರಮಚ್ಯುತಂ  ॥ ೧೨ ॥
ತದಭ್ಯಾಸಪ್ರದಾತಾರಂ ಶಿವಂ ಮತ್ವಾ ಸಮಾಶ್ರಯೇತ್ ।
ಲಬ್ಧ್ವಾ ಶಾಸ್ತ್ರಮಿದಂ ಮಹ್ಯಮನ್ಯೇಷಾಂ ನ ಪ್ರಕಾಶಯೇತ್  ॥ ೧೩ ॥
ತಸ್ಮಾತ್ಸರ್ವಪ್ರಯತ್ನೇನ ಗೋಪನೀಯಂ ವಿಜಾನತಾ ।
ಯತ್ರಾಸ್ತೇ ಚ ಗುರುರ್ಬ್ರಹ್ಮಂದಿವ್ಯಯೋಗಪ್ರದಾಯಕಃ  ॥ ೧೪ ॥
ತತ್ರ ಗತ್ವಾ ಚ ತೇನೋಕ್ತವಿದ್ಯಾಂ ಸ।ನ್ಗೃಹ್ಯ ಖೇಚರೀಂ ।
ತೇನೋಕ್ತಃ ಸಮ್ಯಗಭ್ಯಾಸಂ ಕುರ್ಯಾದಾದಾವತಂದ್ರಿತಃ  ॥ ೧೫ ॥
ಅನಯಾ ವಿದ್ಯಯಾ ಯೋಗೀ ಖೇಚರೀಸಿದ್ಧಿಭಾಗ್ಭವೇತ್ ।
ಖೇಚರ್ಯಾ ಖೇಚರೀಂ ಯುಂಜನ್ಖೇಚರೀಬೀಜಪೂರಯಾ  ॥ ೧೬ ॥
ಖೇಚರಾಧಿಪತಿರ್ಭೂತ್ವಾ ಖೇಚರೇಷು ಸದಾ ವಸೇತ್ ।
ಖೇಚರಾವಸಥಂ ವಹ್ನಿಮಂಬುಮಂಡಲಭೂಷಿತಂ  ॥ ೧೭ ॥
ಆಖ್ಯಾತಂ ಖೇಚರೀಬೀಜಂ ತೇನ ಯೋಗಃ ಪ್ರಸಿಧ್ಯತಿ ।
ಸೋಮಾಂಶನವಕಂ ವರ್ಣಂ ಪ್ರತಿಲೋಮೇನ ಚೋದ್ಧರೇತ್  ॥ ೧೮ ॥
ತಸ್ಮಾತ್ತ್ರ್ಯಂಶಕಮಾಖ್ಯಾತಮಕ್ಷರಂ ಚಂದ್ರರೂಪಕಂ ।
ತಸ್ಮಾದಪ್ಯಷ್ಟಮಂ ವರ್ಣಂ ವಿಲೋಮೇನ ಪರಂ ಮುನೇ  ॥ ೧೯ ॥
ತಥಾ ತತ್ಪರಮಂ ವಿದ್ಧಿ ತದಾದಿರಪಿ ಪಂಚಮೀ ।
ಇಂದೋಶ್ಚ ಬಹುಭಿನ್ನೇ ಚ ಕೂಟೋಽಯಂ ಪರಿಕೀರ್ತಿತಃ  ॥ ೨೦ ॥
ಗುರೂಪದೇಶಲಭ್ಯಂ ಚ ಸರ್ವಯೋಗಪ್ರಸಿದ್ಧಿದಂ ।
ಯತ್ತಸ್ಯ ದೇಹಜಾ ಮಾಯಾ ನಿರುದ್ಧಕರಣಾಶ್ರಯಾ  ॥ ೨೧ ॥
ಸ್ವಪ್ನೇಽಪಿ ನ ಲಭೇತ್ತಸ್ಯ ನಿತ್ಯಂ ದ್ವಾದಶಜಪ್ಯತಃ ।
ಯ ಇಮಾಂ ಪಂಚ ಲಕ್ಷಾಣಿ ಜಪೇದಪಿ ಸುಯಂತ್ರಿತಃ  ॥ ೨೨ ॥
ತಸ್ಯ ಶ್ರೀಖೇಚರೀಸಿದ್ಧಿಃ ಸ್ವಯಮೇವ ಪ್ರವರ್ತತೇ ।
ನಶ್ಯಂತಿ ಸರ್ವವಿಘ್ನಾನಿ ಪ್ರಸೀದಂತಿ ಚ ದೇವತಾಃ  ॥ ೨೩ ॥
ವಲೀಪಲಿತನಾಶಶ್ಚ ಭವಿಷ್ಯತಿ ನ ಸಂಶಯಃ ।
ಏವಂ ಲಬ್ಧ್ವಾ ಮಹಾವಿದ್ಯಾಮಭ್ಯಾಸಂ ಕಾರಯೇತ್ತತಃ  ॥ ೨೪ ॥
ಅನ್ಯಥಾ ಕ್ಲಿಶ್ಯತೇ ಬ್ರಹ್ಮನ್ನ ಸಿದ್ಧಿಃ ಖೇಚರೀಪಥೇ ।
ಯದಭ್ಯಾಸವಿಧೌ ವಿದ್ಯಾಂ ನ ಲಭೇದ್ಯಃ ಸುಧಾಮಯೀಂ  ॥ ೨೫ ॥
ತತಃ ಸಂಮೇಲಕಾದೌ ಚ ಲಬ್ಧ್ವಾ ವಿದ್ಯಾಂ ಸದಾ ಜಪೇತ್ ।
ನಾನ್ಯಥಾ ರಹಿತೋ ಬ್ರಹ್ಮನ್ನ ಕಿಂಚಿತ್ಸಿದ್ಧಿಭಾಗ್ಭವೇತ್  ॥ ೨೬ ॥
ಯದಿದಂ ಲಭ್ಯತೇ ಶಾಸ್ತ್ರಂ ತದಾ ವಿದ್ಯಾಂ ಸಮಾಶ್ರಯೇತ್ ।
ತತಸ್ತದೋದಿತಾಂ ಸಿದ್ಧಿಮಾಶು ತಾಂ ಲಭತೇ ಮುನಿಃ  ॥ ೨೭ ॥
ತಾಲುಮೂಲಂ ಸಮುತ್ಕೃಷ್ಯ ಸಪ್ತವಾಸರಮಾತ್ಮವಿತ್ ।
ಸ್ವಗುರೂಕ್ತಪ್ರಕಾರೇಣ ಮಲಂ ಸರ್ವಂ ವಿಶೋಧಯೇತ್  ॥ ೨೮ ॥
ಸ್ನುಹಿಪತ್ರನಿಭಂ ಶಸ್ತ್ರಂ ಸುತೀಕ್ಷ್ಣಂ ಸ್ನಿಗ್ಧನಿರ್ಮಲಂ ।
ಸಮಾದಾಯ ತತಸ್ತೇನ ರೋಮಮಾತ್ರಂ ಸಮುಚ್ಛಿನೇತ್  ॥ ೨೯ ॥
ಹಿತ್ವಾ ಸೈಂಧವಪಥ್ಯಾಭ್ಯಾಂ ಚೂರ್ಣಿತಾಭ್ಯಾಂ ಪ್ರಕರ್ಷಯೇತ್ ।
ಪುನಃ ಸಪ್ತದಿನೇ ಪ್ರಾಪ್ತೇ ರೋಮಮಾತ್ರಂ ಸಮುಚ್ಛಿನೇತ್  ॥ ೩೦ ॥
ಏವಂ ಕ್ರಮೇಣ ಷಣ್ಮಾಸಂ ನಿತ್ಯೋದ್ಯುಕ್ತಃ ಸಮಾಚರೇತ್ ।
ಷಣ್ಮಾಸಾದ್ರಸನಾಮೂಲಂ ಸಿರಾಬದ್ಧಂ ಪ್ರಣಶ್ಯತಿ  ॥ ೩೧ ॥
ಅಥ ವಾಗೀಶ್ವರೀಧಾಮ ಶಿರೋ ವಸ್ತ್ರೇಣ ವೇಷ್ಟಯೇತ್ ।
ಶನೈರುತ್ಕರ್ಷಯೇದ್ಯೋಗೀ ಕಾಲವೇಲಾವಿಧಾನವಿತ್  ॥ ೩೨ ॥
ಪುನಃ ಷಣ್ಮಾಸಮಾತ್ರೇಣ ನಿತ್ಯಂ ಸಂಘರ್ಷಣಾನ್ಮುನೇ ।
ಭ್ರೂಮಧ್ಯಾವಧಿ ಚಾಪ್ಯೇತಿ ತಿರ್ಯಕ್ಕಣಬಿಲಾವಧಿಃ  ॥ ೩೩ ॥
ಅಧಶ್ಚ ಚುಬುಕಂ ಮೂಲಂ ಪ್ರಯಾತಿ ಕ್ರಮಚಾರಿತಾ ।
ಪುನಃ ಸಂವತ್ಸರಾಣಾಂ ತು ತೃತೀಯಾದೇವ ಲೀಲಯಾ  ॥ ೩೪ ॥
ಕೇಶಾಂತಮೂರ್ಧ್ವಂ ಕ್ರಮತಿ ತಿರ್ಯಕ್ ಶಾಖಾವಧಿರ್ಮುನೇ ।
ಅಧಸ್ತಾತ್ಕಂಠಕೂಪಾಂತಂ ಪುನರ್ವರ್ಷತ್ರಯೇಣ ತು  ॥ ೩೫ ॥
ಬ್ರಹ್ಮರಂಧ್ರಂ ಸಮಾವೃತ್ಯ ತಿಷ್ಠೇದೇವ ನ ಸಂಶಯಃ ।
ತಿರ್ಯಕ್ ಚೂಲಿತಲಂ ಯಾತಿ ಅಧಃ ಕಂಠಬಿಲಾವಧಿ  ॥ ೩೬ ॥
ಶನೈಃ ಶನೈರ್ಮಸ್ತಕಾಚ್ಚ ಮಹಾವಜ್ರಕಪಾಟಭಿತ್ ।
ಪೂರ್ವಂ ಬೀಜಯುತಾ ವಿದ್ಯಾ ಹ್ಯಾಖ್ಯಾತಾ ಯತಿದುರ್ಲಭಾ  ॥ ೩೭ ॥
ತಸ್ಯಾಃ ಷಡಂಗಂ ಕುರ್ವೀತ ತಯಾ ಷಟ್ಸ್ವರಭಿನ್ನಯಾ ।
ಕುರ್ಯಾದೇವಂ ಕರನ್ಯಾಸಂ ಸರ್ವಸಿದ್ಧ್ಯಾದಿಹೇತವೇ  ॥ ೩೮ ॥
ಶನೈರೇವಂ ಪ್ರಕರ್ತವ್ಯಮಭ್ಯಾಸಂ ಯುಗಪನ್ನಹಿ ।
ಯುಗಪದ್ವರ್ತತೇ ಯಸ್ಯ ಶರೀರಂ ವಿಲಯಂ ವ್ರಜೇತ್  ॥ ೩೯ ॥
ತಸ್ಮಾಚ್ಛನೈಃ ಶನೈಃ ಕಾರ್ಯಮಭ್ಯಾಸಂ ಮುನಿಪುಂಗವ ।
ಯದಾ ಚ ಬಾಹ್ಯಮಾರ್ಗೇಣ ಜಿಹ್ವಾ ಬ್ರಹ್ಮಬಿಲಂ ವ್ರಜೇತ್  ॥ ೪೦ ॥
ತದಾ ಬ್ರಹ್ಮಾರ್ಗಲಂ ಬ್ರಹ್ಮಂದುರ್ಭೇದ್ಯಂ ತ್ರಿದಶೈರಪಿ ।
ಅಂಗುಲ್ಯಗ್ರೇಣ ಸ।ನ್ಘೃಷ್ಯ ಜಿಹ್ವಾಮಾತ್ರಂ ನಿವೇಶಯೇತ್  ॥ ೪೧ ॥
ಏವಂ ವರ್ಷತ್ರಯಂ ಕೃತ್ವಾ ಬ್ರಹ್ಮದ್ವಾರಂ ಪ್ರವಿಶ್ಯತಿ ।
ಬ್ರಹ್ಮದ್ವಾರೇ ಪ್ರವಿಷ್ಟೇ ತು ಸಮ್ಯಙ್ಮಥನಮಾಚರೇತ್  ॥ ೪೨ ॥
ಮಥನೇನ ವಿನಾ ಕೇಚಿತ್ಸಾಧಯಂತಿ ವಿಪಶ್ಚಿತಃ ।
ಖೇಚರೀಮಂತ್ರಸಿದ್ಧಸ್ಯ ಸಿಧ್ಯತೇ ಮಥನಂ ವಿನಾ  ॥ ೪೩ ॥
ಜಪಂ ಚ ಮಥನಂ ಚೈವ ಕೃತ್ವಾ ಶೀಘ್ರಂ ಫಲಂ ಲಭೇತ್ ।
ಸ್ವರ್ಣಜಾಂ ರೌಪ್ಯಜಾಂ ವಾಪಿ ಲೋಹಜಾಂ ವಾ ಶಲಾಕಿಕಾಂ  ॥ ೪೪ ॥
ನಿಯೋಜ್ಯ ನಾಸಿಕಾರಂಧ್ರಂ ದುಗ್ಧಸಿಕ್ತೇನ ತಂತುನಾ ।
ಪ್ರಾಣಾನ್ನಿರುಧ್ಯ ಹೃದಯೇ ಸುಖಮಾಸನಮಾತ್ಮನಃ  ॥ ೪೫ ॥
ಶನೈಃ ಸುಮಥನಂ ಕುರ್ಯಾದ್ಭ್ರೂಮಧ್ಯೇ ನ್ಯಸ್ಯ ಚಕ್ಷುಷೀ ।
ಷಣ್ಮಾಸಂ ಮಥನಾವಸ್ಥಾ ಭಾವೇನೈವ ಪ್ರಜಾಯತೇ  ॥ ೪೬ ॥
ಯಥಾ ಸುಷುಪ್ತಿರ್ಬಾಲಾನಾಂ ಯಥಾ ಭಾವಸ್ತಥಾ ಭವೇತ್ ।
ನ ಸದಾ ಮಥನಂ ಶಸ್ತಂ ಮಾಸೇ ಮಾಸೇ ಸಮಾಚರೇತ್  ॥ ೪೭ ॥
ಸದಾ ರಸನಯಾ ಯೋಗೀ ಮಾರ್ಗಂ ನ ಪರಿಸ।ನ್ಕ್ರಮೇತ್ ।
ಏವಂ ದ್ವಾದಶವರ್ಷಾಂತೇ ಸಂಸಿದ್ಧಿರ್ಭವತಿ ಧ್ರುವಾ  ॥ ೪೮ ॥
ಶರೀರೇ ಸಕಲಂ ವಿಶ್ವಂ ಪಶ್ಯತ್ಯಾತ್ಮಾವಿಭೇದತಃ ।
ಬ್ರಹ್ಮಾಂಡೋಽಯಂ ಮಹಾಮಾರ್ಗೋ ರಾಜದಂತೋರ್ಧ್ವಕುಂಡಲೀ  ॥ ೪೯ ॥ ಇತಿ ॥
ಇತಿ ದ್ವಿತೀಯೋಽಧ್ಯಾಯಃ  ॥ ೨ ॥
ಮೇಲನಮನುಃ । ಹ್ರೀಂ ಭಂ ಸಂ ಪಂ ಫಂ ಸಂ ಕ್ಷಂ । ಪದ್ಮಜ ಉವಾಚ ।
ಅಮಾವಾಸ್ಯಾ ಚ ಪ್ರತಿಪತ್ಪೌರ್ಣಮಾಸೀ ಚ ಶಂಕರ ।
ಅಸ್ಯಾಃ ಕಾ ವರ್ಣ್ಯತೇ ಸ।ನ್ಜ್ಞಾ ಏತದಾಖ್ಯಾಹಿ ತತ್ತ್ವತಃ  ॥ ೧ ॥
ಪ್ರತಿಪದ್ದಿನತೋಽಕಾಲೇ ಅಮಾವಾಸ್ಯಾ ತಥೈವ ಚ ।
ಪೌರ್ಣಮಾಸ್ಯಾಂ ಸ್ಥಿರೀಕುರ್ಯಾತ್ಸ ಚ ಪಂಥಾ ಹಿ ನಾನ್ಯಥಾ  ॥ ೨ ॥
ಕಾಮೇನ ವಿಷಯಾಕಾಂಕ್ಷೀ ವಿಷಯಾತ್ಕಾಮಮೋಹಿತಃ ।
ದ್ವಾವೇವ ಸಂತ್ಯಜೇನ್ನಿತ್ಯಂ ನಿರಂಜನಮುಪಾಶ್ರಯೇತ್  ॥ ೩ ॥
ಅಪರಂ ಸಂತ್ಯಜೇತ್ಸರ್ವಂ ಯದಿಚ್ಛೇದಾತ್ಮನೋ ಹಿತಂ ।
ಶಕ್ತಿಮಧ್ಯೇ ಮನಃ ಕೃತ್ವಾ ಮನಃ ಶಕ್ತೇಶ್ಚ ಮಧ್ಯಗಂ  ॥ ೪ ॥
ಮನಸಾ ಮನ ಆಲೋಕ್ಯ ತತ್ತ್ಯಜೇತ್ಪರಮಂ ಪದಂ ।
ಮನ ಏವ ಹಿ ಬಿಂದುಶ್ಚ ಉತ್ಪತ್ತಿಸ್ಥಿತಿಕಾರಣಂ  ॥ ೫ ॥
ಮನಸೋತ್ಪದ್ಯತೇ ಬಿಂದುರ್ಯಥಾ ಕ್ಷೀರಂ ಘೃತಾತ್ಮಕಂ ।
ನ ಚ ಬಂಧನಮಧ್ಯಸ್ಥಂ ತದ್ವೈ ಕಾರಣಮಾನಸಂ  ॥ ೬ ॥
ಚಂದ್ರಾರ್ಕಮಧ್ಯಮಾ ಶಕ್ತಿರ್ಯತ್ರಸ್ಥಾ ತತ್ರ ಬಂಧನಂ ।
ಜ್ಞಾತ್ವಾ ಸುಷುಮ್ನಾಂ ತದ್ಭೇದಂ ಕೃತ್ವಾ ವಾಯುಂ ಚ ಮಧ್ಯಗಂ  ॥ ೭ ॥
ಸ್ಥಿತ್ವಾಸೌ ಬೈಂದವಸ್ಥಾನೇ ಘ್ರಾಣರಂಧ್ರೇ ನಿರೋಧಯೇತ್ ।
ವಾಯುಂ ಬಿಂದುಂ ಸಮಾಖ್ಯಾತಂ ಸತ್ತ್ವಂ ಪ್ರಕೃತಿಮೇವ ಚ  ॥ ೮ ॥
ಷಟ್ ಚಕ್ರಾಣಿ ಪರಿಜ್ಞಾತ್ವಾ ಪ್ರವಿಶೇತ್ಸುಖಮಂಡಲಂ ।
ಮೂಲಾಧಾರಂ ಸ್ವಾಧಿಷ್ಠಾನಂ ಮಣಿಪೂರಂ ತೃತೀಯಕಂ  ॥ ೯ ॥
ಅನಾಹತಂ ವಿಶುದ್ಧಂ ಚ ಆಜ್ಞಾಚಕ್ರಂ ಚ ಷಷ್ಠಕಂ ।
ಆಧಾರಂ ಗುದಮಿತ್ಯುಕ್ತಂ ಸ್ವಾಧಿಷ್ಠಾನಂ ತು ಲೈಂಗಿಕಂ  ॥ ೧೦ ॥
ಮಣಿಪೂರಂ ನಾಭಿದೇಶಂ ಹೃದಯಸ್ಥಮನಾಹತಂ ।
ವಿಶುದ್ಧಿಃ ಕಂಠಮೂಲೇ ಚ ಆಜ್ಞಾಚಕ್ರಂ ಚ ಮಸ್ತಕಂ  ॥ ೧೧ ॥
ಷಟ್ ಚಕ್ರಾಣಿ ಪರಿಜ್ಞಾತ್ವಾ ಪ್ರವಿಶೇತ್ಸುಖಮಂಡಲೇ ।
ಪ್ರವಿಶೇದ್ವಾಯುಮಾಕೃಷ್ಯ ತಯೈವೋರ್ಧ್ವಂ ನಿಯೋಜಯೇತ್  ॥ ೧೨ ॥
ಏವಂ ಸಮಭ್ಯಸೇದ್ವಾಯುಂ ಸ ಬ್ರಹ್ಮಾಂಡಮಯೋ ಭವೇತ್ ।
ವಾಯುಂ ಬಿಂದುಂ ತಥಾ ಚಕ್ರಂ ಚಿತ್ತಂ ಚೈವ ಸಮಭ್ಯಸೇತ್  ॥ ೧೩ ॥
ಸಮಾಧಿಮೇಕೇನ ಸಮಮಮೃತಂ ಯಾಂತಿ ಯೋಗಿನಃ ।
ಯಥಾಗ್ನಿರ್ದಾರುಮಧ್ಯಸ್ಥೋ ನೋತ್ತಿಷ್ಠೇನ್ಮಥನಂ ವಿನಾ  ॥ ೧೪ ॥
ವಿನಾ ಚಾಭ್ಯಾಸಯೋಗೇನ ಜ್ಞಾನದೀಪಸ್ತಥಾ ನ ಹಿ ।
ಘಟಮಧ್ಯಗತೋ ದೀಪೋ ಬಾಹ್ಯೇ ನೈವ ಪ್ರಕಾಶತೇ  ॥ ೧೫ ॥
ಭಿನ್ನೇ ತಸ್ಮಿನ್ಘಟೇ ಚೈವ ದೀಪಜ್ವಾಲಾ ಚ ಭಾಸತೇ ।
ಸ್ವಕಾಯಂ ಘಟಮಿತ್ಯುಕ್ತಂ ಯಥಾ ದೀಪೋ ಹಿ ತತ್ಪದಂ  ॥ ೧೬ ॥
ಗುರುವಾಕ್ಯಸಮಾಭಿನ್ನೇ ಬ್ರಹ್ಮಜ್ಞಾನಂ ಸ್ಫುಟೀಭವೇತ್ ।
ಕರ್ಣಧಾರಂ ಗುರುಂ ಪ್ರಾಪ್ಯ ಕೃತ್ವಾ ಸೂಕ್ಷ್ಮಂ ತರಂತಿ ಚ  ॥ ೧೭ ॥
ಅಭ್ಯಾಸವಾಸನಾಶಕ್ತ್ಯಾ ತರಂತಿ ಭವಸಾಗರಂ ।
ಪರಾಯಾಮಂಕುರೀಭೂಯ ಪಶ್ಯಂತಾಂ ದ್ವಿದಲೀಕೃತಾ  ॥ ೧೮ ॥
ಮಧ್ಯಮಾಯಾಂ ಮುಕುಲಿತಾ ವೈಖರ್ಯಾಂ ವಿಕಸೀಕೃತಾ ।
ಪೂರ್ವಂ ಯಥೋದಿತಾ ಯಾ ವಾಗ್ವಿಲೋಮೇನಾಸ್ತಗಾ ಭವೇತ್  ॥ ೧೯ ॥
ತಸ್ಯಾ ವಾಚಃ ಪರೋ ದೇವಃ ಕೂಟಸ್ಥೋ ವಾಕ್ಪ್ರಬೋಧಕಃ ।
ಸೋಽಹಮಸ್ಮೀತಿ ನಿಶ್ಚಿತ್ಯ ಯಃ ಸದಾ ವರ್ತತೇ ಪುಮಾನ್  ॥ ೨೦ ॥
ಶಬ್ದೈರುಚ್ಚಾವಚೈರ್ನೀಚೈರ್ಭಾಷಿತೋಽಪಿ ನ ಲಿಪ್ಯತೇ ।
ವಿಶ್ವಶ್ಚ ತೈಜಸಶ್ಚೈವ ಪ್ರಾಜ್ಞಶ್ಚೇತಿ ಚ ತೇ ತ್ರಯಃ  ॥ ೨೧ ॥
ವಿರಾಡ್ಢಿರಣ್ಯಗರ್ಭಶ್ಚ ಈಶ್ವರಶ್ಚೇತಿ ತೇ ತ್ರಯಃ ।
ಬ್ರಹ್ಮಾಂಡಂ ಚೈವ ಪಿಂಡಾಂಡಂ ಲೋಕಾ ಭೂರಾದಯಃ ಕ್ರಮಾತ್  ॥ ೨೨ ॥
ಸ್ವಸ್ವೋಪಾಧಿಲಯಾದೇವ ಲೀಯಂತೇ ಪ್ರತ್ಯಗಾತ್ಮನಿ ।
ಅಂಡಂ ಜ್ಞಾನಾಗ್ನಿನಾ ತಪ್ತಂ ಲೀಯತೇ ಕಾರಣೈಃ ಸಹ  ॥ ೨೩ ॥
ಪರಮಾತ್ಮನಿ ಲೀನಂ ತತ್ಪರಂ ಬ್ರಹ್ಮೈವ ಜಾಯತೇ ।
ತತಃ ಸ್ತಿಮಿತಗಂಭೀರಂ ನ ತೇಜೋ ನ ತಮಸ್ತತಂ  ॥ ೨೪ ॥
ಅನಾಖ್ಯಮನಭಿವ್ಯಕ್ತಂ ಸತ್ಕಿಂಚಿದವಶಿಷ್ಯತೇ ।
ಧ್ಯಾತ್ವಾ ಮಧ್ಯಸ್ಥಮಾತ್ಮಾನಂ ಕಲಶಾಂತರದೀಪವತ್  ॥ ೨೫ ॥
ಅಂಗುಷ್ಠಮಾತ್ರಮಾತ್ಮಾನಮಧೂಮಜ್ಯೋತಿರೂಪಕಂ ।
ಪ್ರಕಾಶಯಂತಮಂತಸ್ಥಂ ಧ್ಯಾಯೇತ್ಕೂಟಸ್ಥಮವ್ಯಯಂ  ॥ ೨೬ ॥
ವಿಜ್ಞಾನಾತ್ಮಾ ತಥಾ ದೇಹೇ ಜಾಗ್ರತ್ಸ್ವಪ್ನಸುಷುಪ್ತಿತಃ ।
ಮಾಯಯಾ ಮೋಹಿತಃ ಪಶ್ಚಾದ್ಬಹುಜನ್ಮಾಂತರೇ ಪುನಃ  ॥ ೨೭ ॥
ಸತ್ಕರ್ಮಪರಿಪಾಕಾತ್ತು ಸ್ವವಿಕಾರಂ ಚಿಕೀರ್ಷತಿ ।
ಕೋಽಹಂ ಕಥಮಯಂ ದೋಷಃ ಸಂಸಾರಾಖ್ಯ ಉಪಾಗತಃ  ॥ ೨೮ ॥
ಜಾಗ್ರತ್ಸ್ವಪ್ನೇ ವ್ಯವಹರಂತ್ಸುಷುಪ್ತೌ ಕ್ವ ಗತಿರ್ಮಮ ।
ಇತಿ ಚಿಂತಾಪರೋ ಭೂತ್ವಾ ಸ್ವಭಾಸಾ ಚ ವಿಶೇಷತಃ  ॥ ೨೯ ॥
ಅಜ್ಞಾನಾತ್ತು ಚಿದಾಭಾಸೋ ಬಹಿಸ್ತಾಪೇನ ತಾಪಿತಃ ।
ದಗ್ಧಂ ಭವತ್ಯೇವ ತದಾ ತೂಲಪಿಂಡಮಿವಾಗ್ನಿನಾ  ॥ ೩೦ ॥
ದಹರಸ್ಥಃ ಪ್ರತ್ಯಗಾತ್ಮಾ ನಷ್ಟೇ ಜ್ಞಾನೇ ತತಃ ಪರಂ ।
ವಿತತೋ ವ್ಯಾಪ್ಯ ವಿಜ್ಞಾನಂ ದಹತ್ಯೇವ ಕ್ಷಣೇನ ತು  ॥ ೩೧ ॥
ಮನೋಮಯಜ್ಞಾನಮಯಾನ್ಸಮ್ಯಗ್ದಗ್ಧ್ವಾ ಕ್ರಮೇಣ ತು ।
ಘಟಸ್ಥದೀಪವಚ್ಛಶ್ವದಂತರೇವ ಪ್ರಕಾಶತೇ  ॥ ೩೨ ॥
ಧ್ಯಾಯನ್ನಾಸ್ತೇ ಮುನಿಶ್ಚೈವಮಾಸುಪ್ತೇರಾಮೃತೇಸ್ತು ಯಃ ।
ಜೀವನ್ಮುಕ್ತಃ ಸ ವಿಜ್ಞೇಯಃ ಸ ಧನ್ಯಃ ಕೃತಕೃತ್ಯವಾನ್  ॥ ೩೩ ॥
ಜೀವನ್ಮುಕ್ತಪದಂ ತ್ಯಕ್ತ್ವಾ ಸ್ವದೇಹೇ ಕಾಲಸಾತ್ಕೃತೇ ।
ವಿಶತ್ಯದೇಹಮುಕ್ತತ್ವಂ ಪವನೋಽಸ್ಪಂದತಾಮಿವ  ॥ ೩೪ ॥
ಅಶಬ್ದಮಸ್ಪರ್ಶಮರೂಪಮವ್ಯಯಂ
ತಥಾರಸಂ ನಿತ್ಯಮಗಂಧವಚ್ಚ ಯತ್ ।
ಅನಾದ್ಯನಂತಂ ಮಹತಃ ಪರಂ ಧ್ರುವಂ
ತದೇವ ಶಿಷ್ಯತ್ಯಮಲಂ ನಿರಾಮಯಂ  ॥ ೩೫ ॥ ಇತ್ಯುಪನಿಷತ್  ॥
ಓಂ ಸಹ ನಾವವತು ಸಹ ನೌ ಭುನಕ್ತು ಸಹ ವೀರ್ಯಂ ಕರವಾವಹೈ ।
ತೇಜಸ್ವಿನಾವಧೀತಮಸ್ತು ಮಾ ವಿದ್ವಿಷಾವಹೈ  ॥
ಓಂ ಶಾಂತಿಃ ಶಾಂತಿಃ ಶಾಂತಿಃ  ॥ ಹರಿಃ ಓಂ ತತ್ಸತ್  ॥
ಇತಿ ಯೋಗಕುಂಡಲಿನ್ಯುಪನಿಷತ್ಸಮಾಪ್ತಾ  ॥

॥ ಶಬ್ದಾರ್ಥ ಮತ್ತು ಭಾವಾರ್ಥ ॥

ಶಬ್ದಾರ್ಥ:
• ಕುಂಡಲಿನಿ ಶಕ್ತಿ - ಜೀವಾತ್ಮನ ಮೂಲ ಚೈತನ್ಯ ಶಕ್ತಿ;
• ಸುಷುಮ್ನಾ - ಬೆನ್ನುಮೂಳೆಯ ಮಧ್ಯದಲ್ಲಿರುವ ನಾಡಿ;
• ಖೇಚರೀ ಮುದ್ರಾ - ಯೋಗದ ಒಂದು ಪ್ರಮುಖ ಮುದ್ರೆ;
• ಸಮಾಧಿಃ - ಯೋಗದ ಪರಮ ಸ್ಥಿತಿ.

ಭಾವಾರ್ಥ:
ಯೋಗಕುಂಡಲಿನ್ಯುಪನಿಷತ್ತು ಕೃಷ್ಣ ಯಜುರ್ವೇದಕ್ಕೆ ಸೇರಿದ್ದು, ಪ್ರಾಣಯಾಮ, ಹಠಯೋಗ ಮತ್ತು ಲಯಯೋಗದ ತಂತ್ರಗಳ ಮೂಲಕ ಕುಂಡಲಿನಿ ಶಕ್ತಿಯನ್ನು ಜಾಗೃತಗೊಳಿಸಿ, ಸುಷುಮ್ನಾ ನಾಡಿಯ ಮೂಲಕ ಬ್ರಹ್ಮರಂಧ್ರಕ್ಕೆ ತಲುಪಿಸಿ ಮುಕ್ತಿ ಪಡೆಯುವ ಮಾರ್ಗವನ್ನು ಬೋಧಿಸುತ್ತದೆ.

೧. ಕುಂಡಲಿನಿ ಜಾಗೃತಿ:
ಪ್ರಾಣಾಯಾಮ ಮತ್ತು ಬಂಧಗಳ ಅಭ್ಯಾಸದಿಂದ ಮೂಲಾಧಾರದಲ್ಲಿರುವ ಕುಂಡಲಿನಿ ಶಕ್ತಿಯನ್ನು ಎಚ್ಚರಗೊಳಿಸಿ ಷಟ್ಚಕ್ರಗಳನ್ನು ಭೇದಿಸಿ ಸಹಸ್ರಾರದಲ್ಲಿ ಶಿವನೊಂದಿಗೆ ಐಕ್ಯಗೊಳಿಸಬೇಕು.

೨. ಯೋಗ ಸಿದ್ಧಿ:
ಖೇಚರೀ ಮುದ್ರೆಯ ಅಭ್ಯಾಸವು ಸಾಧಕನಿಗೆ ರೋಗರುಜಿನಗಳಿಂದ ಮುಕ್ತಿ ನೀಡಿ ಮನಸ್ಸನ್ನು ಸಮಾಧಿ ಸ್ಥಿತಿಗೆ ಕೊಂಡೊಯ್ಯುತ್ತದೆ.
