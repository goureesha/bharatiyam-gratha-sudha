ಯದ್ಬ್ರಹ್ಮ ರುದ್ರಹೃದಯಮಹಾವಿದ್ಯಾಪ್ರಕಾಶಿತಂ।ಹ್ ।
ತದ್ಬ್ರಹ್ಮಮಾತ್ರಾವಸ್ಥಾನಪದವೀಮಧುನಾ ಭಜೇ  ॥
ಓಂ ಸಹ ನಾವವತು  ॥ ಸಹ ನೌ ಭುನಕ್ತು  ॥ ಸಹ ವೀರ್ಯಂ ಕರವಾವಹೈ  ॥
ತೇಜಸ್ವಿನಾವಧೀತಮಸ್ತು ಮಾ ವಿದ್ವಿಷಾವಹೈ  ॥
ಓಂ ಶಾಂತಿಃ ಶಾಂತಿಃ ಶಾಂತಿಃ  ॥
ಹರಿಃ ಓಂ  ॥
ಹೃದಯಂ ಕುಂಡಲೀ ಭಸ್ಮರುದ್ರಾಕ್ಷಗಣದರ್ಶನಂ।ಹ್ ।
ತಾರಸಾರಂ ಮಹಾವಾಕ್ಯಂ ಪಂಚಬ್ರಹ್ಮಾಗ್ನಿಹೋತ್ರಕಂ।ಹ್  ॥ ೧ ॥
ಪ್ರಣಮ್ಯ ಶಿರಸಾ ಪಾದೌ ಶುಕೋ ವ್ಯಾಸಮುವಾಚ ಹ ।
ಕೋ ದೇವಃ ಸರ್ವದೇವೇಷು ಕಸ್ಮಿಂದೇವಾಶ್ಚ ಸರ್ವಶಃ  ॥ ೨ ॥
ಕಸ್ಯ ಶುಶ್ರೂಷಣಾನ್ನಿತ್ಯಂ ಪ್ರೀತಾ ದೇವಾ ಭವಂತಿ ಮೇ ।
ತಸ್ಯ ತದ್ವಚನಂ ಶ್ರುತ್ವಾ ಪ್ರತ್ಯುವಾಚ ಪಿತಾ ಶುಕಂ।ಹ್  ॥ ೩ ॥
ಸರ್ವದೇವಾತ್ಮಕೋ ರುದ್ರಃ ಸರ್ವೇ ದೇವಾಃ ಶಿವಾತ್ಮಕಾಃ ।
ರುದ್ರಸ್ಯ ದಕ್ಷಿಣೇ ಪಾರ್ಶ್ವೇ ರವಿರ್ಬ್ರಹ್ಮಾ ತ್ರಯೋಽಗ್ನಯಃ  ॥ ೪ ॥
ವಾಮಪಾರ್ಶ್ವೇ ಉಮಾ ದೇವೀ ವಿಷ್ಣುಃ ಸೋಮೋಽಪಿ ತೇ ತ್ರಯಃ ।
ಯಾ ಉಮಾ ಸಾ ಸ್ವಯಂ ವಿಷ್ಣುರ್ಯೋ ವಿಷ್ಣುಃ ಸ ಹಿ ಚಂದ್ರಮಾಃ  ॥ ೫ ॥
ಯೇ ನಮಸ್ಯಂತಿ ಗೋವಿಂದಂ ತೇ ನಮಸ್ಯಂತಿ ಶಂಕರಂ।ಹ್ ।
ಯೇಽರ್ಚಯಂತಿ ಹರಿಂ ಭಕ್ತ್ಯಾ ತೇಽರ್ಚಯಂತಿ ವೃಷಧ್ವಜಂ।ಹ್  ॥ ೬ ॥
ಯೇ ದ್ವಿಷಂತಿ ವಿರೂಪಾಕ್ಷಂ ತೇ ದ್ವಿಷಂತಿ ಜನಾರ್ದನಂ।ಹ್ ।
ಯೇ ರುದ್ರಂ ನಾಭಿಜಾನಂತಿ ತೇ ನ ಜಾನಂತಿ ಕೇಶವಂ।ಹ್  ॥ ೭ ॥
ರುದ್ರಾತ್ಪ್ರವರ್ತತೇ ಬೀಜಂ ಬೀಜಯೋನಿರ್ಜನಾರ್ದನಃ ।
ಯೋ ರುದ್ರಃ ಸ ಸ್ವಯಂ ಬ್ರಹ್ಮಾ ಯೋ ಬ್ರಹ್ಮಾ ಸ ಹುತಾಶನಃ  ॥ ೮ ॥
ಬ್ರಹ್ಮವಿಷ್ಣುಮಯೋ ರುದ್ರ ಅಗ್ನೀಷೋಮಾತ್ಕಂ ಜಗತ್।ಹ್ ।
ಪುಂಲಿಂಗಂ ಸರ್ವಮೀಶಾನಂ ಸ್ತ್ರೀಲಿಂಗಂ ಭಗವತ್ಯುಮಾ  ॥ ೯ ॥
ಉಮಾರುದ್ರಾತ್ಮಿಕಾಃ ಸರ್ವಾಃ ಗ್ರಜಾಃ ಸ್ಥಾವರಜಂಗಮಾಃ ।
ವ್ಯಕ್ತಂ ಸರ್ವಮುಮಾರೂಪಮವ್ಯಕ್ತಂ ತು ಮಹೇಶ್ವರಂ।ಹ್  ॥ ೧೦ ॥
ಉಮಾ ಶಂಕರಯೋಗೋ ಯಃ ಸ ಯೋಗೋ ವಿಷ್ಣುರುಚ್ಯತೇ ।
ಯಸ್ತು ತಸ್ಮೈ ನಮಸ್ಕಾರಂ ಕುರ್ಯಾದ್ಭಕ್ತಿಸಮನ್ವಿತಃ  ॥ ೧೧ ॥
ಆತ್ಮಾನಂ ಪರಮಾತ್ಮಾನಮಂತರಾತ್ಮಾನಮೇವ ಚ ।
ಜ್ಞಾತ್ವಾ ತ್ರಿವಿಧಮಾತ್ಮಾನಂ ಪರಮಾತ್ಮಾನಮಾಶ್ರಯೇತ್।ಹ್  ॥ ೧೨ ॥
ಅಂತರಾತ್ಮಾ ಭವೇದ್ಬ್ರಹ್ಮಾ ಪರಮಾತ್ಮಾ ಮಹೇಶ್ವರಃ ।
ಸರ್ವೇಷಾಮೇವ ಭೂತಾನಾಂ ವಿಷ್ಣುರಾತ್ಮಾ ಸನಾತನಃ  ॥ ೧೩ ॥
ಅಸ್ಯ ತ್ರೈಲೋಕ್ಯವೃಕ್ಷಸ್ಯ ಭೂಮೌ ವಿಟಪಶಾಖಿನಃ ।
ಅಗ್ರಂ ಮಧ್ಯಂ ತಥಾ ಮೂಲಂ ವಿಷ್ಣುಬ್ರಹ್ಮಮಹೇಶ್ವರಾಃ  ॥ ೧೪ ॥
ಕಾರ್ಯಂ ವಿಷ್ಣುಃ ಕ್ರಿಯಾ ಬ್ರಹ್ಮಾ ಕಾರಣಂ ತು ಮಹೇಶ್ವರಃ ।
ಪ್ರಯೋಜನಾರ್ಥಂ ರುದ್ರೇಣ ಮೂರ್ತಿರೇಕಾ ತ್ರಿಧಾ ಕೃತಾ  ॥ ೧೫ ॥
ಧರ್ಮೋ ರುದ್ರೋ ಜಗದ್ವಿಷ್ಣುಃ ಸರ್ವಜ್ಞಾನಂ ಪಿತಾಮಹಃ ।
ಶ್ರೀರುದ್ರ ರುದ್ರ ರುದ್ರೇತಿ ಯಸ್ತಂ ಬ್ರೂಯಾದ್ವಿಚಕ್ಷಣಃ  ॥ ೧೬ ॥
ಕೀರ್ತನಾತ್ಸರ್ವದೇವಸ್ಯ ಸರ್ವಪಾಪೈಃ ಪ್ರಮುಚ್ಯತೇ ।
ರುದ್ರೋ ನರ ಉಮಾ ನಾರೀ ತಸ್ಮೈ ತಸ್ಯೈ ನಮೋ ನಮಃ  ॥ ೧೭ ॥
ರುದ್ರೋ ಬ್ರಹ್ಮಾ ಉಮಾ ವಾಣೀ ತಸ್ಮೈ ತಸ್ಯೈ ನಮೋ ನಮಃ ।
ರುದ್ರೋ ವಿಷ್ಣುರುಮಾ ಲಕ್ಷ್ಮೀಸ್ತಸ್ಮೈ ತಸ್ಯೈ ನಮೋ ನಮಃ  ॥ ೧೮ ॥
ರುದ್ರಃ ಸೂರ್ಯ ಉಮಾ ಛಾಯಾ ತಸ್ಮೈ ತಸ್ಯೈ ನಮೋ ನಮಃ ।
ರುದ್ರಃ ಸೋಮ ಉಮಾ ತಾರಾ ತಸ್ಮೈ ತಸ್ಯೈ ನಮೋ ನಮಃ  ॥ ೧೯ ॥
ರುದ್ರೋ ದಿವಾ ಉಮಾ ರಾತ್ರಿಸ್ತಸ್ಮೈ ತಸ್ಯೈ ನಮೋ ನಮಃ ।
ರುದ್ರೋ ಯಜ್ಞ ಉಮಾ ವೇದಿಸ್ತಸ್ಮೈ ತಸ್ಯೈ ನಮೋ ನಮಃ  ॥ ೨೦ ॥
ರುದ್ರೋ ವಹ್ನಿರುಮಾ ಸ್ವಾಹಾ ತಸ್ಮೈ ತಸ್ಯೈ ನಮೋ ನಮಃ ।
ರುದ್ರೋ ವೇದ ಉಮಾ ಶಾಸ್ತಂ ತಸ್ಮೈ ತಸ್ಯೈ ನಮೋ ನಮಃ  ॥ ೨೧ ॥
ರುದ್ರೋ ವೃಕ್ಷ ಉಮಾ ವಲ್ಲೀ ತಸ್ಮೈ ತಸ್ಯೈ ನಮೋ ನಮಃ ।
ರುದ್ರೋ ಗಂಧ ಉಮಾ ಪುಷ್ಪಂ ತಸ್ಮೈ ತಸ್ಯೈ ನಮೋ ನಮಃ  ॥ ೨೨ ॥
ರುದ್ರೋಽರ್ಥ ಅಕ್ಷರಃ ಸೋಮಾ ತಸ್ಮೈ ತಸ್ಯೈ ನಮೋ ನಮಃ ।
ರುದ್ರೋ ಲಿಂಗಮುಮಾ ಪೀಠಂ ತಸ್ಮೈ ತಸ್ಯೈ ನಮೋ ನಮಃ  ॥ ೨೩ ॥
ಸರ್ವದೇವಾತ್ಮಕಂ ರುದ್ರಂ ನಮಸ್ಕುರ್ಯಾತ್ಪೃಥಕ್ಪೃಥಕ್।ಹ್ ।
ಏಭಿರ್ಮಂತ್ರಪದೈರೇವ ನಮಸ್ಯಾಮೀಶಪಾರ್ವತೀ  ॥ ೨೪ ॥
ಯತ್ರ ಯತ್ರ ಭವೇತ್ಸಾರ್ಧಮಿಮಂ ಮಂತ್ರಮುದೀರಯೇತ್।ಹ್ ।
ಬ್ರಹ್ಮಹಾ ಜಲಮಧ್ಯೇ ತು ಸರ್ವಪಾಪೈಃ ಪ್ರಮುಚ್ಯತೇ  ॥ ೨೫ ॥
ಸರ್ವಾಧಿಷ್ಠಾನಮದ್ವಂದ್ವಂ ಪರಂ ಬ್ರಹ್ಮ ಸನಾತನಂ।ಹ್ ।
ಸಚ್ಚಿದಾನಂದರೂಪಂ ತದವಾಙ್ಮನಸಗೋಚರಂ।ಹ್  ॥ ೨೬ ॥
ತಸ್ಮಿನ್ಸುವಿದಿತೇ ಸರ್ವಂ ವಿಜ್ಞಾತಂ ಸ್ಯಾದಿದಂ ಶುಕ ।
ತದಾತ್ಮಕತ್ವಾತ್ಸರ್ವಸ್ಯ ತಸ್ಮಾದ್ಭಿನ್ನಂ ನಹಿ ಕ್ವಚಿತ್।ಹ್  ॥ ೨೭ ॥
ದ್ವೇ ವಿದ್ಯೇ ವೇದಿತವ್ಯೇ ಹಿ ಪರಾ ಚೈವಾಪರಾ ಚ ತೇ ।
ತತ್ರಾಪರಾ ತು ವಿದ್ಯೈಷಾ ಋಗ್ವೇದೋ ಯಜುರೇವ ಚ  ॥ ೨೮ ॥
ಸಾಮವೇದಸ್ತಥಾಥರ್ವವೇದಃ ಶಿಕ್ಷಾ ಮುನೀಶ್ವರ ।
ಕಲ್ಪೋ ವ್ಯಾಕರಣಂ ಚೈವ ನಿರುಕ್ತಂ ಛಂದ ಏವ ಚ  ॥ ೨೯ ॥
ಜ್ಯೋತಿಷಂ ಚ ಯಥಾ ನಾತ್ಮವಿಷಯಾ ಅಪಿ ಬುದ್ಧಯಃ ।
ಅಥೈಷಾ ಪರಮಾ ವಿದ್ಯಾ ಯಯಾತ್ಮಾ ಪರಮಾಕ್ಷರಂ।ಹ್  ॥ ೩೦ ॥
ಯತ್ತದದ್ರೇಶ್ಯಮಗ್ರಾಹ್ಯಮಗೋತ್ರಂ ರೂಪವರ್ಜಿತಂ।ಹ್ ।
ಅಚಕ್ಷುಃಶ್ರೋತ್ರಮತ್ಯರ್ಥಂ ತದಪಾಣಿಪದಂ ತಥಾ  ॥ ೩೧ ॥
ನಿತ್ಯಂ ವಿಭುಂ ಸರ್ವಗತಂ ಸುಸೂಕ್ಷ್ಮಂ ಚ ತದವ್ಯಯಂ।ಹ್ ।
ತದ್ಭೂತಯೋನಿಂ ಪಶ್ಯಂತಿ ಧೀರಾ ಆತ್ಮಾನಮಾತ್ಮನಿ  ॥ ೩೨ ॥
ಯಃ ಸರ್ವಜ್ಞಃ ಸರ್ವವಿದ್ಯೋ ಯಸ್ಯ ಜ್ಞಾನಮಯಂ ತಪಃ ।
ತಸ್ಮಾದತ್ರಾನ್ನರೂಪೇಣ ಜಾಯತೇ ಜಗದಾವಲಿಃ  ॥ ೩೩ ॥
ಸತ್ಯವದ್ಭಾತಿ ತತ್ಸರ್ವಂ ರಜ್ಜುಸರ್ಪವದಾಸ್ಥಿತಂ।ಹ್ ।
ತದೇತದಕ್ಷರಂ ಸತ್ಯಂ ತದ್ವಿಜ್ಞಾಯ ವಿಮುಚ್ಯತೇ  ॥ ೩೪ ॥
ಜ್ಞಾನೇನೈವ ಹಿ ಸಂಸಾರವಿನಾಶೋ ನೈವ ಕರ್ಮಣಾ ।
ಶ್ರೋತ್ರಿಯಂ ಬ್ರಹ್ಮನಿಷ್ಠಂ ಸ್ವಗುರುಂ ಗಚ್ಛೇದ್ಯಥಾವಿಧಿ  ॥ ೩೫ ॥
ಗುರುಸ್ತಸ್ಮೈ ಪರಾಂ ವಿದ್ಯಾಂ ದದ್ಯಾದ್ಬ್ರಹ್ಮಾತ್ಮಬೋಧಿನೀಂ।ಹ್ ।
ಗುಹಾಯಾಂ ನಿಹಿತಂ ಸಾಕ್ಷಾದಕ್ಷರಂ ವೇದ ಚೇನ್ನರಃ  ॥ ೩೬ ॥
ಛಿತ್ವಾಽವಿದ್ಯಾಮಹಾಗ್ರಂಥಿಂ ಶಿವಂ ಗಚ್ಛೇತ್ಸನಾತನಂ।ಹ್ ।
ತದೇತದಮೃತಂ ಸತ್ಯಂ ತದ್ಬೋದ್ಧವ್ಯಂ ಮುಮುಕ್ಷಿಭಿಃ  ॥ ೩೭ ॥
ಧನುಸ್ತಾರಂ ಶರೋ ಹ್ಯಾತ್ಮಾ ಬ್ರಹ್ಮ ತಲ್ಲಕ್ಷ್ಯಮುಚ್ಯತೇ ।
ಅಪ್ರಮತ್ತೇನ ವೇದ್ಧವ್ಯಂ ಶರವತ್ತನ್ಮಯೋ ಭವೇತ್।ಹ್  ॥ ೩೮ ॥
ಲಕ್ಷ್ಯಂ ಸರ್ವಗತಂ ಚೈವ ಶರಃ ಸರ್ವಗತೋ ಮುಖಃ ।
ವೇದ್ಧಾ ಸರ್ವಗತಶ್ಚೈವ ಶಿವಲಕ್ಷ್ಯಂ ನ ಸಂಶಯಃ  ॥ ೩೯ ॥
ನ ತತ್ರ ಚಂದ್ರಾರ್ಕವಪುಃ ಪ್ರಕಾಶತೇ
ನ ವಾಂತಿ ವಾತಾಃ ಸಕಲಾ ದೇವತಾಶ್ಚ ।
ಸ ಏಷ ದೇವಃ ಕೃತಭಾವಭೂತಃ
ಸ್ವಯಂ ವಿಶುದ್ಧೋ ವಿರಜಃ ಪ್ರಕಾಶತೇ  ॥ ೪೦ ॥
ದ್ವೌ ಸುಪರ್ಣೌ ಶರೀರೇಽಸ್ಮಿಂಜೀವೇಶಾಕ್ಷ್ಯೌ ಸಹ ಸ್ಥಿತೌ ।
ತಯೋರ್ಜೀವಃ ಫಲಂ ಭುಂಕ್ತೇ ಕರ್ಮಣೋ ನ ಮಹೇಶ್ವರಃ  ॥ ೪೧ ॥
ಕೇವಲಂ ಸಾಕ್ಷಿರೂಪೇಣ ವಿನಾ ಭೋಗಂ ಮಹೇಶ್ವರಃ ।
ಪ್ರಕಾಶತೇ ಸ್ವಯಂ ಭೇದಃ ಕಲ್ಪಿತೋ ಮಾಯಯಾ ತಯೋಃ  ॥ ೪೨ ॥
ಘಟಾಕಾಶಮಠಾಕಾಶೌ ಯಥಾಕಾಶಪ್ರಭೇದತಃ ।
ಕಲ್ಪಿತೌ ಪರಮೌ ಜೀವಶಿವರೂಪೇಣ ಕಲ್ಪಿತೌ  ॥ ೪೩ ॥
ತತ್ತ್ವತಶ್ಚ ಶಿವಃ ಸಾಕ್ಷಾಚ್ಚಿಜ್ಜೀವಶ್ಚ ಸ್ವತಃ ಸದಾ ।
ಚಿಚ್ಚಿದಾಕಾರತೋ ಭಿನ್ನಾ ನ ಭಿನ್ನಾ ಚಿತ್ತ್ವಹಾನಿತಃ  ॥ ೪೪ ॥
ಚಿತಶ್ಚಿನ್ನ ಚಿದಾಕಾರದ್ಭಿದ್ಯತೇ ಜಡರೂಪತಃ ।
ಭಿದ್ಯತೇ ಚೇಜ್ಜಡೋ ಭೇದಶ್ಚಿದೇಕಾ ಸರ್ವದಾ ಖಲು  ॥ ೪೫ ॥
ತರ್ಕತಶ್ಚ ಪ್ರಮಾಣಾಚ್ಚ ಚಿದೇಕತ್ವವ್ಯವಸ್ಥಿತೇಃ ।
ಚಿದೇಕತ್ವಪರಿಜ್ಞಾನೇ ನ ಶೋಚತಿ ನ ಮುಹ್ಯತಿ  ॥ ೪೬ ॥
ಅದ್ವೈತಂ ಪರಮಾನಂದಂ ಶಿವಂ ಯಾತಿ ತು ಕೈವಲಂ।ಹ್  ॥ ೪೭ ॥
ಅಧಿಷ್ಠಾನಂ ಸಮಸ್ತಸ್ಯ ಜಗತಃ ಸತ್ಯಚಿದ್ಘನಂ।ಹ್ ।
ಅಹಮಸ್ಮೀತಿ ನಿಶ್ಚಿತ್ಯ ವೀತಶೋಕೋ ಭವೇನ್ಮುನಿಃ  ॥ ೪೮ ॥
ಸ್ವಶರೀರೇ ಸ್ವಯಂ ಜ್ಯೋತಿಃಸ್ವರೂಪಂ ಸರ್ವಸಾಕ್ಷಿಣಂ।ಹ್ ।
ಕ್ಷೀಣದೋಷಾಃ ಪ್ರಪಶ್ಯಂತಿ ನೇತರೇ ಮಾಯಯಾವೃತಾಃ  ॥ ೪೯ ॥
ಏವಂ ರೂಪಪರಿಜ್ಞಾನಂ ಯಸ್ಯಾಸ್ತಿ ಪರಯೋಗಿನಃ ।
ಕುತ್ರಚಿದ್ಗಮನಂ ನಾಸ್ತಿ ತಸ್ಯ ಪೂರ್ಣಸ್ವರೂಪಿಣಃ  ॥ ೫೦ ॥
ಆಕಾಶಮೇಕಂ ಸಂಪೂರ್ಣಂ ಕುತ್ರಚಿನ್ನೈವ ಗಚ್ಛತಿ ।
ತದ್ವತ್ಸ್ವಾತ್ಮಪರಿಜ್ಞಾನೀ ಕುತ್ರಚಿನ್ನೈವ ಗಚ್ಛತಿ  ॥ ೫೧ ॥
ಸ ಯೋ ಹ ವೈ ತತ್ಪರಮಂ ಬ್ರಹ್ಮ ಯೋ ವೇದ ವೈ ಮುನಿಃ ।
ಬ್ರಹ್ಮೈವ ಭವತಿ ಸ್ವಸ್ಥಃ ಸಚ್ಚಿದಾನಂದ ಮಾತೃಕಃ  ॥ ೫೨ ॥
ಇತ್ಯುಪನಿಷತ್।ಹ್  ॥
ಓಂ ಸಹ ನಾವವತು  ॥ ಸಹ ನೌ ಭುನಕ್ತು  ॥ ಸಹ ವೀರ್ಯಂ ಕರವಾವಹೈ  ॥
ತೇಜಸ್ವಿನಾವಧೀತಮಸ್ತು ಮಾ ವಿದ್ವಿಷಾವಹೈ  ॥
ಓಂ ಶಾಂತಿಃ ಶಾಂತಿಃ ಶಾಂತಿಃ  ॥
ಇತಿ ರುದ್ರಹೃದಯೋಪನಿಷತ್ಸಮಾಪ್ತಾ  ॥

॥ ಶಬ್ದಾರ್ಥ ಮತ್ತು ಭಾವಾರ್ಥ ॥

ಶಬ್ದಾರ್ಥ:
• ರುದ್ರಃ - ಶಿವನು;
• ನರಃ - ಪುರುಷ ರೂಪವು;
• ಉಮಾ - ಉಮಾದೇವಿಯು;
• ನಾರೀ - ಸ್ತ್ರೀ ರೂಪವು;
• ತಸ್ಮೈ - ಅವನಿಗೆ;
• ತಸ್ಯೈ - ಅವಳಿಗೆ;
• ನಮಃ ನಮಃ - ನಮಸ್ಕಾರಗಳು;
• ವಿಷ್ಣುಃ - ವಿಷ್ಣುವು;
• ಶಿವಮಯಃ - ಶಿವನ ಸ್ವರೂಪವೇ.

ಭಾವಾರ್ಥ:
ರುದ್ರಹೃದಯೋಪನಿಷತ್ತು ಕೃಷ್ಣ ಯಜುರ್ವೇದಕ್ಕೆ ಸೇರಿದ ಅತ್ಯಂತ ಸುಂದರ ಉಪನಿಷತ್ತಾಗಿದ್ದು, ಶಿವ ಮತ್ತು ವಿಷ್ಣು ತತ್ತ್ವಗಳ ಏಕತೆಯನ್ನು ಸಾರುತ್ತದೆ.

೧. ಹರಿ-ಹರ ಅಭೇದ ತತ್ತ್ವ:
ಶಿವನೇ ವಿಷ್ಣು, ವಿಷ್ಣುವೇ ಶಿವ. ಶಿವನನ್ನು ದ್ವೇಷಿಸುವವನು ವಿಷ್ಣುವನ್ನು ದ್ವೇಷಿಸಿದಂತೆ. ಅವರಿಬ್ಬರ ಹೃದಯಗಳು ಒಂದೇ ಚೈತನ್ಯದ ಎರಡು ಮುಖಗಳು.

೨. ರುದ್ರ ಮತ್ತು ಉಮಾ (ಶಿವ-ಶಕ್ತಿ):
ಜಗತ್ತಿನಲ್ಲಿರುವ ಸಮಸ್ತ ಪುರುಷ ವಾಚಕ ಶಕ್ತಿಗಳು ರುದ್ರನ ರೂಪಗಳು ಮತ್ತು ಸಮಸ್ತ ಸ್ತ್ರೀ ವಾಚಕ ಶಕ್ತಿಗಳು ಉಮಾದೇವಿಯ ರೂಪಗಳು. ಇಡೀ ಬ್ರಹ್ಮಾಂಡವು ಈ ಜಗತ್ಪಿತೃಗಳ ಮೈತ್ರಿಯಿಂದ ಕೂಡಿದೆ.
